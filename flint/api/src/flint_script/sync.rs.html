1//! ScriptSync — entity discovery, .rhai file loading, hot-reload
2//!
3//! Scans the ECS world for entities with a `script` component and loads
4//! the corresponding .rhai source files. Watches for file changes to
5//! support live hot-reload during play.
6
7use crate::engine::ScriptEngine;
8use flint_core::EntityId;
9use flint_ecs::FlintWorld;
10use std::collections::{HashMap, HashSet};
11use std::path::{Path, PathBuf};
12use std::time::SystemTime;
13
14/// Tracks which entities have scripts and manages file-based hot-reload
15pub struct ScriptSync {
16    /// Set of entity IDs that have been discovered and loaded
17    discovered: HashSet<EntityId>,
18    /// Source path → last modified time for hot-reload detection
19    file_timestamps: HashMap<PathBuf, SystemTime>,
20    /// Base scripts directory
21    scripts_dir: Option<PathBuf>,
22}
23
24impl ScriptSync {
25    pub fn new() -> Self {
26        Self {
27            discovered: HashSet::new(),
28            file_timestamps: HashMap::new(),
29            scripts_dir: None,
30        }
31    }
32
33    /// Set the scripts directory (called during initialization)
34    pub fn set_scripts_dir(&mut self, dir: PathBuf) {
35        self.scripts_dir = Some(dir);
36    }
37
38    /// Discover entities with `script` component and compile their scripts
39    pub fn discover_and_load(&mut self, world: &FlintWorld, engine: &mut ScriptEngine) {
40        let scripts_dir = match &self.scripts_dir {
41            Some(d) => d.clone(),
42            None => return,
43        };
44
45        for entity in world.all_entities() {
46            if self.discovered.contains(&entity.id) {
47                continue;
48            }
49
50            let script_comp = world.get_components(entity.id)
51                .and_then(|comps| comps.get("script").cloned());
52
53            let Some(script_data) = script_comp else { continue };
54
55            // Check enabled
56            let enabled = script_data.get("enabled")
57                .and_then(|v| v.as_bool())
58                .unwrap_or(true);
59            if !enabled {
60                self.discovered.insert(entity.id);
61                continue;
62            }
63
64            // Get source file path
65            let source = script_data.get("source")
66                .and_then(|v| v.as_str())
67                .unwrap_or("");
68            if source.is_empty() {
69                self.discovered.insert(entity.id);
70                continue;
71            }
72
73            let script_path = scripts_dir.join(source);
74            if !script_path.exists() {
75                eprintln!("[script] File not found: {}", script_path.display());
76                self.discovered.insert(entity.id);
77                continue;
78            }
79
80            match engine.compile_file(&script_path) {
81                Ok(ast) => {
82                    println!("[script] Loaded: {} → {}", entity.name, source);
83                    // Record file timestamp
84                    if let Ok(meta) = std::fs::metadata(&script_path) {
85                        if let Ok(modified) = meta.modified() {
86                            self.file_timestamps.insert(script_path.clone(), modified);
87                        }
88                    }
89                    engine.add_script(entity.id, ast, source.to_string());
90                }
91                Err(e) => {
92                    eprintln!("[script] Compile error in {}: {}", source, e);
93                }
94            }
95
96            self.discovered.insert(entity.id);
97        }
98    }
99
100    /// Check for modified script files and hot-reload them
101    pub fn check_hot_reload(&mut self, engine: &mut ScriptEngine) {
102        let scripts_dir = match &self.scripts_dir {
103            Some(d) => d.clone(),
104            None => return,
105        };
106
107        // Collect scripts that need reloading
108        let mut to_reload: Vec<(EntityId, PathBuf)> = Vec::new();
109
110        for (entity_id, script) in &engine.scripts {
111            let script_path = scripts_dir.join(&script.source_path);
112
113            let current_modified = match std::fs::metadata(&script_path) {
114                Ok(meta) => meta.modified().ok(),
115                Err(_) => continue,
116            };
117
118            let Some(current) = current_modified else { continue };
119            let last = self.file_timestamps.get(&script_path);
120
121            if last.is_none_or(|last| current > *last) {
122                to_reload.push((*entity_id, script_path));
123            }
124        }
125
126        // Reload changed scripts
127        for (entity_id, script_path) in to_reload {
128            match engine.compile_file(&script_path) {
129                Ok(ast) => {
130                    if let Some(script) = engine.scripts.get_mut(&entity_id) {
131                        println!("[script] Hot-reloaded: {}", script.source_path);
132                        script.hot_reload(ast);
133                    }
134                    if let Ok(meta) = std::fs::metadata(&script_path) {
135                        if let Ok(modified) = meta.modified() {
136                            self.file_timestamps.insert(script_path, modified);
137                        }
138                    }
139                }
140                Err(e) => {
141                    // Keep old AST on compile error
142                    eprintln!("[script] Hot-reload compile error: {}", e);
143                }
144            }
145        }
146    }
147}

155/// Load scripts from the `scripts/` directory next to the scene file.
156/// Also checks one level up (e.g. game root) for projects that use a `scenes/` subdirectory.
157pub fn load_scripts_from_scene(scene_path: &str, sync: &mut ScriptSync) {
158    let scene_dir = Path::new(scene_path)
159        .parent()
160        .unwrap_or_else(|| Path::new("."));
161
162    let scripts_dir = scene_dir.join("scripts");
163    if scripts_dir.is_dir() {
164        sync.set_scripts_dir(scripts_dir);
165        return;
166    }
167
168    // Check parent directory (game project structure: scenes/ and scripts/ are siblings)
169    if let Some(parent) = scene_dir.parent() {
170        let scripts_dir = parent.join("scripts");
171        if scripts_dir.is_dir() {
172            sync.set_scripts_dir(scripts_dir);
173        }
174    }
