1//! Script execution context and command types
2//!
3//! ScriptCallContext is the shared state accessed by Rhai API functions during script execution.
4//! ScriptCommand represents deferred actions (audio, events, logging) collected during a script call.

11/// Snapshot of input state for script access (no winit dependency needed)
12#[derive(Clone, Default)]
13pub struct InputSnapshot {
14    pub actions_pressed: HashSet<String>,
15    pub actions_just_pressed: HashSet<String>,
16    pub mouse_delta: (f64, f64),
17}

19/// Log severity levels
20#[derive(Debug, Clone)]
21pub enum LogLevel {
22    Info,
23    Warn,
24    Error,
25}

27/// Deferred commands produced by scripts, processed by PlayerApp after script update
28#[derive(Debug, Clone)]
29pub enum ScriptCommand {
30    PlaySound { name: String, volume: f64 },
31    PlaySoundAt { name: String, position: (f64, f64, f64), volume: f64 },
32    StopSound { name: String },
33    FireEvent { name: String, data: toml::Value },
34    Log { level: LogLevel, message: String },
35}
36
37/// 2D draw command issued by scripts each frame (immediate mode)
38#[derive(Debug, Clone)]
39pub enum DrawCommand {
40    Text {
41        x: f32, y: f32,
42        text: String,
43        size: f32,
44        color: [f32; 4],
45        layer: i32,
46    },
47    RectFilled {
48        x: f32, y: f32, w: f32, h: f32,
49        color: [f32; 4],
50        rounding: f32,
51        layer: i32,
52    },
53    RectOutline {
54        x: f32, y: f32, w: f32, h: f32,
55        color: [f32; 4],
56        thickness: f32,
57        layer: i32,
58    },
59    CircleFilled {
60        x: f32, y: f32,
61        radius: f32,
62        color: [f32; 4],
63        layer: i32,
64    },
65    CircleOutline {
66        x: f32, y: f32,
67        radius: f32,
68        color: [f32; 4],
69        thickness: f32,
70        layer: i32,
71    },
72    Line {
73        x1: f32, y1: f32, x2: f32, y2: f32,
74        color: [f32; 4],
75        thickness: f32,
76        layer: i32,
77    },
78    Sprite {
79        x: f32, y: f32, w: f32, h: f32,
80        name: String,
81        uv: [f32; 4],
82        tint: [f32; 4],
83        layer: i32,
84    },
85}
86
87impl DrawCommand {
88    pub fn layer(&self) -> i32 {
89        match self {
90            DrawCommand::Text { layer, .. } => *layer,
91            DrawCommand::RectFilled { layer, .. } => *layer,
92            DrawCommand::RectOutline { layer, .. } => *layer,
93            DrawCommand::CircleFilled { layer, .. } => *layer,
94            DrawCommand::CircleOutline { layer, .. } => *layer,
95            DrawCommand::Line { layer, .. } => *layer,
96            DrawCommand::Sprite { layer, .. } => *layer,
97        }
98    }
99}

101/// Shared context set before each script call and read by registered Rhai functions.
102///
103/// Safety: the `world` pointer is only valid during the scope of `call_update` /
104/// `process_events`. It is set to null immediately after each call batch.
105pub struct ScriptCallContext {
106    /// Raw pointer to the FlintWorld — valid only during call scope
107    pub world: *mut FlintWorld,
108    /// Raw pointer to the PhysicsSystem — valid only during call scope
109    pub physics: *const PhysicsSystem,
110    /// Camera position and direction for weapon aiming
111    pub camera_position: [f32; 3],
112    pub camera_direction: [f32; 3],
113    /// Entity currently being scripted
114    pub current_entity: EntityId,
115    /// Accumulated commands to be drained after all scripts run
116    pub commands: Vec<ScriptCommand>,
117    /// Accumulated 2D draw commands for the current frame
118    pub draw_commands: Vec<DrawCommand>,
119    /// Input snapshot for this frame
120    pub input: InputSnapshot,
121    /// Frame delta time
122    pub delta_time: f64,
123    /// Total elapsed game time
124    pub total_time: f64,
125    /// Screen dimensions in pixels (set before scripts run)
126    pub screen_width: f32,
127    pub screen_height: f32,
128}
129
130// SAFETY: ScriptCallContext is only accessed from the main thread within
131// controlled call scopes. The world pointer is valid only during those scopes.
132unsafe impl Send for ScriptCallContext {}
133unsafe impl Sync for ScriptCallContext {}
134
135impl Default for ScriptCallContext {
136    fn default() -> Self {
137        Self::new()
138    }
139}
140
141impl ScriptCallContext {
142    pub fn new() -> Self {
143        Self {
144            world: std::ptr::null_mut(),
145            physics: std::ptr::null(),
146            camera_position: [0.0; 3],
147            camera_direction: [0.0, 0.0, 1.0],
148            current_entity: EntityId::from_raw(0),
149            commands: Vec::new(),
150            draw_commands: Vec::new(),
151            input: InputSnapshot::default(),
152            delta_time: 0.0,
153            total_time: 0.0,
154            screen_width: 1280.0,
155            screen_height: 720.0,
156        }
157    }
158
159    /// Get a reference to the world. Panics if called outside a valid scope.
160    ///
161    /// # Safety
162    /// Caller must ensure the world pointer was set and is still valid
163    /// (i.e., called within the scope of `call_update` or `process_events`).
164    pub unsafe fn world_ref(&self) -> &FlintWorld {
165        assert!(!self.world.is_null(), "ScriptCallContext: world pointer is null (called outside scope)");
166        unsafe { &*self.world }
167    }
168
169    /// Get a mutable reference to the world. Panics if called outside a valid scope.
170    ///
171    /// # Safety
172    /// Caller must ensure the world pointer was set, is still valid, and no other
173    /// references to the world exist (i.e., called within the scope of
174    /// `call_update` or `process_events`).
175    #[allow(clippy::mut_from_ref)]
176    pub unsafe fn world_mut(&self) -> &mut FlintWorld {
177        assert!(!self.world.is_null(), "ScriptCallContext: world pointer is null (called outside scope)");
178        unsafe { &mut *self.world }
179    }
180
181    /// Get a reference to the physics system. Returns None if not set.
182    ///
183    /// # Safety
184    /// Caller must ensure the physics pointer was set and is still valid.
185    pub unsafe fn physics_ref(&self) -> Option<&PhysicsSystem> {
186        if self.physics.is_null() {
187            None
188        } else {
189            Some(unsafe { &*self.physics })
190        }
191    }
192}
