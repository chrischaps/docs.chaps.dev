1//! Script execution context and command types
2//!
3//! ScriptCallContext is the shared state accessed by Rhai API functions during script execution.
4//! ScriptCommand represents deferred actions (audio, events, logging) collected during a script call.
5
6use flint_core::EntityId;
7use flint_ecs::FlintWorld;
8use std::collections::HashSet;
9
10/// Snapshot of input state for script access (no winit dependency needed)
11#[derive(Clone, Default)]
12pub struct InputSnapshot {
13    pub actions_pressed: HashSet<String>,
14    pub actions_just_pressed: HashSet<String>,
15    pub mouse_delta: (f64, f64),
16}
17
18/// Log severity levels
19#[derive(Debug, Clone)]
20pub enum LogLevel {
21    Info,
22    Warn,
23    Error,
24}
25
26/// Deferred commands produced by scripts, processed by PlayerApp after script update
27#[derive(Debug, Clone)]
28pub enum ScriptCommand {
29    PlaySound { name: String, volume: f64 },
30    PlaySoundAt { name: String, position: (f64, f64, f64), volume: f64 },
31    StopSound { name: String },
32    FireEvent { name: String, data: toml::Value },
33    Log { level: LogLevel, message: String },
34}
35
36/// Shared context set before each script call and read by registered Rhai functions.
37///
38/// Safety: the `world` pointer is only valid during the scope of `call_update` /
39/// `process_events`. It is set to null immediately after each call batch.
40pub struct ScriptCallContext {
41    /// Raw pointer to the FlintWorld â€” valid only during call scope
42    pub world: *mut FlintWorld,
43    /// Entity currently being scripted
44    pub current_entity: EntityId,
45    /// Accumulated commands to be drained after all scripts run
46    pub commands: Vec<ScriptCommand>,
47    /// Input snapshot for this frame
48    pub input: InputSnapshot,
49    /// Frame delta time
50    pub delta_time: f64,
51    /// Total elapsed game time
52    pub total_time: f64,
53}
54
55// SAFETY: ScriptCallContext is only accessed from the main thread within
56// controlled call scopes. The world pointer is valid only during those scopes.
57unsafe impl Send for ScriptCallContext {}
58unsafe impl Sync for ScriptCallContext {}
59
60impl Default for ScriptCallContext {
61    fn default() -> Self {
62        Self::new()
63    }
64}
65
66impl ScriptCallContext {
67    pub fn new() -> Self {
68        Self {
69            world: std::ptr::null_mut(),
70            current_entity: EntityId::from_raw(0),
71            commands: Vec::new(),
72            input: InputSnapshot::default(),
73            delta_time: 0.0,
74            total_time: 0.0,
75        }
76    }
77
78    /// Get a reference to the world. Panics if called outside a valid scope.
79    ///
80    /// # Safety
81    /// Caller must ensure the world pointer was set and is still valid
82    /// (i.e., called within the scope of `call_update` or `process_events`).
83    pub unsafe fn world_ref(&self) -> &FlintWorld {
84        assert!(!self.world.is_null(), "ScriptCallContext: world pointer is null (called outside scope)");
85        unsafe { &*self.world }
86    }
87
88    /// Get a mutable reference to the world. Panics if called outside a valid scope.
89    ///
90    /// # Safety
91    /// Caller must ensure the world pointer was set, is still valid, and no other
92    /// references to the world exist (i.e., called within the scope of
93    /// `call_update` or `process_events`).
94    #[allow(clippy::mut_from_ref)]
95    pub unsafe fn world_mut(&self) -> &mut FlintWorld {
96        assert!(!self.world.is_null(), "ScriptCallContext: world pointer is null (called outside scope)");
97        unsafe { &mut *self.world }
98    }
99}
