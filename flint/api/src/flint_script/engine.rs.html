1//! ScriptEngine — Rhai engine setup, AST storage, call helpers
2//!
3//! Wraps rhai::Engine with all API functions registered. Manages compiled ASTs
4//! and per-entity Scopes. Provides the call_update / process_events interface
5//! that temporarily lends the FlintWorld to scripts.

16/// Per-entity script instance
17pub struct ScriptInstance {
18    pub ast: AST,
19    pub scope: Scope<'static>,
20    pub source_path: String,
21    pub has_on_init: bool,
22    pub has_on_update: bool,
23    pub has_on_collision: bool,
24    pub has_on_trigger_enter: bool,
25    pub has_on_trigger_exit: bool,
26    pub has_on_action: bool,
27    pub has_on_interact: bool,
28    pub init_called: bool,
29}

31impl ScriptInstance {
32    pub fn new(ast: AST, source_path: String) -> Self {
33        let has_on_init = has_function(&ast, "on_init");
34        let has_on_update = has_function(&ast, "on_update");
35        let has_on_collision = has_function(&ast, "on_collision");
36        let has_on_trigger_enter = has_function(&ast, "on_trigger_enter");
37        let has_on_trigger_exit = has_function(&ast, "on_trigger_exit");
38        let has_on_action = has_function(&ast, "on_action");
39        let has_on_interact = has_function(&ast, "on_interact");
40
41        Self {
42            ast,
43            scope: Scope::new(),
44            source_path,
45            has_on_init,
46            has_on_update,
47            has_on_collision,
48            has_on_trigger_enter,
49            has_on_trigger_exit,
50            has_on_action,
51            has_on_interact,
52            init_called: false,
53        }
54    }

56    /// Recompile with a new AST but preserve the scope (persistent state)
57    pub fn hot_reload(&mut self, ast: AST) {
58        self.has_on_init = has_function(&ast, "on_init");
59        self.has_on_update = has_function(&ast, "on_update");
60        self.has_on_collision = has_function(&ast, "on_collision");
61        self.has_on_trigger_enter = has_function(&ast, "on_trigger_enter");
62        self.has_on_trigger_exit = has_function(&ast, "on_trigger_exit");
63        self.has_on_action = has_function(&ast, "on_action");
64        self.has_on_interact = has_function(&ast, "on_interact");
65        self.ast = ast;
66        // Don't reset init_called — hot-reload preserves state
67    }
68}

70/// Check if an AST contains a function definition with the given name
71fn has_function(ast: &AST, name: &str) -> bool {
72    ast.iter_functions().any(|f| f.name == name)
73}

75/// The scripting engine — owns the Rhai Engine and per-entity script instances
76pub struct ScriptEngine {
77    engine: Engine,
78    pub ctx: Arc<Mutex<ScriptCallContext>>,
79    pub scripts: HashMap<EntityId, ScriptInstance>,
80}

82impl ScriptEngine {
83    pub fn new() -> Self {
84        let ctx = Arc::new(Mutex::new(ScriptCallContext::new()));
85        let mut engine = Engine::new();
86
87        // Register all API functions
88        api::register_all(&mut engine, ctx.clone());
89
90        Self {
91            engine,
92            ctx,
93            scripts: HashMap::new(),
94        }
95    }
96
97    /// Compile a Rhai source file into an AST
98    pub fn compile(&self, source: &str) -> Result<AST, String> {
99        self.engine.compile(source).map_err(|e| format!("{}", e))
100    }
101
102    /// Compile from a file path
103    pub fn compile_file(&self, path: &std::path::Path) -> Result<AST, String> {
104        let source = std::fs::read_to_string(path)
105            .map_err(|e| format!("Failed to read {}: {}", path.display(), e))?;
106        self.compile(&source)
107    }
108
109    /// Add a compiled script for an entity
110    pub fn add_script(&mut self, entity: EntityId, ast: AST, source_path: String) {
111        let mut instance = ScriptInstance::new(ast, source_path);
112        // Evaluate top-level statements to populate scope with module-level variables.
113        // Without this, `let x = 0.0;` at the top of a script never enters the Scope,
114        // so `call_fn` can't find it from within `on_init`/`on_update`.
115        if let Err(e) = self.engine.run_ast_with_scope(&mut instance.scope, &instance.ast) {
116            eprintln!("[script] module init error ({}): {}", instance.source_path, e);
117        }
118        self.scripts.insert(entity, instance);
119    }
120
121    /// Provide context for the current frame
122    pub fn provide_context(&self, input: InputSnapshot, delta_time: f64, total_time: f64) {
123        let mut c = self.ctx.lock().unwrap();
124        c.input = input;
125        c.delta_time = delta_time;
126        c.total_time = total_time;
127    }
128
129    /// Call on_init() for all scripts that haven't been initialized yet
130    pub fn call_inits(&mut self, world: &mut FlintWorld) {
131        // Set world pointer for the duration of this call
132        {
133            let mut c = self.ctx.lock().unwrap();
134            c.world = world as *mut FlintWorld;
135        }
136
137        let entity_ids: Vec<EntityId> = self.scripts.keys().copied().collect();
138        for entity_id in entity_ids {
139            let script = self.scripts.get_mut(&entity_id).unwrap();
140            if script.has_on_init && !script.init_called {
141                {
142                    let mut c = self.ctx.lock().unwrap();
143                    c.current_entity = entity_id;
144                }
145                script.init_called = true;
146                if let Err(e) = self.engine.call_fn::<()>(&mut script.scope, &script.ast, "on_init", ()) {
147                    eprintln!("[script] on_init error ({}): {}", script.source_path, e);
148                }
149            }
150        }
151
152        // Clear world pointer
153        {
154            let mut c = self.ctx.lock().unwrap();
155            c.world = std::ptr::null_mut();
156        }
157    }
158
159    /// Call on_update(dt) for all scripts
160    pub fn call_updates(&mut self, world: &mut FlintWorld, dt: f64) {
161        {
162            let mut c = self.ctx.lock().unwrap();
163            c.world = world as *mut FlintWorld;
164        }
165
166        let entity_ids: Vec<EntityId> = self.scripts.keys().copied().collect();
167        for entity_id in entity_ids {
168            let script = self.scripts.get_mut(&entity_id).unwrap();
169            if script.has_on_update {
170                {
171                    let mut c = self.ctx.lock().unwrap();
172                    c.current_entity = entity_id;
173                }
174                if let Err(e) = self.engine.call_fn::<()>(&mut script.scope, &script.ast, "on_update", (dt,)) {
175                    eprintln!("[script] on_update error ({}): {}", script.source_path, e);
176                }
177            }
178        }
179
180        {
181            let mut c = self.ctx.lock().unwrap();
182            c.world = std::ptr::null_mut();
183        }
184    }

186    /// Route game events to appropriate script callbacks
187    pub fn process_events(&mut self, events: &[GameEvent], world: &mut FlintWorld) {
188        if events.is_empty() {
189            return;
190        }
191
192        {
193            let mut c = self.ctx.lock().unwrap();
194            c.world = world as *mut FlintWorld;
195        }
196
197        for event in events {
198            match event {
199                GameEvent::CollisionStarted { entity_a, entity_b } => {
200                    self.call_collision(*entity_a, *entity_b);
201                    self.call_collision(*entity_b, *entity_a);
202                }
203                GameEvent::TriggerEntered { entity, trigger } => {
204                    self.call_trigger_enter(*trigger, *entity);
205                }
206                GameEvent::TriggerExited { entity, trigger } => {
207                    self.call_trigger_exit(*trigger, *entity);
208                }
209                GameEvent::ActionPressed(action) => {
210                    self.call_action_on_all(action, world);
211                }
212                _ => {}
213            }
214        }
215
216        {
217            let mut c = self.ctx.lock().unwrap();
218            c.world = std::ptr::null_mut();
219        }
220    }
221
222    fn call_collision(&mut self, entity: EntityId, other: EntityId) {
223        if let Some(script) = self.scripts.get_mut(&entity) {
224            if script.has_on_collision {
225                {
226                    let mut c = self.ctx.lock().unwrap();
227                    c.current_entity = entity;
228                }
229                let other_id = other.raw() as i64;
230                if let Err(e) = self.engine.call_fn::<()>(&mut script.scope, &script.ast, "on_collision", (other_id,)) {
231                    eprintln!("[script] on_collision error ({}): {}", script.source_path, e);
232                }
233            }
234        }
235    }
236
237    fn call_trigger_enter(&mut self, trigger: EntityId, entity: EntityId) {
238        if let Some(script) = self.scripts.get_mut(&trigger) {
239            if script.has_on_trigger_enter {
240                {
241                    let mut c = self.ctx.lock().unwrap();
242                    c.current_entity = trigger;
243                }
244                let entity_id = entity.raw() as i64;
245                if let Err(e) = self.engine.call_fn::<()>(&mut script.scope, &script.ast, "on_trigger_enter", (entity_id,)) {
246                    eprintln!("[script] on_trigger_enter error ({}): {}", script.source_path, e);
247                }
248            }
249        }
250    }

255                {
256                    let mut c = self.ctx.lock().unwrap();

261                    eprintln!("[script] on_trigger_exit error ({}): {}", script.source_path, e);
262                }
263            }
264        }

267    fn call_action_on_all(&mut self, action: &str, world: &FlintWorld) {
268        let entity_ids: Vec<EntityId> = self.scripts.keys().copied().collect();
269        for entity_id in entity_ids {
270            let script = self.scripts.get_mut(&entity_id).unwrap();
271
272            // on_action callback
273            if script.has_on_action {
274                {
275                    let mut c = self.ctx.lock().unwrap();
276                    c.current_entity = entity_id;
277                }
278                let action_str = action.to_string();
279                if let Err(e) = self.engine.call_fn::<()>(&mut script.scope, &script.ast, "on_action", (action_str,)) {
280                    eprintln!("[script] on_action error ({}): {}", script.source_path, e);
281                }
282            }
283
284            // on_interact sugar: ActionPressed("interact") + proximity + interactable check
285            if script.has_on_interact && action == "interact" {
286                // Read range from interactable component, fall back to 3.0
287                let (range, enabled) = get_interactable_config(entity_id, world);
288                if !enabled {
289                    continue;
290                }
291                let close_enough = is_near_player(entity_id, world, range);
292                if close_enough {
293                    {
294                        let mut c = self.ctx.lock().unwrap();
295                        c.current_entity = entity_id;
296                    }
297                    if let Err(e) = self.engine.call_fn::<()>(&mut script.scope, &script.ast, "on_interact", ()) {
298                        eprintln!("[script] on_interact error ({}): {}", script.source_path, e);
299                    }
300                }
301            }
302        }
303    }

306    pub fn drain_commands(&self) -> Vec<ScriptCommand> {
307        let mut c = self.ctx.lock().unwrap();
308        std::mem::take(&mut c.commands)
309    }

318/// Read interactable range and enabled state from entity components.
319/// Returns (range, enabled) — defaults to (3.0, true) if no interactable component.
320fn get_interactable_config(entity: EntityId, world: &FlintWorld) -> (f64, bool) {
321    let Some(comps) = world.get_components(entity) else {
322        return (3.0, true);
323    };
324    let Some(interactable) = comps.get("interactable") else {
325        return (3.0, true);
326    };
327    let range = interactable
328        .get("range")
329        .and_then(|v| v.as_float().or_else(|| v.as_integer().map(|i| i as f64)))
330        .unwrap_or(3.0);
331    let enabled = interactable
332        .get("enabled")
333        .and_then(|v| v.as_bool())
334        .unwrap_or(true);
335    (range, enabled)
336}

338/// Information about the nearest interactable entity
339pub struct NearestInteractable {
340    pub entity_id: EntityId,
341    pub prompt_text: String,
342    pub interaction_type: String,
343    pub distance: f64,
344}
345
346/// Find the nearest in-range interactable entity to the player.
347/// Checks all entities with an `interactable` component that are enabled and within range.
348pub fn find_nearest_interactable(world: &FlintWorld) -> Option<NearestInteractable> {
349    // Find player entity
350    let player_id = world.all_entities().iter()
351        .find(|e| {
352            world.get_components(e.id)
353                .map(|c| c.has("character_controller"))
354                .unwrap_or(false)
355        })
356        .map(|e| e.id)?;
357    let pt = world.get_transform(player_id)?;
358
359    let mut best: Option<NearestInteractable> = None;
360
361    for entity in world.all_entities() {
362        let Some(comps) = world.get_components(entity.id) else { continue; };
363        let Some(interactable) = comps.get("interactable") else { continue; };
364
365        let enabled = interactable.get("enabled").and_then(|v| v.as_bool()).unwrap_or(true);
366        if !enabled {
367            continue;
368        }
369
370        let range = interactable
371            .get("range")
372            .and_then(|v| v.as_float().or_else(|| v.as_integer().map(|i| i as f64)))
373            .unwrap_or(3.0);
374
375        let Some(et) = world.get_transform(entity.id) else { continue; };
376        let dx = (pt.position.x - et.position.x) as f64;
377        let dy = (pt.position.y - et.position.y) as f64;
378        let dz = (pt.position.z - et.position.z) as f64;
379        let dist = (dx * dx + dy * dy + dz * dz).sqrt();
380
381        if dist <= range {
382            let is_closer = best.as_ref().is_none_or(|b| dist < b.distance);
383            if is_closer {
384                let prompt_text = interactable
385                    .get("prompt_text")
386                    .and_then(|v| v.as_str())
387                    .unwrap_or("Interact")
388                    .to_string();
389                let interaction_type = interactable
390                    .get("interaction_type")
391                    .and_then(|v| v.as_str())
392                    .unwrap_or("use")
393                    .to_string();
394                best = Some(NearestInteractable {
395                    entity_id: entity.id,
396                    prompt_text,
397                    interaction_type,
398                    distance: dist,
399                });
400            }
401        }
402    }
403
404    best
405}

407/// Check if an entity is within `range` distance of the player entity
408fn is_near_player(entity: EntityId, world: &FlintWorld, range: f64) -> bool {
409    // Find player entity (entity with character_controller component)
410    let player_id = world.all_entities().iter()
411        .find(|e| {
412            world.get_components(e.id)
413                .map(|c| c.has("character_controller"))
414                .unwrap_or(false)
415        })
416        .map(|e| e.id);
417
418    let Some(player) = player_id else { return false; };
419    let Some(pt) = world.get_transform(player) else { return false; };
420    let Some(et) = world.get_transform(entity) else { return false; };
421
422    let dx = (pt.position.x - et.position.x) as f64;
423    let dy = (pt.position.y - et.position.y) as f64;
424    let dz = (pt.position.z - et.position.z) as f64;
425    (dx * dx + dy * dy + dz * dz).sqrt() <= range
426}

764        // from both on_init and on_update via Scope population.

771            m.insert("value".into(), toml::Value::Float(0.0));
772            m
773        })).unwrap();
774
775        let ast = engine.compile(r#"

777
778            fn on_init() {
779                // Modify the module-level variable
780                counter = 42.0;
781            }
782
783            fn on_update(dt) {
784                // Read the value set in on_init
785                let me = self_entity();
786                set_field(me, "state", "value", counter);

788        "#).unwrap();

790        engine.add_script(id, ast, "test.rhai".into());

794        let value = world.get_components(id).unwrap()
795            .get_field("state", "value").unwrap()
796            .as_float().unwrap();
