1//! Rhai API function registration
2//!
3//! All functions accessible from Rhai scripts are registered here.
4//! They access the world through the shared ScriptCallContext.
5
6use crate::context::{DrawCommand, LogLevel, ScriptCallContext, ScriptCommand};
7use flint_core::EntityId;
8use rhai::{Dynamic, Engine, Map};
9use std::sync::{Arc, Mutex};
10
11/// Register all API functions on the Rhai engine
12pub fn register_all(engine: &mut Engine, ctx: Arc<Mutex<ScriptCallContext>>) {
13    register_entity_api(engine, ctx.clone());
14    register_input_api(engine, ctx.clone());
15    register_time_api(engine, ctx.clone());
16    register_audio_api(engine, ctx.clone());
17    register_animation_api(engine, ctx.clone());
18    register_physics_api(engine, ctx.clone());
19    register_math_api(engine);
20    register_event_api(engine, ctx.clone());
21    register_ui_api(engine, ctx.clone());
22    register_log_api(engine, ctx);
23}

27fn register_entity_api(engine: &mut Engine, ctx: Arc<Mutex<ScriptCallContext>>) {
28    // self_entity() -> i64
29    {
30        let ctx = ctx.clone();
31        engine.register_fn("self_entity", move || -> i64 {
32            let c = ctx.lock().unwrap();
33            c.current_entity.raw() as i64
34        });
35    }
36
37    // this_entity() -> i64 (alias for self_entity)
38    {
39        let ctx = ctx.clone();
40        engine.register_fn("this_entity", move || -> i64 {
41            let c = ctx.lock().unwrap();
42            c.current_entity.raw() as i64
43        });
44    }
45
46    // get_entity(name: &str) -> i64
47    {
48        let ctx = ctx.clone();
49        engine.register_fn("get_entity", move |name: &str| -> i64 {
50            let c = ctx.lock().unwrap();
51            let world = unsafe { c.world_ref() };
52            world.get_id(name).map(|id| id.raw() as i64).unwrap_or(-1)
53        });
54    }
55
56    // entity_exists(id: i64) -> bool
57    {
58        let ctx = ctx.clone();
59        engine.register_fn("entity_exists", move |id: i64| -> bool {
60            if id < 0 { return false; }
61            let c = ctx.lock().unwrap();
62            let world = unsafe { c.world_ref() };
63            world.contains(EntityId::from_raw(id as u64))
64        });
65    }
66
67    // entity_name(id: i64) -> String
68    {
69        let ctx = ctx.clone();
70        engine.register_fn("entity_name", move |id: i64| -> String {
71            if id < 0 { return String::new(); }
72            let c = ctx.lock().unwrap();
73            let world = unsafe { c.world_ref() };
74            world.get_name(EntityId::from_raw(id as u64))
75                .unwrap_or("")
76                .to_string()
77        });
78    }
79
80    // has_component(id: i64, comp: &str) -> bool
81    {
82        let ctx = ctx.clone();
83        engine.register_fn("has_component", move |id: i64, comp: &str| -> bool {
84            if id < 0 { return false; }
85            let c = ctx.lock().unwrap();
86            let world = unsafe { c.world_ref() };
87            world.get_components(EntityId::from_raw(id as u64))
88                .map(|comps| comps.has(comp))
89                .unwrap_or(false)
90        });
91    }
92
93    // get_component(id: i64, comp: &str) -> Dynamic (returns full component as Map, or () if missing)
94    {
95        let ctx = ctx.clone();
96        engine.register_fn("get_component", move |id: i64, comp: &str| -> Dynamic {
97            if id < 0 { return Dynamic::UNIT; }
98            let c = ctx.lock().unwrap();
99            let world = unsafe { c.world_ref() };
100            match world.get_components(EntityId::from_raw(id as u64)) {
101                Some(comps) => {
102                    match comps.get(comp) {
103                        Some(val) => toml_to_dynamic(val.clone()),
104                        None => Dynamic::UNIT,
105                    }
106                }
107                None => Dynamic::UNIT,
108            }
109        });
110    }
111
112    // get_field(id: i64, comp: &str, field: &str) -> Dynamic
113    {
114        let ctx = ctx.clone();
115        engine.register_fn("get_field", move |id: i64, comp: &str, field: &str| -> Dynamic {
116            if id < 0 { return Dynamic::UNIT; }
117            let c = ctx.lock().unwrap();
118            let world = unsafe { c.world_ref() };
119            world.get_components(EntityId::from_raw(id as u64))
120                .and_then(|comps| comps.get_field(comp, field).cloned())
121                .map(toml_to_dynamic)
122                .unwrap_or(Dynamic::UNIT)
123        });
124    }
125
126    // set_field(id: i64, comp: &str, field: &str, val: Dynamic)
127    {
128        let ctx = ctx.clone();
129        engine.register_fn("set_field", move |id: i64, comp: &str, field: &str, val: Dynamic| {
130            if id < 0 { return; }
131            let c = ctx.lock().unwrap();
132            let world = unsafe { c.world_mut() };
133            if let Some(tv) = dynamic_to_toml(&val) {
134                if let Some(comps) = world.get_components_mut(EntityId::from_raw(id as u64)) {
135                    comps.set_field(comp, field, tv);
136                }
137            }
138        });
139    }
140
141    // get_position(id: i64) -> Map #{x, y, z}
142    {
143        let ctx = ctx.clone();
144        engine.register_fn("get_position", move |id: i64| -> Map {
145            let mut map = Map::new();
146            if id < 0 { return map; }
147            let c = ctx.lock().unwrap();
148            let world = unsafe { c.world_ref() };
149            if let Some(transform) = world.get_transform(EntityId::from_raw(id as u64)) {
150                map.insert("x".into(), Dynamic::from(transform.position.x as f64));
151                map.insert("y".into(), Dynamic::from(transform.position.y as f64));
152                map.insert("z".into(), Dynamic::from(transform.position.z as f64));
153            }
154            map
155        });
156    }
157
158    // set_position(id: i64, x: f64, y: f64, z: f64)
159    {
160        let ctx = ctx.clone();
161        engine.register_fn("set_position", move |id: i64, x: f64, y: f64, z: f64| {
162            if id < 0 { return; }
163            let c = ctx.lock().unwrap();
164            let world = unsafe { c.world_mut() };
165            let eid = EntityId::from_raw(id as u64);
166            if let Some(comps) = world.get_components_mut(eid) {
167                comps.set_field("transform", "position", toml::Value::Array(vec![
168                    toml::Value::Float(x),
169                    toml::Value::Float(y),
170                    toml::Value::Float(z),
171                ]));
172            }
173        });
174    }
175
176    // get_rotation(id: i64) -> Map #{x, y, z}
177    {
178        let ctx = ctx.clone();
179        engine.register_fn("get_rotation", move |id: i64| -> Map {
180            let mut map = Map::new();
181            if id < 0 { return map; }
182            let c = ctx.lock().unwrap();
183            let world = unsafe { c.world_ref() };
184            if let Some(transform) = world.get_transform(EntityId::from_raw(id as u64)) {
185                map.insert("x".into(), Dynamic::from(transform.rotation.x as f64));
186                map.insert("y".into(), Dynamic::from(transform.rotation.y as f64));
187                map.insert("z".into(), Dynamic::from(transform.rotation.z as f64));
188            }
189            map
190        });
191    }
192
193    // set_rotation(id: i64, x: f64, y: f64, z: f64)
194    {
195        let ctx = ctx.clone();
196        engine.register_fn("set_rotation", move |id: i64, x: f64, y: f64, z: f64| {
197            if id < 0 { return; }
198            let c = ctx.lock().unwrap();
199            let world = unsafe { c.world_mut() };
200            let eid = EntityId::from_raw(id as u64);
201            if let Some(comps) = world.get_components_mut(eid) {
202                comps.set_field("transform", "rotation", toml::Value::Array(vec![
203                    toml::Value::Float(x),
204                    toml::Value::Float(y),
205                    toml::Value::Float(z),
206                ]));
207            }
208        });
209    }
210
211    // distance(a: i64, b: i64) -> f64
212    {
213        let ctx = ctx.clone();
214        engine.register_fn("distance", move |a: i64, b: i64| -> f64 {
215            if a < 0 || b < 0 { return f64::MAX; }
216            let c = ctx.lock().unwrap();
217            let world = unsafe { c.world_ref() };
218            let ta = world.get_transform(EntityId::from_raw(a as u64));
219            let tb = world.get_transform(EntityId::from_raw(b as u64));
220            match (ta, tb) {
221                (Some(a), Some(b)) => {
222                    let dx = (a.position.x - b.position.x) as f64;
223                    let dy = (a.position.y - b.position.y) as f64;
224                    let dz = (a.position.z - b.position.z) as f64;
225                    (dx * dx + dy * dy + dz * dz).sqrt()
226                }
227                _ => f64::MAX,
228            }
229        });
230    }
231
232    // spawn_entity(name: &str) -> i64
233    {
234        let ctx = ctx.clone();
235        engine.register_fn("spawn_entity", move |name: &str| -> i64 {
236            let c = ctx.lock().unwrap();
237            let world = unsafe { c.world_mut() };
238            world.spawn(name).map(|id| id.raw() as i64).unwrap_or(-1)
239        });
240    }
241
242    // despawn_entity(id: i64)
243    {
244        let ctx = ctx.clone();
245        engine.register_fn("despawn_entity", move |id: i64| {
246            if id < 0 { return; }
247            let c = ctx.lock().unwrap();
248            let world = unsafe { c.world_mut() };
249            let _ = world.despawn(EntityId::from_raw(id as u64));
250        });
251    }
252}

429fn register_physics_api(engine: &mut Engine, ctx: Arc<Mutex<ScriptCallContext>>) {
430    // raycast(ox, oy, oz, dx, dy, dz, max_dist) -> Map or ()
431    {
432        let ctx = ctx.clone();
433        engine.register_fn("raycast", move |ox: f64, oy: f64, oz: f64, dx: f64, dy: f64, dz: f64, max_dist: f64| -> Dynamic {
434            let c = ctx.lock().unwrap();
435            let physics = unsafe { c.physics_ref() };
436            let physics = match physics {
437                Some(p) => p,
438                None => return Dynamic::UNIT,
439            };
440
441            // Exclude the current entity from the raycast
442            let exclude = Some(c.current_entity);
443
444            match physics.raycast(
445                [ox as f32, oy as f32, oz as f32],
446                [dx as f32, dy as f32, dz as f32],
447                max_dist as f32,
448                exclude,
449            ) {
450                Some(hit) => {
451                    let mut map = Map::new();
452                    map.insert("entity".into(), Dynamic::from(hit.entity_id.raw() as i64));
453                    map.insert("distance".into(), Dynamic::from(hit.distance as f64));
454                    map.insert("point_x".into(), Dynamic::from(hit.point[0] as f64));
455                    map.insert("point_y".into(), Dynamic::from(hit.point[1] as f64));
456                    map.insert("point_z".into(), Dynamic::from(hit.point[2] as f64));
457                    map.insert("normal_x".into(), Dynamic::from(hit.normal[0] as f64));
458                    map.insert("normal_y".into(), Dynamic::from(hit.normal[1] as f64));
459                    map.insert("normal_z".into(), Dynamic::from(hit.normal[2] as f64));
460                    Dynamic::from(map)
461                }
462                None => Dynamic::UNIT,
463            }
464        });
465    }
466
467    // get_camera_direction() -> Map #{x, y, z}
468    {
469        let ctx = ctx.clone();
470        engine.register_fn("get_camera_direction", move || -> Map {
471            let c = ctx.lock().unwrap();
472            let mut map = Map::new();
473            map.insert("x".into(), Dynamic::from(c.camera_direction[0] as f64));
474            map.insert("y".into(), Dynamic::from(c.camera_direction[1] as f64));
475            map.insert("z".into(), Dynamic::from(c.camera_direction[2] as f64));
476            map
477        });
478    }
479
480    // get_camera_position() -> Map #{x, y, z}
481    {
482        let ctx = ctx.clone();
483        engine.register_fn("get_camera_position", move || -> Map {
484            let c = ctx.lock().unwrap();
485            let mut map = Map::new();
486            map.insert("x".into(), Dynamic::from(c.camera_position[0] as f64));
487            map.insert("y".into(), Dynamic::from(c.camera_position[1] as f64));
488            map.insert("z".into(), Dynamic::from(c.camera_position[2] as f64));
489            map
490        });
491    }
492}

494// ─── Math API ────────────────────────────────────────────
495
496fn register_math_api(engine: &mut Engine) {
497    engine.register_fn("clamp", |val: f64, min: f64, max: f64| -> f64 {
498        val.clamp(min, max)
499    });
500
501    engine.register_fn("lerp", |a: f64, b: f64, t: f64| -> f64 {
502        a + (b - a) * t
503    });
504
505    engine.register_fn("random", || -> f64 {
506        // Simple pseudo-random based on time — no external crate needed
507        let t = std::time::SystemTime::now()
508            .duration_since(std::time::UNIX_EPOCH)
509            .unwrap_or_default()
510            .subsec_nanos();
511        // Simple hash to get a 0..1 value
512        let hash = t.wrapping_mul(2654435761);
513        (hash as f64) / (u32::MAX as f64)
514    });
515
516    engine.register_fn("random_range", |min: f64, max: f64| -> f64 {
517        let t = std::time::SystemTime::now()
518            .duration_since(std::time::UNIX_EPOCH)
519            .unwrap_or_default()
520            .subsec_nanos();
521        let hash = t.wrapping_mul(2654435761);
522        let r = (hash as f64) / (u32::MAX as f64);
523        min + (max - min) * r
524    });
525
526    engine.register_fn("sin", |x: f64| -> f64 { x.sin() });
527    engine.register_fn("cos", |x: f64| -> f64 { x.cos() });
528    engine.register_fn("abs", |x: f64| -> f64 { x.abs() });
529    engine.register_fn("sqrt", |x: f64| -> f64 { x.sqrt() });
530    engine.register_fn("floor", |x: f64| -> f64 { x.floor() });
531    engine.register_fn("ceil", |x: f64| -> f64 { x.ceil() });
532    engine.register_fn("min", |a: f64, b: f64| -> f64 { a.min(b) });
533    engine.register_fn("max", |a: f64, b: f64| -> f64 { a.max(b) });
534    engine.register_fn("atan2", |y: f64, x: f64| -> f64 { y.atan2(x) });
535}

820/// Convert a toml::Value to rhai::Dynamic
821pub fn toml_to_dynamic(val: toml::Value) -> Dynamic {
822    match val {
823        toml::Value::Boolean(b) => Dynamic::from(b),
824        toml::Value::Integer(i) => Dynamic::from(i),
825        toml::Value::Float(f) => Dynamic::from(f),
826        toml::Value::String(s) => Dynamic::from(s),
827        toml::Value::Array(arr) => {
828            let items: Vec<Dynamic> = arr.into_iter().map(toml_to_dynamic).collect();
829            Dynamic::from(items)
830        }
831        toml::Value::Table(table) => {
832            let mut map = Map::new();
833            for (k, v) in table {
834                map.insert(k.into(), toml_to_dynamic(v));
835            }
836            Dynamic::from(map)
837        }
838        toml::Value::Datetime(dt) => Dynamic::from(dt.to_string()),
839    }
840}
841
842/// Convert a rhai::Dynamic to toml::Value
843pub fn dynamic_to_toml(val: &Dynamic) -> Option<toml::Value> {
844    if val.is_bool() {
845        return val.as_bool().ok().map(toml::Value::Boolean);
846    }
847    if val.is_int() {
848        return val.as_int().ok().map(toml::Value::Integer);
849    }
850    if val.is_float() {
851        return val.as_float().ok().map(toml::Value::Float);
852    }
853    if val.is_string() {
854        return val.clone().into_string().ok().map(toml::Value::String);
855    }
856    if val.is_array() {
857        let arr = val.clone().into_array().ok()?;
858        let items: Vec<toml::Value> = arr.iter().filter_map(dynamic_to_toml).collect();
859        return Some(toml::Value::Array(items));
860    }
861    if val.is_map() {
862        return Some(map_to_toml_from_dynamic(val));
863    }
864    None
865}
866
867fn map_to_toml(map: &Map) -> toml::Value {
868    let mut table = toml::map::Map::new();
869    for (k, v) in map {
870        if let Some(tv) = dynamic_to_toml(v) {
871            table.insert(k.to_string(), tv);
872        }
873    }
874    toml::Value::Table(table)
875}
876
877fn map_to_toml_from_dynamic(val: &Dynamic) -> toml::Value {
878    if let Some(map) = val.clone().try_cast::<Map>() {
879        return map_to_toml(&map);
880    }
881    toml::Value::String(val.to_string())
882}
