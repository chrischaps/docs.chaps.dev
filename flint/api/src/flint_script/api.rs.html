1//! Rhai API function registration
2//!
3//! All functions accessible from Rhai scripts are registered here.
4//! They access the world through the shared ScriptCallContext.
5
6use crate::context::{LogLevel, ScriptCallContext, ScriptCommand};
7use flint_core::EntityId;
8use rhai::{Dynamic, Engine, Map};
9use std::sync::{Arc, Mutex};

11/// Register all API functions on the Rhai engine
12pub fn register_all(engine: &mut Engine, ctx: Arc<Mutex<ScriptCallContext>>) {
13    register_entity_api(engine, ctx.clone());
14    register_input_api(engine, ctx.clone());
15    register_time_api(engine, ctx.clone());
16    register_audio_api(engine, ctx.clone());
17    register_animation_api(engine, ctx.clone());
18    register_math_api(engine);
19    register_event_api(engine, ctx.clone());
20    register_log_api(engine, ctx);
21}

25fn register_entity_api(engine: &mut Engine, ctx: Arc<Mutex<ScriptCallContext>>) {
26    // self_entity() -> i64
27    {
28        let ctx = ctx.clone();
29        engine.register_fn("self_entity", move || -> i64 {
30            let c = ctx.lock().unwrap();
31            c.current_entity.raw() as i64
32        });
33    }
34
35    // get_entity(name: &str) -> i64
36    {
37        let ctx = ctx.clone();
38        engine.register_fn("get_entity", move |name: &str| -> i64 {
39            let c = ctx.lock().unwrap();
40            let world = unsafe { c.world_ref() };
41            world.get_id(name).map(|id| id.raw() as i64).unwrap_or(-1)
42        });
43    }
44
45    // entity_exists(id: i64) -> bool
46    {
47        let ctx = ctx.clone();
48        engine.register_fn("entity_exists", move |id: i64| -> bool {
49            if id < 0 { return false; }
50            let c = ctx.lock().unwrap();
51            let world = unsafe { c.world_ref() };
52            world.contains(EntityId::from_raw(id as u64))
53        });
54    }
55
56    // entity_name(id: i64) -> String
57    {
58        let ctx = ctx.clone();
59        engine.register_fn("entity_name", move |id: i64| -> String {
60            if id < 0 { return String::new(); }
61            let c = ctx.lock().unwrap();
62            let world = unsafe { c.world_ref() };
63            world.get_name(EntityId::from_raw(id as u64))
64                .unwrap_or("")
65                .to_string()
66        });
67    }
68
69    // has_component(id: i64, comp: &str) -> bool
70    {
71        let ctx = ctx.clone();
72        engine.register_fn("has_component", move |id: i64, comp: &str| -> bool {
73            if id < 0 { return false; }
74            let c = ctx.lock().unwrap();
75            let world = unsafe { c.world_ref() };
76            world.get_components(EntityId::from_raw(id as u64))
77                .map(|comps| comps.has(comp))
78                .unwrap_or(false)
79        });
80    }
81
82    // get_field(id: i64, comp: &str, field: &str) -> Dynamic
83    {
84        let ctx = ctx.clone();
85        engine.register_fn("get_field", move |id: i64, comp: &str, field: &str| -> Dynamic {
86            if id < 0 { return Dynamic::UNIT; }
87            let c = ctx.lock().unwrap();
88            let world = unsafe { c.world_ref() };
89            world.get_components(EntityId::from_raw(id as u64))
90                .and_then(|comps| comps.get_field(comp, field).cloned())
91                .map(toml_to_dynamic)
92                .unwrap_or(Dynamic::UNIT)
93        });
94    }
95
96    // set_field(id: i64, comp: &str, field: &str, val: Dynamic)
97    {
98        let ctx = ctx.clone();
99        engine.register_fn("set_field", move |id: i64, comp: &str, field: &str, val: Dynamic| {
100            if id < 0 { return; }
101            let c = ctx.lock().unwrap();
102            let world = unsafe { c.world_mut() };
103            if let Some(tv) = dynamic_to_toml(&val) {
104                if let Some(comps) = world.get_components_mut(EntityId::from_raw(id as u64)) {
105                    comps.set_field(comp, field, tv);
106                }
107            }
108        });
109    }
110
111    // get_position(id: i64) -> Map #{x, y, z}
112    {
113        let ctx = ctx.clone();
114        engine.register_fn("get_position", move |id: i64| -> Map {
115            let mut map = Map::new();
116            if id < 0 { return map; }
117            let c = ctx.lock().unwrap();
118            let world = unsafe { c.world_ref() };
119            if let Some(transform) = world.get_transform(EntityId::from_raw(id as u64)) {
120                map.insert("x".into(), Dynamic::from(transform.position.x as f64));
121                map.insert("y".into(), Dynamic::from(transform.position.y as f64));
122                map.insert("z".into(), Dynamic::from(transform.position.z as f64));
123            }
124            map
125        });
126    }
127
128    // set_position(id: i64, x: f64, y: f64, z: f64)
129    {
130        let ctx = ctx.clone();
131        engine.register_fn("set_position", move |id: i64, x: f64, y: f64, z: f64| {
132            if id < 0 { return; }
133            let c = ctx.lock().unwrap();
134            let world = unsafe { c.world_mut() };
135            let eid = EntityId::from_raw(id as u64);
136            if let Some(comps) = world.get_components_mut(eid) {
137                comps.set_field("transform", "position", toml::Value::Array(vec![
138                    toml::Value::Float(x),
139                    toml::Value::Float(y),
140                    toml::Value::Float(z),
141                ]));
142            }
143        });
144    }
145
146    // get_rotation(id: i64) -> Map #{x, y, z}
147    {
148        let ctx = ctx.clone();
149        engine.register_fn("get_rotation", move |id: i64| -> Map {
150            let mut map = Map::new();
151            if id < 0 { return map; }
152            let c = ctx.lock().unwrap();
153            let world = unsafe { c.world_ref() };
154            if let Some(transform) = world.get_transform(EntityId::from_raw(id as u64)) {
155                map.insert("x".into(), Dynamic::from(transform.rotation.x as f64));
156                map.insert("y".into(), Dynamic::from(transform.rotation.y as f64));
157                map.insert("z".into(), Dynamic::from(transform.rotation.z as f64));
158            }
159            map
160        });
161    }
162
163    // set_rotation(id: i64, x: f64, y: f64, z: f64)
164    {
165        let ctx = ctx.clone();
166        engine.register_fn("set_rotation", move |id: i64, x: f64, y: f64, z: f64| {
167            if id < 0 { return; }
168            let c = ctx.lock().unwrap();
169            let world = unsafe { c.world_mut() };
170            let eid = EntityId::from_raw(id as u64);
171            if let Some(comps) = world.get_components_mut(eid) {
172                comps.set_field("transform", "rotation", toml::Value::Array(vec![
173                    toml::Value::Float(x),
174                    toml::Value::Float(y),
175                    toml::Value::Float(z),
176                ]));
177            }
178        });
179    }
180
181    // distance(a: i64, b: i64) -> f64
182    {
183        let ctx = ctx.clone();
184        engine.register_fn("distance", move |a: i64, b: i64| -> f64 {
185            if a < 0 || b < 0 { return f64::MAX; }
186            let c = ctx.lock().unwrap();
187            let world = unsafe { c.world_ref() };
188            let ta = world.get_transform(EntityId::from_raw(a as u64));
189            let tb = world.get_transform(EntityId::from_raw(b as u64));
190            match (ta, tb) {
191                (Some(a), Some(b)) => {
192                    let dx = (a.position.x - b.position.x) as f64;
193                    let dy = (a.position.y - b.position.y) as f64;
194                    let dz = (a.position.z - b.position.z) as f64;
195                    (dx * dx + dy * dy + dz * dz).sqrt()
196                }
197                _ => f64::MAX,
198            }
199        });
200    }
201
202    // spawn_entity(name: &str) -> i64
203    {
204        let ctx = ctx.clone();
205        engine.register_fn("spawn_entity", move |name: &str| -> i64 {
206            let c = ctx.lock().unwrap();
207            let world = unsafe { c.world_mut() };
208            world.spawn(name).map(|id| id.raw() as i64).unwrap_or(-1)
209        });
210    }
211
212    // despawn_entity(id: i64)
213    {
214        let ctx = ctx.clone();
215        engine.register_fn("despawn_entity", move |id: i64| {
216            if id < 0 { return; }
217            let c = ctx.lock().unwrap();
218            let world = unsafe { c.world_mut() };
219            let _ = world.despawn(EntityId::from_raw(id as u64));
220        });
221    }
222}

387fn register_math_api(engine: &mut Engine) {
388    engine.register_fn("clamp", |val: f64, min: f64, max: f64| -> f64 {
389        val.clamp(min, max)
390    });
391
392    engine.register_fn("lerp", |a: f64, b: f64, t: f64| -> f64 {
393        a + (b - a) * t
394    });
395
396    engine.register_fn("random", || -> f64 {
397        // Simple pseudo-random based on time — no external crate needed
398        let t = std::time::SystemTime::now()
399            .duration_since(std::time::UNIX_EPOCH)
400            .unwrap_or_default()
401            .subsec_nanos();
402        // Simple hash to get a 0..1 value
403        let hash = t.wrapping_mul(2654435761);
404        (hash as f64) / (u32::MAX as f64)
405    });
406
407    engine.register_fn("random_range", |min: f64, max: f64| -> f64 {
408        let t = std::time::SystemTime::now()
409            .duration_since(std::time::UNIX_EPOCH)
410            .unwrap_or_default()
411            .subsec_nanos();
412        let hash = t.wrapping_mul(2654435761);
413        let r = (hash as f64) / (u32::MAX as f64);
414        min + (max - min) * r
415    });
416
417    engine.register_fn("sin", |x: f64| -> f64 { x.sin() });
418    engine.register_fn("cos", |x: f64| -> f64 { x.cos() });
419    engine.register_fn("abs", |x: f64| -> f64 { x.abs() });
420    engine.register_fn("sqrt", |x: f64| -> f64 { x.sqrt() });
421    engine.register_fn("floor", |x: f64| -> f64 { x.floor() });
422    engine.register_fn("ceil", |x: f64| -> f64 { x.ceil() });
423    engine.register_fn("min", |a: f64, b: f64| -> f64 { a.min(b) });
424    engine.register_fn("max", |a: f64, b: f64| -> f64 { a.max(b) });
425    engine.register_fn("atan2", |y: f64, x: f64| -> f64 { y.atan2(x) });
426}

497/// Convert a toml::Value to rhai::Dynamic
498pub fn toml_to_dynamic(val: toml::Value) -> Dynamic {
499    match val {
500        toml::Value::Boolean(b) => Dynamic::from(b),
501        toml::Value::Integer(i) => Dynamic::from(i),
502        toml::Value::Float(f) => Dynamic::from(f),
503        toml::Value::String(s) => Dynamic::from(s),
504        toml::Value::Array(arr) => {
505            let items: Vec<Dynamic> = arr.into_iter().map(toml_to_dynamic).collect();
506            Dynamic::from(items)
507        }
508        toml::Value::Table(table) => {
509            let mut map = Map::new();
510            for (k, v) in table {
511                map.insert(k.into(), toml_to_dynamic(v));
512            }
513            Dynamic::from(map)
514        }
515        toml::Value::Datetime(dt) => Dynamic::from(dt.to_string()),
516    }
517}
518
519/// Convert a rhai::Dynamic to toml::Value
520pub fn dynamic_to_toml(val: &Dynamic) -> Option<toml::Value> {
521    if val.is_bool() {
522        return val.as_bool().ok().map(toml::Value::Boolean);
523    }
524    if val.is_int() {
525        return val.as_int().ok().map(toml::Value::Integer);
526    }
527    if val.is_float() {
528        return val.as_float().ok().map(toml::Value::Float);
529    }
530    if val.is_string() {
531        return val.clone().into_string().ok().map(toml::Value::String);
532    }
533    if val.is_array() {
534        let arr = val.clone().into_array().ok()?;
535        let items: Vec<toml::Value> = arr.iter().filter_map(dynamic_to_toml).collect();
536        return Some(toml::Value::Array(items));
537    }
538    if val.is_map() {
539        return Some(map_to_toml_from_dynamic(val));
540    }
541    None
542}
543
544fn map_to_toml(map: &Map) -> toml::Value {
545    let mut table = toml::map::Map::new();
546    for (k, v) in map {
547        if let Some(tv) = dynamic_to_toml(v) {
548            table.insert(k.to_string(), tv);
549        }
550    }
551    toml::Value::Table(table)
552}
553
554fn map_to_toml_from_dynamic(val: &Dynamic) -> toml::Value {
555    if let Some(map) = val.clone().try_cast::<Map>() {
556        return map_to_toml(&map);
557    }
558    toml::Value::String(val.to_string())
559}
