1//! Content-addressed asset storage
2
3use flint_core::{ContentHash, FlintError, Result};
4use std::fs;
5use std::path::{Path, PathBuf};
6
7/// Content-addressed file storage
8///
9/// Stores files at `.flint/assets/<first-2-hex>/<full-hash>.<ext>`
10pub struct ContentStore {
11    root: PathBuf,
12}
13
14impl ContentStore {
15    /// Create a new content store at the given root directory
16    pub fn new<P: AsRef<Path>>(root: P) -> Self {
17        Self {
18            root: root.as_ref().to_path_buf(),
19        }
20    }
21
22    /// Store a file and return its content hash
23    pub fn store<P: AsRef<Path>>(&self, source_path: P) -> Result<ContentHash> {
24        let source = source_path.as_ref();
25        let hash = ContentHash::from_file(source)
26            .map_err(|e| FlintError::AssetError(format!("Failed to hash file: {}", e)))?;
27
28        let dest = self.path_for_hash(&hash, source);
29        if dest.exists() {
30            return Ok(hash); // Already stored (dedup)
31        }
32
33        if let Some(parent) = dest.parent() {
34            fs::create_dir_all(parent)?;
35        }
36
37        fs::copy(source, &dest)?;
38        Ok(hash)
39    }
40
41    /// Get the storage path for a hash
42    pub fn get(&self, hash: &ContentHash) -> Option<PathBuf> {
43        let hex = hash.to_hex();
44        let prefix = &hex[..2];
45        let dir = self.root.join(prefix);
46
47        if !dir.exists() {
48            return None;
49        }
50
51        // Look for any file matching the hash
52        if let Ok(entries) = fs::read_dir(&dir) {
53            for entry in entries.flatten() {
54                let name = entry.file_name();
55                let name_str = name.to_string_lossy();
56                if name_str.starts_with(&hex) {
57                    return Some(entry.path());
58                }
59            }
60        }
61
62        None
63    }
64
65    /// Check if a hash exists in the store
66    pub fn contains(&self, hash: &ContentHash) -> bool {
67        self.get(hash).is_some()
68    }
69
70    /// List all stored asset hashes
71    pub fn list(&self) -> Result<Vec<ContentHash>> {
72        let mut hashes = Vec::new();
73
74        if !self.root.exists() {
75            return Ok(hashes);
76        }
77
78        for entry in fs::read_dir(&self.root)? {
79            let entry = entry?;
80            if entry.file_type()?.is_dir() {
81                for file_entry in fs::read_dir(entry.path())? {
82                    let file_entry = file_entry?;
83                    let name = file_entry.file_name();
84                    let name_str = name.to_string_lossy();
85                    // Extract hash from filename (before the extension)
86                    if let Some(hash_hex) = name_str.split('.').next() {
87                        if let Some(hash) = ContentHash::from_prefixed_hex(&format!("sha256:{}", hash_hex)) {
88                            hashes.push(hash);
89                        }
90                    }
91                }
92            }
93        }
94
95        Ok(hashes)
96    }
97
98    /// Remove a stored asset
99    pub fn remove(&self, hash: &ContentHash) -> Result<bool> {
100        if let Some(path) = self.get(hash) {
101            fs::remove_file(path)?;
102            Ok(true)
103        } else {
104            Ok(false)
105        }
106    }
107
108    /// Build the storage path for a hash with extension from source
109    fn path_for_hash(&self, hash: &ContentHash, source: &Path) -> PathBuf {
110        let hex = hash.to_hex();
111        let prefix = &hex[..2];
112        let ext = source
113            .extension()
114            .and_then(|e| e.to_str())
115            .unwrap_or("bin");
116
117        self.root.join(prefix).join(format!("{}.{}", hex, ext))
118    }
119}

126    fn temp_dir() -> PathBuf {
127        let dir = std::env::temp_dir().join(format!("flint_test_{}", uuid::Uuid::new_v4()));
128        fs::create_dir_all(&dir).unwrap();
129        dir
130    }

132    #[test]
133    fn test_store_and_retrieve() {
134        let store_dir = temp_dir();
135        let store = ContentStore::new(&store_dir);
136
137        // Create a test file
138        let test_file = store_dir.join("test.txt");
139        let mut f = fs::File::create(&test_file).unwrap();
140        f.write_all(b"hello world").unwrap();
141
142        let hash = store.store(&test_file).unwrap();
143        assert!(store.contains(&hash));
144
145        let retrieved = store.get(&hash).unwrap();
146        let content = fs::read_to_string(retrieved).unwrap();
147        assert_eq!(content, "hello world");
148
149        // Cleanup
150        fs::remove_dir_all(&store_dir).ok();
151    }

153    #[test]
154    fn test_store_dedup() {
155        let store_dir = temp_dir();
156        let store = ContentStore::new(&store_dir);
157
158        let test_file = store_dir.join("test.txt");

161        let hash1 = store.store(&test_file).unwrap();

163        assert_eq!(hash1, hash2);
