1//! Hierarchical scene tree panel showing all entities
2
3use flint_core::EntityId;
4use flint_ecs::FlintWorld;
5
6/// An entry in the scene tree
7struct TreeEntry {
8    id: EntityId,
9    name: String,
10    archetype: String,
11}
12
13/// Scene tree panel â€” shows entities in a scrollable list
14#[derive(Default)]
15pub struct SceneTree {
16    entries: Vec<TreeEntry>,
17    selected: Option<EntityId>,
18}
19
20impl SceneTree {
21    pub fn new() -> Self {
22        Self::default()
23    }
24
25    /// Rebuild the tree from the current world state
26    pub fn update(&mut self, world: &FlintWorld) {
27        self.entries.clear();
28
29        for entity in world.all_entities() {
30            self.entries.push(TreeEntry {
31                id: entity.id,
32                name: entity.name.clone(),
33                archetype: entity.archetype.clone().unwrap_or_else(|| "unknown".to_string()),
34            });
35        }
36
37        // Sort by name for consistent ordering
38        self.entries.sort_by(|a, b| a.name.cmp(&b.name));
39    }
40
41    /// Get the currently selected entity
42    pub fn selected_entity(&self) -> Option<EntityId> {
43        self.selected
44    }
45
46    /// Draw the scene tree UI
47    pub fn ui(&mut self, ui: &mut egui::Ui) {
48        ui.heading("Scene Tree");
49        ui.separator();
50
51        ui.label(format!("{} entities", self.entries.len()));
52        ui.separator();
53
54        egui::ScrollArea::vertical().show(ui, |ui| {
55            for entry in &self.entries {
56                let is_selected = self.selected == Some(entry.id);
57                let label = format!("{} ({})", entry.name, entry.archetype);
58
59                let response = ui.selectable_label(is_selected, &label);
60                if response.clicked() {
61                    self.selected = Some(entry.id);
62                }
63            }
64        });
65    }
66}
