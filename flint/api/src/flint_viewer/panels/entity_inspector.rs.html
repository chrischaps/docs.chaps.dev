1//! Read-only entity inspector panel — displays components and their properties
2
3use flint_core::EntityId;
4use flint_ecs::FlintWorld;
5
6/// Entity inspector — read-only display of all components for a selected entity
7#[derive(Default)]
8pub struct EntityInspector;
9
10impl EntityInspector {
11    pub fn new() -> Self {
12        Self
13    }
14
15    pub fn ui(&self, ui: &mut egui::Ui, world: &FlintWorld, entity_id: EntityId) {
16        ui.heading("Entity Inspector");
17        ui.separator();
18
19        // Entity info
20        let entities = world.all_entities();
21        let entity = entities.iter().find(|e| e.id == entity_id);
22
23        let Some(entity) = entity else {
24            ui.label("Entity not found");
25            return;
26        };
27
28        ui.label(format!("Name: {}", entity.name));
29        ui.label(format!(
30            "Archetype: {}",
31            entity.archetype.as_deref().unwrap_or("(none)")
32        ));
33        ui.label(format!("ID: {}", entity.id.raw()));
34
35        // Transform
36        let transform = world.get_transform(entity_id).unwrap_or_default();
37        ui.separator();
38        ui.collapsing("Transform", |ui| {
39            ui.horizontal(|ui| {
40                ui.label("Position:");
41                ui.monospace(format!(
42                    "[{:.2}, {:.2}, {:.2}]",
43                    transform.position.x, transform.position.y, transform.position.z
44                ));
45            });
46            ui.horizontal(|ui| {
47                ui.label("Rotation:");
48                ui.monospace(format!(
49                    "[{:.2}, {:.2}, {:.2}]",
50                    transform.rotation.x, transform.rotation.y, transform.rotation.z
51                ));
52            });
53            ui.horizontal(|ui| {
54                ui.label("Scale:");
55                ui.monospace(format!(
56                    "[{:.2}, {:.2}, {:.2}]",
57                    transform.scale.x, transform.scale.y, transform.scale.z
58                ));
59            });
60        });
61
62        // Dynamic components
63        if let Some(components) = world.get_components(entity_id) {
64            for (comp_name, comp_value) in components.data.iter() {
65                ui.separator();
66                ui.collapsing(comp_name, |ui| {
67                    display_toml_value(ui, comp_value);
68                });
69            }
70        }
71    }
72}

74/// Recursively display a TOML value as a read-only tree
75fn display_toml_value(ui: &mut egui::Ui, value: &toml::Value) {
76    match value {
77        toml::Value::Table(table) => {
78            for (key, val) in table {
79                match val {
80                    toml::Value::Table(_) | toml::Value::Array(_) => {
81                        ui.collapsing(key, |ui| {
82                            display_toml_value(ui, val);
83                        });
84                    }
85                    _ => {
86                        ui.horizontal(|ui| {
87                            ui.label(format!("{}:", key));
88                            ui.monospace(format_toml_leaf(val));
89                        });
90                    }
91                }
92            }
93        }
94        toml::Value::Array(arr) => {
95            for (i, val) in arr.iter().enumerate() {
96                match val {
97                    toml::Value::Table(_) | toml::Value::Array(_) => {
98                        ui.collapsing(format!("[{}]", i), |ui| {
99                            display_toml_value(ui, val);
100                        });
101                    }
102                    _ => {
103                        ui.horizontal(|ui| {
104                            ui.label(format!("[{}]:", i));
105                            ui.monospace(format_toml_leaf(val));
106                        });
107                    }
108                }
109            }
110        }
111        _ => {
112            ui.monospace(format_toml_leaf(value));
113        }
114    }
115}
116
117fn format_toml_leaf(value: &toml::Value) -> String {
118    match value {
119        toml::Value::String(s) => format!("\"{}\"", s),
120        toml::Value::Integer(i) => i.to_string(),
121        toml::Value::Float(f) => format!("{:.4}", f),
122        toml::Value::Boolean(b) => b.to_string(),
123        toml::Value::Datetime(d) => d.to_string(),
124        _ => value.to_string(),
125    }
126}
