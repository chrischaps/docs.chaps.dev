1//! Runtime system trait
2
3use flint_core::Result;
4use flint_ecs::FlintWorld;
5
6/// A system that can be ticked by the game loop
7///
8/// Systems are updated in registration order. Fixed update runs at a constant
9/// rate (physics), while update runs once per frame (rendering, input).
10pub trait RuntimeSystem {
11    /// Called once when the system is first registered
12    fn initialize(&mut self, world: &mut FlintWorld) -> Result<()>;
13
14    /// Called at a fixed rate (e.g. 60Hz) for deterministic simulation
15    fn fixed_update(&mut self, world: &mut FlintWorld, dt: f64) -> Result<()>;
16
17    /// Called once per frame for variable-rate logic
18    fn update(&mut self, world: &mut FlintWorld, dt: f64) -> Result<()>;
19
20    /// Called when the system is being shut down
21    fn shutdown(&mut self) -> Result<()>;
22
23    /// Human-readable name for this system
24    fn name(&self) -> &str;
25}
