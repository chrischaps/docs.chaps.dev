1//! Game clock with fixed-timestep accumulator
2
3use std::time::Instant;
4
5/// Tracks game time and provides a fixed-timestep accumulator for physics updates
6pub struct GameClock {
7    /// Total elapsed game time in seconds
8    pub total_time: f64,
9    /// Time since last frame in seconds
10    pub delta_time: f64,
11    /// Fixed timestep interval (default: 1/60 second)
12    pub fixed_timestep: f64,
13    /// Accumulated time for fixed-step consumption
14    accumulator: f64,
15    /// Last tick instant
16    last_instant: Instant,
17    /// Whether this is the first tick
18    first_tick: bool,
19}
20
21impl Default for GameClock {
22    fn default() -> Self {
23        Self {
24            total_time: 0.0,
25            delta_time: 0.0,
26            fixed_timestep: 1.0 / 60.0,
27            accumulator: 0.0,
28            last_instant: Instant::now(),
29            first_tick: true,
30        }
31    }
32}
33
34impl GameClock {
35    /// Create a new game clock with default 60Hz fixed timestep
36    pub fn new() -> Self {
37        Self::default()
38    }
39
40    /// Create a game clock with a custom fixed timestep
41    pub fn with_fixed_timestep(hz: f64) -> Self {
42        Self {
43            fixed_timestep: 1.0 / hz,
44            ..Self::default()
45        }
46    }
47
48    /// Advance the clock. Call once per frame.
49    pub fn tick(&mut self) {
50        let now = Instant::now();
51
52        if self.first_tick {
53            self.first_tick = false;
54            self.last_instant = now;
55            self.delta_time = 0.0;
56            return;
57        }
58
59        let elapsed = now.duration_since(self.last_instant).as_secs_f64();
60        self.last_instant = now;
61
62        // Clamp to avoid spiral of death (max 250ms frame time)
63        self.delta_time = elapsed.min(0.25);
64        self.total_time += self.delta_time;
65        self.accumulator += self.delta_time;
66    }
67
68    /// Returns true if there's enough accumulated time for a fixed update step
69    pub fn should_fixed_update(&self) -> bool {
70        self.accumulator >= self.fixed_timestep
71    }
72
73    /// Consume one fixed timestep from the accumulator
74    pub fn consume_fixed_step(&mut self) {
75        self.accumulator -= self.fixed_timestep;
76    }
77
78    /// Get the interpolation alpha for rendering between fixed steps
79    pub fn interpolation_alpha(&self) -> f64 {
80        self.accumulator / self.fixed_timestep
81    }
82}

84#[cfg(test)]
85mod tests {
86    use super::*;
87
88    #[test]
89    fn test_clock_defaults() {
90        let clock = GameClock::new();
91        assert!((clock.fixed_timestep - 1.0 / 60.0).abs() < 1e-10);
92        assert_eq!(clock.total_time, 0.0);
93        assert_eq!(clock.delta_time, 0.0);
94    }
95
96    #[test]
97    fn test_custom_timestep() {
98        let clock = GameClock::with_fixed_timestep(30.0);
99        assert!((clock.fixed_timestep - 1.0 / 30.0).abs() < 1e-10);
100    }
101
102    #[test]
103    fn test_first_tick_zero_delta() {
104        let mut clock = GameClock::new();
105        clock.tick();
106        assert_eq!(clock.delta_time, 0.0);
107    }
108
109    #[test]
110    fn test_accumulator_logic() {
111        let mut clock = GameClock::new();
112        clock.fixed_timestep = 1.0 / 60.0;
113        // Simulate adding time directly
114        clock.accumulator = 1.0 / 30.0; // Two fixed steps worth
115
116        assert!(clock.should_fixed_update());
117        clock.consume_fixed_step();
118        assert!(clock.should_fixed_update());
119        clock.consume_fixed_step();
120        assert!(!clock.should_fixed_update());
121    }
122
123    #[test]
124    fn test_interpolation_alpha() {
125        let mut clock = GameClock::new();
126        clock.fixed_timestep = 1.0 / 60.0;
127        clock.accumulator = clock.fixed_timestep * 0.5;
128        let alpha = clock.interpolation_alpha();
129        assert!((alpha - 0.5).abs() < 1e-10);
130    }
