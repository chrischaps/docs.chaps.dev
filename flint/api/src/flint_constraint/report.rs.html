1//! Validation report types
2
3use crate::types::Severity;
4use flint_core::EntityId;
5
6/// A single constraint violation
7#[derive(Debug, Clone)]
8pub struct Violation {
9    pub constraint_name: String,
10    pub entity_name: String,
11    pub entity_id: EntityId,
12    pub severity: Severity,
13    pub message: String,
14    pub has_auto_fix: bool,
15}
16
17/// A complete validation report
18#[derive(Debug, Default)]
19pub struct ValidationReport {
20    pub violations: Vec<Violation>,
21}
22
23impl ValidationReport {
24    /// Create an empty report
25    pub fn new() -> Self {
26        Self::default()
27    }
28
29    /// Check if the scene is valid (no errors)
30    pub fn is_valid(&self) -> bool {
31        !self
32            .violations
33            .iter()
34            .any(|v| v.severity == Severity::Error)
35    }
36
37    /// Count violations by severity
38    pub fn error_count(&self) -> usize {
39        self.violations
40            .iter()
41            .filter(|v| v.severity == Severity::Error)
42            .count()
43    }
44
45    pub fn warning_count(&self) -> usize {
46        self.violations
47            .iter()
48            .filter(|v| v.severity == Severity::Warning)
49            .count()
50    }
51
52    pub fn info_count(&self) -> usize {
53        self.violations
54            .iter()
55            .filter(|v| v.severity == Severity::Info)
56            .count()
57    }
58
59    /// Get a human-readable summary
60    pub fn summary(&self) -> String {
61        let total = self.violations.len();
62        if total == 0 {
63            return "No violations found.".to_string();
64        }
65
66        format!(
67            "{} violation(s): {} error(s), {} warning(s), {} info",
68            total,
69            self.error_count(),
70            self.warning_count(),
71            self.info_count(),
72        )
73    }
74}
