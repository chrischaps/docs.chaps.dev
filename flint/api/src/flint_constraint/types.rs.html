3use serde::{Deserialize, Serialize};
4use std::collections::HashMap;
5
6/// Severity level for constraint violations
7#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
8#[serde(rename_all = "lowercase")]
9pub enum Severity {
10    Error,
11    Warning,
12    Info,
13}
14
15/// The kind of constraint to enforce
16#[derive(Debug, Clone, Serialize, Deserialize)]
17#[serde(tag = "type", rename_all = "snake_case")]
18pub enum ConstraintKind {
19    /// Entity with given archetype must have a specific component
20    RequiredComponent {
21        archetype: String,
22        component: String,
23    },
24    /// Entity must have a child with the given archetype
25    RequiredChild { child_archetype: String },
26    /// A numeric field must be within a range
27    ValueRange {
28        field: String,
29        min: f64,
30        max: f64,
31    },
32    /// A reference field must point to a valid entity
33    ReferenceValid { field: String },
34    /// A simple query-based rule expression
35    QueryRule { rule: String },
36}
37
38/// Strategy for automatically fixing a constraint violation
39#[derive(Debug, Clone, Serialize, Deserialize)]
40#[serde(tag = "strategy", rename_all = "snake_case")]
41pub enum AutoFixStrategy {
42    /// Add a child entity with the given archetype and defaults
43    AddChild {
44        archetype: String,
45        #[serde(default)]
46        defaults: HashMap<String, toml::Value>,
47    },
48    /// Set a field to a default value
49    SetDefault { field: String, value: toml::Value },
50    /// Remove the invalid field/component
51    RemoveInvalid,
52    /// Copy a field value from the parent entity
53    AssignFromParent {
54        field: String,
55        source_field: String,
56    },
57}
58
59/// Auto-fix configuration for a constraint
60#[derive(Debug, Clone, Serialize, Deserialize)]
61pub struct AutoFix {
62    pub enabled: bool,
63    #[serde(flatten)]
64    pub strategy: AutoFixStrategy,
65}
66
67/// A complete constraint definition
68#[derive(Debug, Clone, Serialize, Deserialize)]
69pub struct ConstraintDef {
70    pub name: String,
71    #[serde(default)]
72    pub description: Option<String>,
73    /// Query in flint-query syntax to select entities
74    pub query: String,
75    pub kind: ConstraintKind,
76    pub severity: Severity,
77    /// Message template with `{name}` and `{archetype}` placeholders
78    pub message: String,
79    #[serde(default)]
80    pub auto_fix: Option<AutoFix>,
81}
82
83/// TOML file format for constraint definitions
84#[derive(Debug, Deserialize)]
85pub struct ConstraintFile {
86    pub constraint: Vec<ConstraintDef>,
87}

93    #[test]
94    fn test_parse_constraint_from_toml() {
95        let toml_str = r#"
96[[constraint]]
97name = "doors_have_handles"
98query = "entities where archetype == 'door'"
99severity = "warning"
100message = "Door '{name}' is missing a handle"
101
102[constraint.kind]
103type = "required_child"
104child_archetype = "handle"
105"#;
106
107        let file: ConstraintFile = toml::from_str(toml_str).unwrap();
108        assert_eq!(file.constraint.len(), 1);
109        let c = &file.constraint[0];
110        assert_eq!(c.name, "doors_have_handles");
111        assert_eq!(c.severity, Severity::Warning);
112        assert!(matches!(&c.kind, ConstraintKind::RequiredChild { child_archetype } if child_archetype == "handle"));
113        assert!(c.auto_fix.is_none());
114    }

116    #[test]
117    fn test_parse_constraint_with_autofix() {
118        let toml_str = r#"
119[[constraint]]
120name = "doors_have_handles"
121query = "entities where archetype == 'door'"
122severity = "warning"
123message = "Door '{name}' is missing a handle"
124
125[constraint.kind]
126type = "required_child"
127child_archetype = "handle"
128
129[constraint.auto_fix]
130enabled = true
131strategy = "add_child"
132archetype = "handle"
133"#;
134
135        let file: ConstraintFile = toml::from_str(toml_str).unwrap();
136        let c = &file.constraint[0];
137        let fix = c.auto_fix.as_ref().unwrap();
138        assert!(fix.enabled);
139        assert!(matches!(&fix.strategy, AutoFixStrategy::AddChild { archetype, .. } if archetype == "handle"));
140    }
