1//! HUD overlay â€” crosshair and interaction prompts
2//!
3//! Renders a minimal egui overlay on top of the 3D scene:
4//! - Center crosshair dot
5//! - Interaction prompt when near an interactable entity (fade in/out)
6
7use flint_core::EntityId;
8use flint_ecs::FlintWorld;
9use flint_script::engine::find_nearest_interactable;
10
11/// Current HUD state tracked across frames
12pub struct HudState {
13    /// Currently targeted interactable
14    pub target_entity: Option<EntityId>,
15    pub prompt_text: String,
16    pub interaction_type: String,
17    /// Fade alpha for the interaction prompt (0.0..1.0)
18    pub prompt_alpha: f32,
19}
20
21impl HudState {
22    pub fn new() -> Self {
23        Self {
24            target_entity: None,
25            prompt_text: String::new(),
26            interaction_type: String::new(),
27            prompt_alpha: 0.0,
28        }
29    }
30
31    /// Scan the world for the nearest interactable and update fade state
32    pub fn update(&mut self, world: &FlintWorld, dt: f64) {
33        let fade_speed = 5.0_f32; // ~0.2s fade
34
35        match find_nearest_interactable(world) {
36            Some(nearest) => {
37                self.target_entity = Some(nearest.entity_id);
38                self.prompt_text = nearest.prompt_text;
39                self.interaction_type = nearest.interaction_type;
40                self.prompt_alpha = (self.prompt_alpha + fade_speed * dt as f32).min(1.0);
41            }
42            None => {
43                self.prompt_alpha = (self.prompt_alpha - fade_speed * dt as f32).max(0.0);
44                if self.prompt_alpha <= 0.0 {
45                    self.target_entity = None;
46                }
47            }
48        }
49    }
50
51    /// Render the HUD via egui
52    pub fn render(&self, ctx: &egui::Context) {
53        // Draw crosshair and prompt as an overlay area (no panel, no background)
54        egui::Area::new(egui::Id::new("hud_overlay"))
55            .order(egui::Order::Foreground)
56            .anchor(egui::Align2::CENTER_CENTER, egui::Vec2::ZERO)
57            .interactable(false)
58            .show(ctx, |ui| {
59                // Crosshair dot
60                let painter = ui.painter();
61                let screen_center = ctx.screen_rect().center();
62                painter.circle_filled(
63                    screen_center,
64                    2.0,
65                    egui::Color32::from_rgba_unmultiplied(255, 255, 255, 140),
66                );
67                // Thin ring around dot for visibility on bright backgrounds
68                painter.circle_stroke(
69                    screen_center,
70                    3.5,
71                    egui::Stroke::new(0.8, egui::Color32::from_rgba_unmultiplied(0, 0, 0, 80)),
72                );
73            });
74
75        // Interaction prompt (bottom-center, fades in/out)
76        if self.prompt_alpha > 0.01 {
77            let alpha = (self.prompt_alpha * 255.0) as u8;
78
79            // Build the prompt string based on interaction type
80            let verb = match self.interaction_type.as_str() {
81                "talk" => "Talk to",
82                "examine" => "Examine",
83                _ => "Use",
84            };
85            let prompt = format!("[E] {} {}", verb, self.prompt_text);
86
87            egui::Area::new(egui::Id::new("hud_prompt"))
88                .order(egui::Order::Foreground)
89                .anchor(egui::Align2::CENTER_BOTTOM, egui::Vec2::new(0.0, -60.0))
90                .interactable(false)
91                .show(ctx, |ui| {
92                    let bg = egui::Color32::from_rgba_unmultiplied(10, 10, 10, (180.0 * self.prompt_alpha) as u8);
93                    let text_color = egui::Color32::from_rgba_unmultiplied(240, 235, 220, alpha);
94
95                    egui::Frame::none()
96                        .fill(bg)
97                        .rounding(12.0)
98                        .inner_margin(egui::Margin::symmetric(20.0, 10.0))
99                        .show(ui, |ui: &mut egui::Ui| {
100                            ui.label(
101                                egui::RichText::new(&prompt)
102                                    .color(text_color)
103                                    .size(16.0),
104                            );
105                        });
106                });
107        }
108    }
109}
110
111impl Default for HudState {
112    fn default() -> Self {
113        Self::new()
114    }
