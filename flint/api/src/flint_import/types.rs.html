3use flint_asset::AssetMeta;
4
5/// Result of importing a file
6#[derive(Debug)]
7pub struct ImportResult {
8    /// Metadata about the imported asset
9    pub asset_meta: AssetMeta,
10    /// Extracted meshes
11    pub meshes: Vec<ImportedMesh>,
12    /// Extracted textures
13    pub textures: Vec<ImportedTexture>,
14    /// Extracted materials
15    pub materials: Vec<ImportedMaterial>,
16}
17
18/// An imported mesh with vertex data
19#[derive(Debug, Clone)]
20pub struct ImportedMesh {
21    pub name: String,
22    pub positions: Vec<[f32; 3]>,
23    pub normals: Vec<[f32; 3]>,
24    pub uvs: Vec<[f32; 2]>,
25    pub indices: Vec<u32>,
26    pub material_index: Option<usize>,
27}
28
29/// An imported texture
30#[derive(Debug, Clone)]
31pub struct ImportedTexture {
32    pub name: String,
33    pub width: u32,
34    pub height: u32,
35    pub format: String,
36    pub data: Vec<u8>,
37}
38
39/// An imported PBR material
40#[derive(Debug, Clone)]
41pub struct ImportedMaterial {
42    pub name: String,
43    pub base_color: [f32; 4],
44    pub metallic: f32,
45    pub roughness: f32,
46    pub base_color_texture: Option<String>,
47    pub normal_texture: Option<String>,
48    pub metallic_roughness_texture: Option<String>,
49}
