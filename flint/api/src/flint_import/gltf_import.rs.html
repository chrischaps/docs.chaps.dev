12/// Import a glTF or GLB file
13pub fn import_gltf<P: AsRef<Path>>(path: P) -> Result<ImportResult> {
14    let path = path.as_ref();
15    let (document, buffers, images) = gltf::import(path).map_err(|e| {
16        FlintError::ImportError(format!("Failed to import glTF: {}", e))
17    })?;
18
19    let hash = ContentHash::from_file(path)
20        .map_err(|e| FlintError::ImportError(format!("Failed to hash file: {}", e)))?;
21
22    let file_name = path
23        .file_stem()
24        .and_then(|n| n.to_str())
25        .unwrap_or("unnamed")
26        .to_string();
27
28    let ext = path
29        .extension()
30        .and_then(|e| e.to_str())
31        .unwrap_or("glb")
32        .to_string();
33
34    // Build a map from glTF node index -> skin index for mesh-skin association
35    let mut node_skin_map: HashMap<usize, usize> = HashMap::new();
36    for node in document.nodes() {
37        if let Some(skin) = node.skin() {
38            // Associate the mesh node with its skin
39            if node.mesh().is_some() {
40                node_skin_map.insert(node.mesh().unwrap().index(), skin.index());
41            }
42        }
43    }
44
45    let mut meshes = Vec::new();
46    let mut total_vertices = 0u64;
47
48    for mesh in document.meshes() {
49        let mesh_name = mesh
50            .name()
51            .map(String::from)
52            .unwrap_or_else(|| format!("mesh_{}", mesh.index()));
53
54        let skin_index = node_skin_map.get(&mesh.index()).copied();
55
56        for primitive in mesh.primitives() {
57            let reader = primitive.reader(|buffer| Some(&buffers[buffer.index()]));
58
59            let positions: Vec<[f32; 3]> = reader
60                .read_positions()
61                .map(|iter| iter.collect())
62                .unwrap_or_default();
63
64            let normals: Vec<[f32; 3]> = reader
65                .read_normals()
66                .map(|iter| iter.collect())
67                .unwrap_or_default();
68
69            let uvs: Vec<[f32; 2]> = reader
70                .read_tex_coords(0)
71                .map(|iter| iter.into_f32().collect())
72                .unwrap_or_default();
73
74            let indices: Vec<u32> = reader
75                .read_indices()
76                .map(|iter| iter.into_u32().collect())
77                .unwrap_or_default();
78
79            let material_index = primitive.material().index();
80
81            // Read joint indices and weights for skinned meshes
82            let joint_indices: Option<Vec<[u16; 4]>> = reader
83                .read_joints(0)
84                .map(|iter| iter.into_u16().collect());
85
86            let joint_weights: Option<Vec<[f32; 4]>> = reader
87                .read_weights(0)
88                .map(|iter| iter.into_f32().collect());
89
90            total_vertices += positions.len() as u64;
91
92            meshes.push(ImportedMesh {
93                name: mesh_name.clone(),
94                positions,
95                normals,
96                uvs,
97                indices,
98                material_index,
99                joint_indices,
100                joint_weights,
101                skin_index,
102            });
103        }
104    }
105
106    let mut textures = Vec::new();
107    for (i, image) in images.iter().enumerate() {
108        let tex_name = document
109            .textures()
110            .nth(i)
111            .and_then(|t| t.name().map(String::from))
112            .unwrap_or_else(|| format!("texture_{}", i));
113
114        let format = match image.format {
115            gltf::image::Format::R8 => "r8",
116            gltf::image::Format::R8G8 => "rg8",
117            gltf::image::Format::R8G8B8 => "rgb8",
118            gltf::image::Format::R8G8B8A8 => "rgba8",
119            gltf::image::Format::R16 => "r16",
120            gltf::image::Format::R16G16 => "rg16",
121            gltf::image::Format::R16G16B16 => "rgb16",
122            gltf::image::Format::R16G16B16A16 => "rgba16",
123            gltf::image::Format::R32G32B32FLOAT => "rgb32f",
124            gltf::image::Format::R32G32B32A32FLOAT => "rgba32f",
125        };
126
127        textures.push(ImportedTexture {
128            name: tex_name,
129            width: image.width,
130            height: image.height,
131            format: format.to_string(),
132            data: image.pixels.clone(),
133        });
134    }
135
136    let mut materials = Vec::new();
137    for material in document.materials() {
138        let mat_name = material
139            .name()
140            .map(String::from)
141            .unwrap_or_else(|| format!("material_{}", material.index().unwrap_or(0)));
142
143        let pbr = material.pbr_metallic_roughness();
144        let base_color = pbr.base_color_factor();
145        let metallic = pbr.metallic_factor();
146        let roughness = pbr.roughness_factor();
147
148        let base_color_texture = pbr
149            .base_color_texture()
150            .map(|info| {
151                info.texture()
152                    .name()
153                    .map(String::from)
154                    .unwrap_or_else(|| format!("texture_{}", info.texture().index()))
155            });
156
157        let normal_texture = material
158            .normal_texture()
159            .map(|info| {
160                info.texture()
161                    .name()
162                    .map(String::from)
163                    .unwrap_or_else(|| format!("texture_{}", info.texture().index()))
164            });
165
166        let metallic_roughness_texture = pbr
167            .metallic_roughness_texture()
168            .map(|info| {
169                info.texture()
170                    .name()
171                    .map(String::from)
172                    .unwrap_or_else(|| format!("texture_{}", info.texture().index()))
173            });
174
175        materials.push(ImportedMaterial {
176            name: mat_name,
177            base_color,
178            metallic,
179            roughness,
180            base_color_texture,
181            normal_texture,
182            metallic_roughness_texture,
183        });
184    }
185
186    // --- Extract skeletons (skins) ---
187    let skeletons = extract_skeletons(&document, &buffers);
188
189    // --- Extract skeletal animation clips ---
190    let skeletal_clips = extract_skeletal_clips(&document, &buffers, &skeletons);
191
192    let mut properties = HashMap::new();
193    properties.insert(
194        "vertex_count".to_string(),
195        toml::Value::Integer(total_vertices as i64),
196    );
197    properties.insert(
198        "mesh_count".to_string(),
199        toml::Value::Integer(meshes.len() as i64),
200    );
201    properties.insert(
202        "material_count".to_string(),
203        toml::Value::Integer(materials.len() as i64),
204    );
205    if !skeletons.is_empty() {
206        properties.insert(
207            "skeleton_count".to_string(),
208            toml::Value::Integer(skeletons.len() as i64),
209        );
210    }
211    if !skeletal_clips.is_empty() {
212        properties.insert(
213            "skeletal_clip_count".to_string(),
214            toml::Value::Integer(skeletal_clips.len() as i64),
215        );
216    }
217
218    let asset_meta = AssetMeta {
219        name: file_name,
220        asset_type: AssetType::Mesh,
221        hash: hash.to_prefixed_hex(),
222        source_path: Some(path.to_string_lossy().to_string()),
223        format: Some(ext),
224        properties,
225        tags: vec![],
226    };
227
228    Ok(ImportResult {
229        asset_meta,
230        meshes,
231        textures,
232        materials,
233        skeletons,
234        skeletal_clips,
235    })
236}

238/// Extract skeleton data from glTF skins
239fn extract_skeletons(
240    document: &gltf::Document,
241    buffers: &[gltf::buffer::Data],
242) -> Vec<ImportedSkeleton> {
243    let mut skeletons = Vec::new();
244
245    for skin in document.skins() {
246        let skin_name = skin
247            .name()
248            .map(String::from)
249            .unwrap_or_else(|| format!("skin_{}", skin.index()));
250
251        let joint_nodes: Vec<gltf::Node> = skin.joints().collect();
252
253        // Build a map from glTF node index -> joint index within this skin
254        let mut node_to_joint: HashMap<usize, usize> = HashMap::new();
255        for (joint_idx, node) in joint_nodes.iter().enumerate() {
256            node_to_joint.insert(node.index(), joint_idx);
257        }
258
259        // Read inverse bind matrices
260        let reader = skin.reader(|buf| Some(&buffers[buf.index()]));
261        let ibms: Vec<[[f32; 4]; 4]> = reader
262            .read_inverse_bind_matrices()
263            .map(|iter| iter.collect())
264            .unwrap_or_else(|| vec![identity_4x4(); joint_nodes.len()]);
265
266        // Build joint hierarchy
267        let mut joints = Vec::with_capacity(joint_nodes.len());
268        for (joint_idx, node) in joint_nodes.iter().enumerate() {
269            let joint_name = node
270                .name()
271                .map(String::from)
272                .unwrap_or_else(|| format!("joint_{}", joint_idx));
273
274            // Find parent: walk up the glTF node tree and check if the parent is also a joint
275            let parent = find_parent_joint(document, node.index(), &node_to_joint);
276
277            let ibm = ibms.get(joint_idx).copied().unwrap_or_else(identity_4x4);
278
279            joints.push(ImportedJoint {
280                name: joint_name,
281                index: joint_idx,
282                parent,
283                inverse_bind_matrix: ibm,
284            });
285        }
286
287        skeletons.push(ImportedSkeleton {
288            name: skin_name,
289            joints,
290        });
291    }
292
293    skeletons
294}

296/// Find the parent joint index for a given node, if the parent is within the skin's joint set
297fn find_parent_joint(
298    document: &gltf::Document,
299    node_index: usize,
300    node_to_joint: &HashMap<usize, usize>,
301) -> Option<usize> {
302    // Walk all nodes to find the parent of node_index
303    for node in document.nodes() {
304        for child in node.children() {
305            if child.index() == node_index {
306                // Found the parent node â€” check if it's a joint
307                return node_to_joint.get(&node.index()).copied();
308            }
309        }
310    }
311    None
312}

314/// Extract skeletal animation clips from glTF animations
315fn extract_skeletal_clips(
316    document: &gltf::Document,
317    buffers: &[gltf::buffer::Data],
318    skeletons: &[ImportedSkeleton],
319) -> Vec<ImportedSkeletalClip> {
320    if skeletons.is_empty() {
321        return Vec::new();
322    }
323
324    // Build a map from glTF node index -> (skeleton_index, joint_index)
325    // We rebuild from skeletons since each skeleton was built from skin joints
326    let mut node_to_skeleton_joint: HashMap<usize, (usize, usize)> = HashMap::new();
327    for skin in document.skins() {
328        let skel_idx = skin.index();
329        if skel_idx >= skeletons.len() {
330            continue;
331        }
332        for (joint_idx, node) in skin.joints().enumerate() {
333            node_to_skeleton_joint.insert(node.index(), (skel_idx, joint_idx));
334        }
335    }
336
337    let mut clips = Vec::new();
338
339    for animation in document.animations() {
340        let clip_name = animation
341            .name()
342            .map(String::from)
343            .unwrap_or_else(|| format!("clip_{}", animation.index()));
344
345        let mut channels = Vec::new();
346        let mut max_time: f32 = 0.0;
347
348        for channel in animation.channels() {
349            let target_node = channel.target().node().index();
350
351            // Only process channels that target skeleton joints
352            let Some(&(_skel_idx, joint_idx)) = node_to_skeleton_joint.get(&target_node) else {
353                continue;
354            };
355
356            let property = match channel.target().property() {
357                gltf::animation::Property::Translation => JointProperty::Translation,
358                gltf::animation::Property::Rotation => JointProperty::Rotation,
359                gltf::animation::Property::Scale => JointProperty::Scale,
360                _ => continue, // Skip morph targets etc.
361            };
362
363            let reader = channel.reader(|buf| Some(&buffers[buf.index()]));
364
365            let timestamps: Vec<f32> = reader
366                .read_inputs()
367                .map(|iter| iter.collect())
368                .unwrap_or_default();
369
370            let interpolation = match channel.sampler().interpolation() {
371                gltf::animation::Interpolation::Step => "STEP",
372                gltf::animation::Interpolation::Linear => "LINEAR",
373                gltf::animation::Interpolation::CubicSpline => "CUBICSPLINE",
374            };
375
376            let outputs: Vec<Vec<f32>> = match &property {
377                JointProperty::Translation | JointProperty::Scale => {
378                    reader
379                        .read_outputs()
380                        .map(|out| match out {
381                            gltf::animation::util::ReadOutputs::Translations(iter) => {
382                                iter.map(|v| vec![v[0], v[1], v[2]]).collect()
383                            }
384                            gltf::animation::util::ReadOutputs::Scales(iter) => {
385                                iter.map(|v| vec![v[0], v[1], v[2]]).collect()
386                            }
387                            _ => Vec::new(),
388                        })
389                        .unwrap_or_default()
390                }
391                JointProperty::Rotation => {
392                    reader
393                        .read_outputs()
394                        .map(|out| match out {
395                            gltf::animation::util::ReadOutputs::Rotations(iter) => {
396                                iter.into_f32()
397                                    .map(|v| vec![v[0], v[1], v[2], v[3]])
398                                    .collect()
399                            }
400                            _ => Vec::new(),
401                        })
402                        .unwrap_or_default()
403                }
404            };
405
406            let keyframes: Vec<ImportedKeyframe> = timestamps
407                .iter()
408                .zip(outputs.iter())
409                .map(|(&time, value)| {
410                    if time > max_time {
411                        max_time = time;
412                    }
413                    ImportedKeyframe {
414                        time,
415                        value: value.clone(),
416                    }
417                })
418                .collect();
419
420            channels.push(ImportedChannel {
421                joint_index: joint_idx,
422                property,
423                interpolation: interpolation.to_string(),
424                keyframes,
425            });
426        }
427
428        if !channels.is_empty() {
429            clips.push(ImportedSkeletalClip {
430                name: clip_name,
431                duration: max_time,
432                channels,
433            });
434        }
435    }
436
437    clips
438}

440fn identity_4x4() -> [[f32; 4]; 4] {
441    [
442        [1.0, 0.0, 0.0, 0.0],
443        [0.0, 1.0, 0.0, 0.0],
444        [0.0, 0.0, 1.0, 0.0],
445        [0.0, 0.0, 0.0, 1.0],
446    ]
447}
