1//! Dynamic component storage
2
3use serde::{Deserialize, Serialize};
4use std::collections::HashMap;
5
6/// Dynamic components stored as TOML values
7///
8/// This allows archetypes to be defined at runtime in schema files
9/// rather than requiring Rust types for each component.
10#[derive(Debug, Clone, Default, Serialize, Deserialize)]
11pub struct DynamicComponents {
12    /// The archetype name for this entity (if any)
13    #[serde(skip_serializing_if = "Option::is_none")]
14    pub archetype: Option<String>,
15    /// Component data: component_name -> field data
16    #[serde(flatten)]
17    pub data: HashMap<String, toml::Value>,
18}
19
20impl DynamicComponents {
21    /// Create empty components
22    pub fn new() -> Self {
23        Self::default()
24    }
25
26    /// Create components with an archetype
27    pub fn with_archetype(archetype: impl Into<String>) -> Self {
28        Self {
29            archetype: Some(archetype.into()),
30            data: HashMap::new(),
31        }
32    }
33
34    /// Get component data by name
35    pub fn get(&self, component: &str) -> Option<&toml::Value> {
36        self.data.get(component)
37    }
38
39    /// Get mutable component data by name
40    pub fn get_mut(&mut self, component: &str) -> Option<&mut toml::Value> {
41        self.data.get_mut(component)
42    }
43
44    /// Set component data
45    pub fn set(&mut self, component: impl Into<String>, data: toml::Value) {
46        self.data.insert(component.into(), data);
47    }
48
49    /// Remove a component
50    pub fn remove(&mut self, component: &str) -> Option<toml::Value> {
51        self.data.remove(component)
52    }
53
54    /// Check if a component exists
55    pub fn has(&self, component: &str) -> bool {
56        self.data.contains_key(component)
57    }
58
59    /// Get all component names
60    pub fn component_names(&self) -> Vec<&str> {
61        self.data.keys().map(|s| s.as_str()).collect()
62    }
63
64    /// Get a field value from a component
65    pub fn get_field(&self, component: &str, field: &str) -> Option<&toml::Value> {
66        self.data.get(component).and_then(|v| v.get(field))
67    }
68
69    /// Set a field value in a component
70    pub fn set_field(&mut self, component: &str, field: &str, value: toml::Value) {
71        let comp = self
72            .data
73            .entry(component.to_string())
74            .or_insert_with(|| toml::Value::Table(toml::map::Map::new()));
75
76        if let Some(table) = comp.as_table_mut() {
77            table.insert(field.to_string(), value);
78        }
79    }
80
81    /// Merge data from another DynamicComponents
82    pub fn merge(&mut self, other: &DynamicComponents) {
83        for (name, value) in &other.data {
84            if let Some(existing) = self.data.get_mut(name) {
85                // Merge tables, overwrite scalars
86                if let (Some(existing_table), Some(other_table)) =
87                    (existing.as_table_mut(), value.as_table())
88                {
89                    for (k, v) in other_table {
90                        existing_table.insert(k.clone(), v.clone());
91                    }
92                } else {
93                    self.data.insert(name.clone(), value.clone());
94                }
95            } else {
96                self.data.insert(name.clone(), value.clone());
97            }
98        }
99    }
100}
