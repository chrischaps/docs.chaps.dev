19/// Load a scene from a TOML string
20pub fn load_scene_string(
21    content: &str,
22    registry: &SchemaRegistry,
23) -> Result<(FlintWorld, SceneFile)> {
24    let scene_file: SceneFile = toml::from_str(content)?;
25    let mut world = FlintWorld::new();
26
27    // First pass: create all entities
28    for (name, _) in &scene_file.entities {
29        world.spawn(name.clone())?;
30    }
31
32    // Second pass: set up components and relationships
33    for (name, entity_def) in &scene_file.entities {
34        let id = world.get_id(name).unwrap();
35
36        // Set archetype
37        if let Some(archetype) = &entity_def.archetype {
38            let components = world.get_components_mut(id).unwrap();
39            components.archetype = Some(archetype.clone());
40
41            // Apply archetype defaults
42            if let Some(arch_schema) = registry.get_archetype(archetype) {
43                for (comp_name, defaults) in &arch_schema.defaults {
44                    if !components.has(comp_name) {
45                        components.set(comp_name.clone(), defaults.clone());
46                    }
47                }
48            }
49        }
50
51        // Set component data
52        for (comp_name, comp_data) in &entity_def.components {
53            world.set_component(id, comp_name, comp_data.clone())?;
54        }
55
56        // Set parent relationship
57        if let Some(parent_name) = &entity_def.parent {
58            world.set_parent_by_name(name, parent_name)?;
59        }
60    }
61
62    Ok((world, scene_file))
63}

128    #[test]
129    fn test_load_scene_string() {
130        let toml_str = r#"
131[scene]
132name = "Test Scene"
133
134[entities.room1]
135archetype = "room"
136
137[entities.room1.bounds]
138min = [0, 0, 0]
139max = [10, 4, 8]
140
141[entities.door1]
142archetype = "door"
143parent = "room1"
144
145[entities.door1.transform]
146position = [5, 0, 0]
147"#;
148
149        let registry = SchemaRegistry::new();
150        let (world, scene) = load_scene_string(toml_str, &registry).unwrap();
151
152        assert_eq!(scene.scene.name, "Test Scene");
153        assert_eq!(world.entity_count(), 2);
154        assert!(world.contains_name("room1"));
155        assert!(world.contains_name("door1"));
156
157        let door_id = world.get_id("door1").unwrap();
158        let room_id = world.get_id("room1").unwrap();
159        assert_eq!(world.get_parent(door_id), Some(room_id));
160    }

162    #[test]
163    fn test_transform_parsing_pipeline() {
164        let toml_str = r#"
165[scene]
166name = "Transform Test"
167
168[entities.bar_counter]
169archetype = "furniture"
170
171[entities.bar_counter.transform]
172position = [-4, 0, 0]
173
174[entities.kitchen]
175archetype = "room"
176
177[entities.kitchen.transform]
178position = [0, 0, -9]
179
180[entities.table]
181archetype = "furniture"
182
183[entities.table.transform]
184position = [2.5, 1.0, -3.5]
185"#;
186
187        let registry = SchemaRegistry::new();
188        let (world, _) = load_scene_string(toml_str, &registry).unwrap();
189
190        // Verify bar_counter transform
191        let bar_id = world.get_id("bar_counter").unwrap();
192        let bar_transform = world.get_transform(bar_id);
193        eprintln!("bar_counter transform: {:?}", bar_transform);
194        let bar_t = bar_transform.expect("bar_counter should have a transform");
195        assert!((bar_t.position.x - (-4.0)).abs() < 0.001, "bar x={}, expected -4", bar_t.position.x);
196        assert!((bar_t.position.y - 0.0).abs() < 0.001, "bar y={}, expected 0", bar_t.position.y);
197        assert!((bar_t.position.z - 0.0).abs() < 0.001, "bar z={}, expected 0", bar_t.position.z);
198
199        // Verify kitchen transform
200        let kitchen_id = world.get_id("kitchen").unwrap();
201        let kitchen_transform = world.get_transform(kitchen_id);
202        eprintln!("kitchen transform: {:?}", kitchen_transform);
203        let kitchen_t = kitchen_transform.expect("kitchen should have a transform");
204        assert!((kitchen_t.position.z - (-9.0)).abs() < 0.001, "kitchen z={}, expected -9", kitchen_t.position.z);
205
206        // Verify table transform with float values
207        let table_id = world.get_id("table").unwrap();
208        let table_transform = world.get_transform(table_id);
209        eprintln!("table transform: {:?}", table_transform);
210        let table_t = table_transform.expect("table should have a transform");
211        assert!((table_t.position.x - 2.5).abs() < 0.001, "table x={}, expected 2.5", table_t.position.x);
212        assert!((table_t.position.y - 1.0).abs() < 0.001, "table y={}, expected 1.0", table_t.position.y);
213        assert!((table_t.position.z - (-3.5)).abs() < 0.001, "table z={}, expected -3.5", table_t.position.z);
214    }
