1//! Scene file format definitions
2
3use serde::{Deserialize, Serialize};
4use std::collections::HashMap;
5
6/// Root structure of a scene TOML file
7#[derive(Debug, Clone, Serialize, Deserialize)]
8pub struct SceneFile {
9    pub scene: SceneMetadata,
10    #[serde(default)]
11    pub entities: HashMap<String, EntityDef>,
12}
13
14/// Scene metadata
15#[derive(Debug, Clone, Serialize, Deserialize)]
16pub struct SceneMetadata {
17    pub name: String,
18    #[serde(default = "default_version")]
19    pub version: String,
20    #[serde(default)]
21    pub description: Option<String>,
22}
23
24fn default_version() -> String {
25    "1.0".to_string()
26}
27
28/// Definition of an entity in a scene file
29#[derive(Debug, Clone, Serialize, Deserialize)]
30pub struct EntityDef {
31    /// Optional archetype name
32    #[serde(skip_serializing_if = "Option::is_none")]
33    pub archetype: Option<String>,
34    /// Optional parent entity name
35    #[serde(skip_serializing_if = "Option::is_none")]
36    pub parent: Option<String>,
37    /// Component data - all other fields are treated as components
38    #[serde(flatten)]
39    pub components: HashMap<String, toml::Value>,
40}
41
42impl EntityDef {
43    pub fn new() -> Self {
44        Self {
45            archetype: None,
46            parent: None,
47            components: HashMap::new(),
48        }
49    }
50
51    pub fn with_archetype(mut self, archetype: impl Into<String>) -> Self {
52        self.archetype = Some(archetype.into());
53        self
54    }
55
56    pub fn with_parent(mut self, parent: impl Into<String>) -> Self {
57        self.parent = Some(parent.into());
58        self
59    }
60
61    pub fn with_component(mut self, name: impl Into<String>, data: toml::Value) -> Self {
62        self.components.insert(name.into(), data);
63        self
64    }
65}
66
67impl Default for EntityDef {
68    fn default() -> Self {
69        Self::new()
70    }
71}
72
73impl SceneFile {
74    /// Create a new scene file
75    pub fn new(name: impl Into<String>) -> Self {
76        Self {
77            scene: SceneMetadata {
78                name: name.into(),
79                version: default_version(),
80                description: None,
81            },
82            entities: HashMap::new(),
83        }
84    }
85
86    /// Add an entity to the scene
87    pub fn add_entity(&mut self, name: impl Into<String>, entity: EntityDef) {
88        self.entities.insert(name.into(), entity);
89    }
90}

99        scene.add_entity(

118[entities.room1]
119archetype = "room"
