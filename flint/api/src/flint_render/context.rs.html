3use std::sync::Arc;

7#[derive(Debug, Error)]
8pub enum RenderError {
9    #[error("Failed to create surface: {0}")]
10    SurfaceCreation(String),
11    #[error("Failed to get adapter")]
12    AdapterNotFound,
13    #[error("Failed to create device: {0}")]
14    DeviceCreation(String),
15    #[error("Surface error: {0}")]
16    SurfaceError(String),
17    #[error("Failed to read render buffer: {0}")]
18    BufferReadFailed(String),
19}

21/// wgpu render context containing device, queue, and surface
22pub struct RenderContext {
23    pub surface: wgpu::Surface<'static>,
24    pub device: wgpu::Device,
25    pub queue: wgpu::Queue,
26    pub config: wgpu::SurfaceConfiguration,
27    pub size: winit::dpi::PhysicalSize<u32>,
28    pub depth_texture: wgpu::Texture,
29    pub depth_view: wgpu::TextureView,
30}
31
32impl RenderContext {
33    /// Create a new render context for a window
34    pub async fn new(window: Arc<Window>) -> Result<Self, RenderError> {
35        let size = window.inner_size();
36
37        let instance = wgpu::Instance::new(wgpu::InstanceDescriptor {
38            backends: wgpu::Backends::all(),
39            ..Default::default()
40        });
41
42        let surface = instance
43            .create_surface(window.clone())
44            .map_err(|e| RenderError::SurfaceCreation(e.to_string()))?;
45
46        let adapter = instance
47            .request_adapter(&wgpu::RequestAdapterOptions {
48                power_preference: wgpu::PowerPreference::default(),
49                compatible_surface: Some(&surface),
50                force_fallback_adapter: false,
51            })
52            .await
53            .ok_or(RenderError::AdapterNotFound)?;
54
55        let (device, queue) = adapter
56            .request_device(
57                &wgpu::DeviceDescriptor {
58                    label: Some("Flint Device"),
59                    required_features: wgpu::Features::empty(),
60                    required_limits: wgpu::Limits::default(),
61                    memory_hints: Default::default(),
62                },
63                None,
64            )
65            .await
66            .map_err(|e| RenderError::DeviceCreation(e.to_string()))?;
67
68        let surface_caps = surface.get_capabilities(&adapter);
69        let surface_format = surface_caps
70            .formats
71            .iter()
72            .find(|f| f.is_srgb())
73            .copied()
74            .unwrap_or(surface_caps.formats[0]);
75
76        let config = wgpu::SurfaceConfiguration {
77            usage: wgpu::TextureUsages::RENDER_ATTACHMENT,
78            format: surface_format,
79            width: size.width.max(1),
80            height: size.height.max(1),
81            present_mode: wgpu::PresentMode::AutoVsync,
82            alpha_mode: surface_caps.alpha_modes[0],
83            view_formats: vec![],
84            desired_maximum_frame_latency: 2,
85        };
86        surface.configure(&device, &config);
87
88        let (depth_texture, depth_view) = create_depth_texture(&device, &config);
89
90        Ok(Self {
91            surface,
92            device,
93            queue,
94            config,
95            size,
96            depth_texture,
97            depth_view,
98        })
99    }
100
101    /// Resize the surface
102    pub fn resize(&mut self, new_size: winit::dpi::PhysicalSize<u32>) {
103        if new_size.width > 0 && new_size.height > 0 {
104            self.size = new_size;
105            self.config.width = new_size.width;
106            self.config.height = new_size.height;
107            self.surface.configure(&self.device, &self.config);
108
109            let (depth_texture, depth_view) = create_depth_texture(&self.device, &self.config);
110            self.depth_texture = depth_texture;
111            self.depth_view = depth_view;
112        }
113    }
114
115    /// Get aspect ratio
116    pub fn aspect_ratio(&self) -> f32 {
117        self.size.width as f32 / self.size.height as f32
118    }
119}

121fn create_depth_texture(
122    device: &wgpu::Device,
123    config: &wgpu::SurfaceConfiguration,
124) -> (wgpu::Texture, wgpu::TextureView) {
125    let size = wgpu::Extent3d {
126        width: config.width.max(1),
127        height: config.height.max(1),
128        depth_or_array_layers: 1,
129    };
130
131    let desc = wgpu::TextureDescriptor {
132        label: Some("Depth Texture"),
133        size,
134        mip_level_count: 1,
135        sample_count: 1,
136        dimension: wgpu::TextureDimension::D2,
137        format: wgpu::TextureFormat::Depth32Float,
138        usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::TEXTURE_BINDING,
139        view_formats: &[],
140    };
141
142    let texture = device.create_texture(&desc);
143    let view = texture.create_view(&wgpu::TextureViewDescriptor::default());
144
145    (texture, view)
146}
