1//! Render pipeline for skinned meshes with bone matrix storage buffer
2
3use crate::primitives::SkinnedVertex;
4
5/// Maximum bone count per skeleton (determines storage buffer size)
6pub const MAX_BONES: usize = 256;
7
8/// The skinned mesh render pipeline
9pub struct SkinnedPipeline {
10    pub pipeline: wgpu::RenderPipeline,
11    pub bone_bind_group_layout: wgpu::BindGroupLayout,
12}
13
14impl SkinnedPipeline {
15    /// Create the skinned mesh pipeline.
16    ///
17    /// Uses bind groups 0-2 from the standard pipeline (transform, material, lights)
18    /// and adds bind group 3 for bone matrices (storage buffer).
19    pub fn new(
20        device: &wgpu::Device,
21        format: wgpu::TextureFormat,
22        transform_bind_group_layout: &wgpu::BindGroupLayout,
23        material_bind_group_layout: &wgpu::BindGroupLayout,
24        light_bind_group_layout: &wgpu::BindGroupLayout,
25    ) -> Self {
26        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
27            label: Some("Skinned PBR Shader"),
28            source: wgpu::ShaderSource::Wgsl(include_str!("skinned_shader.wgsl").into()),
29        });
30
31        // Bind group 3: Bone matrices storage buffer (vertex shader only)
32        let bone_bind_group_layout =
33            device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
34                entries: &[wgpu::BindGroupLayoutEntry {
35                    binding: 0,
36                    visibility: wgpu::ShaderStages::VERTEX,
37                    ty: wgpu::BindingType::Buffer {
38                        ty: wgpu::BufferBindingType::Storage { read_only: true },
39                        has_dynamic_offset: false,
40                        min_binding_size: None,
41                    },
42                    count: None,
43                }],
44                label: Some("Bone Bind Group Layout"),
45            });
46
47        let pipeline_layout = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
48            label: Some("Skinned PBR Pipeline Layout"),
49            bind_group_layouts: &[
50                transform_bind_group_layout,
51                material_bind_group_layout,
52                light_bind_group_layout,
53                &bone_bind_group_layout,
54            ],
55            push_constant_ranges: &[],
56        });
57
58        let pipeline = device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
59            label: Some("Skinned PBR Render Pipeline"),
60            layout: Some(&pipeline_layout),
61            vertex: wgpu::VertexState {
62                module: &shader,
63                entry_point: Some("vs_skinned"),
64                buffers: &[SkinnedVertex::desc()],
65                compilation_options: Default::default(),
66            },
67            fragment: Some(wgpu::FragmentState {
68                module: &shader,
69                entry_point: Some("fs_skinned"),
70                targets: &[Some(wgpu::ColorTargetState {
71                    format,
72                    blend: Some(wgpu::BlendState::ALPHA_BLENDING),
73                    write_mask: wgpu::ColorWrites::ALL,
74                })],
75                compilation_options: Default::default(),
76            }),
77            primitive: wgpu::PrimitiveState {
78                topology: wgpu::PrimitiveTopology::TriangleList,
79                strip_index_format: None,
80                front_face: wgpu::FrontFace::Ccw,
81                cull_mode: Some(wgpu::Face::Back),
82                polygon_mode: wgpu::PolygonMode::Fill,
83                unclipped_depth: false,
84                conservative: false,
85            },
86            depth_stencil: Some(wgpu::DepthStencilState {
87                format: wgpu::TextureFormat::Depth32Float,
88                depth_write_enabled: true,
89                depth_compare: wgpu::CompareFunction::Less,
90                stencil: wgpu::StencilState::default(),
91                bias: wgpu::DepthBiasState::default(),
92            }),
93            multisample: wgpu::MultisampleState::default(),
94            multiview: None,
95            cache: None,
96        });
97
98        Self {
99            pipeline,
100            bone_bind_group_layout,
101        }
102    }
103}
