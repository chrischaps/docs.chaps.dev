8/// A single GPU-resident mesh primitive with its material data
9pub struct GpuMesh {
10    pub vertex_buffer: wgpu::Buffer,
11    pub index_buffer: wgpu::Buffer,
12    pub index_count: u32,
13    pub material: ImportedMaterial,
14    // Keep raw data for creating per-entity copies
15    vertex_data: Vec<u8>,
16    index_data: Vec<u8>,
17}
18
19impl GpuMesh {
20    /// Create a copy of the vertex buffer for a new draw call
21    pub fn create_vertex_buffer_copy(&self, device: &wgpu::Device) -> wgpu::Buffer {
22        device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
23            label: Some("Imported Vertex Buffer Copy"),
24            contents: &self.vertex_data,
25            usage: wgpu::BufferUsages::VERTEX,
26        })
27    }
28
29    /// Create a copy of the index buffer for a new draw call
30    pub fn create_index_buffer_copy(&self, device: &wgpu::Device) -> wgpu::Buffer {
31        device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
32            label: Some("Imported Index Buffer Copy"),
33            contents: &self.index_data,
34            usage: wgpu::BufferUsages::INDEX,
35        })
36    }
37
38    /// Reinterpret the raw index data as a slice of u32 triangle indices
39    pub fn triangle_indices(&self) -> Vec<u32> {
40        bytemuck::cast_slice::<u8, u32>(&self.index_data).to_vec()
41    }
42
43    /// Reinterpret the raw vertex data as a slice of Vertex structs
44    pub fn vertices(&self) -> Vec<Vertex> {
45        bytemuck::cast_slice::<u8, Vertex>(&self.vertex_data).to_vec()
46    }
47}

49/// Cache of imported meshes uploaded to the GPU, keyed by asset name
50#[derive(Default)]
51pub struct MeshCache {
52    meshes: HashMap<String, Vec<GpuMesh>>,
53}
54
55impl MeshCache {
56    pub fn new() -> Self {
57        Self::default()
58    }
59
60    /// Upload an imported model's meshes to the GPU
61    pub fn upload_imported(
62        &mut self,
63        device: &wgpu::Device,
64        name: &str,
65        import_result: &ImportResult,
66        default_color: [f32; 4],
67    ) {
68        let default_material = ImportedMaterial {
69            name: "default".to_string(),
70            base_color: default_color,
71            metallic: 0.0,
72            roughness: 0.5,
73            base_color_texture: None,
74            normal_texture: None,
75            metallic_roughness_texture: None,
76        };
77
78        let gpu_meshes: Vec<GpuMesh> = import_result
79            .meshes
80            .iter()
81            .map(|mesh| {
82                let material = mesh
83                    .material_index
84                    .and_then(|i| import_result.materials.get(i))
85                    .cloned()
86                    .unwrap_or_else(|| default_material.clone());
87
88                let vertex_count = mesh.positions.len();
89                let vertices: Vec<Vertex> = (0..vertex_count)
90                    .map(|i| {
91                        let position = mesh.positions[i];
92                        let normal = if i < mesh.normals.len() {
93                            mesh.normals[i]
94                        } else {
95                            [0.0, 1.0, 0.0]
96                        };
97                        let uv = if i < mesh.uvs.len() {
98                            mesh.uvs[i]
99                        } else {
100                            [0.0, 0.0]
101                        };
102
103                        Vertex {
104                            position,
105                            normal,
106                            color: material.base_color,
107                            uv,
108                        }
109                    })
110                    .collect();
111
112                let vertex_data = bytemuck::cast_slice(&vertices).to_vec();
113                let index_data = bytemuck::cast_slice(&mesh.indices).to_vec();
114
115                let vertex_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
116                    label: Some(&format!("{} Vertex Buffer", name)),
117                    contents: &vertex_data,
118                    usage: wgpu::BufferUsages::VERTEX,
119                });
120
121                let index_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
122                    label: Some(&format!("{} Index Buffer", name)),
123                    contents: &index_data,
124                    usage: wgpu::BufferUsages::INDEX,
125                });
126
127                GpuMesh {
128                    vertex_buffer,
129                    index_buffer,
130                    index_count: mesh.indices.len() as u32,
131                    material,
132                    vertex_data,
133                    index_data,
134                }
135            })
136            .collect();
137
138        self.meshes.insert(name.to_string(), gpu_meshes);
139    }
140
141    /// Get cached GPU meshes by asset name
142    pub fn get(&self, name: &str) -> Option<&Vec<GpuMesh>> {
143        self.meshes.get(name)
144    }
145
146    /// Check if a model is already cached
147    pub fn contains(&self, name: &str) -> bool {
148        self.meshes.contains_key(name)
149    }
