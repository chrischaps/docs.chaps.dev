6/// A vertex with position, normal, color, and UV coordinates
7#[repr(C)]
8#[derive(Copy, Clone, Debug, Pod, Zeroable)]
9pub struct Vertex {
10    pub position: [f32; 3],
11    pub normal: [f32; 3],
12    pub color: [f32; 4],
13    pub uv: [f32; 2],
14}
15
16impl Vertex {
17    const ATTRIBS: [wgpu::VertexAttribute; 4] = wgpu::vertex_attr_array![
18        0 => Float32x3,
19        1 => Float32x3,
20        2 => Float32x4,
21        3 => Float32x2,
22    ];
23
24    pub fn desc() -> wgpu::VertexBufferLayout<'static> {
25        wgpu::VertexBufferLayout {
26            array_stride: std::mem::size_of::<Vertex>() as wgpu::BufferAddress,
27            step_mode: wgpu::VertexStepMode::Vertex,
28            attributes: &Self::ATTRIBS,
29        }
30    }
31}

33/// A mesh with vertices and indices
34pub struct Mesh {

39impl Mesh {

49/// Create a box mesh with the given dimensions and color
50pub fn create_box_mesh(
51    width: f32,
52    height: f32,
53    depth: f32,
54    color: [f32; 4],
55) -> Mesh {
56    let hw = width / 2.0;
57    let hh = height / 2.0;
58    let hd = depth / 2.0;
59
60    // 8 corners
61    let positions = [
62        [-hw, -hh, -hd], // 0: back-bottom-left
63        [hw, -hh, -hd],  // 1: back-bottom-right
64        [hw, hh, -hd],   // 2: back-top-right
65        [-hw, hh, -hd],  // 3: back-top-left
66        [-hw, -hh, hd],  // 4: front-bottom-left
67        [hw, -hh, hd],   // 5: front-bottom-right
68        [hw, hh, hd],    // 6: front-top-right
69        [-hw, hh, hd],   // 7: front-top-left
70    ];
71
72    let normals = [
73        [0.0, 0.0, -1.0], // back
74        [0.0, 0.0, 1.0],  // front
75        [-1.0, 0.0, 0.0], // left
76        [1.0, 0.0, 0.0],  // right
77        [0.0, -1.0, 0.0], // bottom
78        [0.0, 1.0, 0.0],  // top
79    ];
80
81    // Build vertices with proper normals per face (6 faces x 4 vertices = 24)
82    // Vertex order per face must produce CCW winding for the outward normal
83    // when indexed with [base, base+1, base+2, base, base+2, base+3]
84    let uv = [0.0, 0.0];
85    let vertices = vec![
86        // Back face (z-), normal [0,0,-1]
87        Vertex { position: positions[0], normal: normals[0], color, uv },
88        Vertex { position: positions[3], normal: normals[0], color, uv },
89        Vertex { position: positions[2], normal: normals[0], color, uv },
90        Vertex { position: positions[1], normal: normals[0], color, uv },
91        // Front face (z+), normal [0,0,1]
92        Vertex { position: positions[4], normal: normals[1], color, uv },
93        Vertex { position: positions[5], normal: normals[1], color, uv },
94        Vertex { position: positions[6], normal: normals[1], color, uv },
95        Vertex { position: positions[7], normal: normals[1], color, uv },
96        // Left face (x-), normal [-1,0,0]
97        Vertex { position: positions[0], normal: normals[2], color, uv },
98        Vertex { position: positions[4], normal: normals[2], color, uv },
99        Vertex { position: positions[7], normal: normals[2], color, uv },
100        Vertex { position: positions[3], normal: normals[2], color, uv },
101        // Right face (x+), normal [1,0,0]
102        Vertex { position: positions[5], normal: normals[3], color, uv },
103        Vertex { position: positions[1], normal: normals[3], color, uv },
104        Vertex { position: positions[2], normal: normals[3], color, uv },
105        Vertex { position: positions[6], normal: normals[3], color, uv },
106        // Bottom face (y-), normal [0,-1,0]
107        Vertex { position: positions[0], normal: normals[4], color, uv },
108        Vertex { position: positions[1], normal: normals[4], color, uv },
109        Vertex { position: positions[5], normal: normals[4], color, uv },
110        Vertex { position: positions[4], normal: normals[4], color, uv },
111        // Top face (y+), normal [0,1,0]
112        Vertex { position: positions[3], normal: normals[5], color, uv },
113        Vertex { position: positions[7], normal: normals[5], color, uv },
114        Vertex { position: positions[6], normal: normals[5], color, uv },
115        Vertex { position: positions[2], normal: normals[5], color, uv },
116    ];
117
118    // Indices (two triangles per face)
119    let indices: Vec<u32> = (0..6u32)
120        .flat_map(|face| {
121            let base = face * 4;
122            [base, base + 1, base + 2, base, base + 2, base + 3]
123        })
124        .collect();
125
126    Mesh { vertices, indices }
127}

129/// Create a wireframe box (edges only)
130pub fn create_wireframe_box_mesh(
131    width: f32,
132    height: f32,
133    depth: f32,
134    color: [f32; 4],
135) -> Mesh {
136    let hw = width / 2.0;
137    let hh = height / 2.0;
138    let hd = depth / 2.0;
139
140    let positions = [
141        [-hw, -hh, -hd],
142        [hw, -hh, -hd],
143        [hw, hh, -hd],
144        [-hw, hh, -hd],
145        [-hw, -hh, hd],
146        [hw, -hh, hd],
147        [hw, hh, hd],
148        [-hw, hh, hd],
149    ];
150
151    let vertices: Vec<Vertex> = positions
152        .iter()
153        .map(|&position| Vertex {
154            position,
155            normal: [0.0, 1.0, 0.0],
156            color,
157            uv: [0.0, 0.0],
158        })
159        .collect();
160
161    // Line indices for 12 edges
162    let indices: Vec<u32> = vec![
163        // Bottom face edges
164        0, 1, 1, 5, 5, 4, 4, 0,
165        // Top face edges
166        3, 2, 2, 6, 6, 7, 7, 3,
167        // Vertical edges
168        0, 3, 1, 2, 5, 6, 4, 7,
169    ];
170
171    Mesh { vertices, indices }
172}

174/// Create a plane mesh
175pub fn create_plane_mesh(width: f32, depth: f32, color: [f32; 4]) -> Mesh {
176    let hw = width / 2.0;
177    let hd = depth / 2.0;
178
179    let vertices = vec![

182            normal: [0.0, 1.0, 0.0],
183            color,

195            color,
196            uv: [1.0, 1.0],

201            color,
202            uv: [0.0, 1.0],
203        },
204    ];
205
206    let indices: Vec<u32> = vec![0, 1, 2, 0, 2, 3];

211/// Create a grid mesh for the ground plane
212pub fn create_grid_mesh(size: f32, divisions: u32, color: [f32; 4]) -> Mesh {
213    let half = size / 2.0;
214    let step = size / divisions as f32;
215
216    let mut vertices = Vec::new();
217    let mut indices = Vec::new();
218
219    let mut idx = 0u32;
220
221    // Lines along X axis
222    for i in 0..=divisions {
223        let z = -half + i as f32 * step;
224        vertices.push(Vertex {
225            position: [-half, 0.0, z],
226            normal: [0.0, 1.0, 0.0],
227            color,
228            uv: [0.0, 0.0],
229        });
230        vertices.push(Vertex {
231            position: [half, 0.0, z],
232            normal: [0.0, 1.0, 0.0],
233            color,
234            uv: [1.0, 0.0],
235        });
236        indices.push(idx);
237        indices.push(idx + 1);
238        idx += 2;
239    }
240
241    // Lines along Z axis
242    for i in 0..=divisions {
243        let x = -half + i as f32 * step;
244        vertices.push(Vertex {
245            position: [x, 0.0, -half],
246            normal: [0.0, 1.0, 0.0],
247            color,
248            uv: [0.0, 0.0],
249        });
250        vertices.push(Vertex {
251            position: [x, 0.0, half],
252            normal: [0.0, 1.0, 0.0],
253            color,
254            uv: [0.0, 1.0],
255        });
256        indices.push(idx);
257        indices.push(idx + 1);
258        idx += 2;
259    }
260
261    Mesh { vertices, indices }
262}

264/// Extract unique edges from a triangle index list for wireframe rendering.
265///
266/// Returns line-list indices that reuse the same vertex buffer. Shared edges
267/// between adjacent triangles are deduplicated using canonical ordering.
268pub fn triangles_to_wireframe_indices(indices: &[u32]) -> Vec<u32> {
269    let mut edges: HashSet<(u32, u32)> = HashSet::new();
270    let mut line_indices = Vec::new();
271
272    for tri in indices.chunks_exact(3) {
273        let (a, b, c) = (tri[0], tri[1], tri[2]);
274        for &(i, j) in &[(a, b), (b, c), (c, a)] {
275            let edge = if i < j { (i, j) } else { (j, i) };
276            if edges.insert(edge) {
277                line_indices.push(edge.0);
278                line_indices.push(edge.1);
279            }
280        }
281    }
282
283    line_indices
284}

286/// Generate line-segment geometry showing face-normal directions.
287///
288/// For each triangle, computes the face center and averaged face normal, then
289/// emits a line from the center to `center + normal * arrow_length`. The
290/// resulting mesh uses cyan coloring and is rendered with the line pipeline.
291pub fn generate_normal_arrows(vertices: &[Vertex], indices: &[u32], arrow_length: f32) -> Mesh {
292    let cyan = [0.0, 1.0, 1.0, 1.0];
293    let mut arrow_verts = Vec::new();
294    let mut arrow_indices = Vec::new();
295    let mut idx = 0u32;
296
297    for tri in indices.chunks_exact(3) {
298        let (a, b, c) = (tri[0] as usize, tri[1] as usize, tri[2] as usize);
299        if a >= vertices.len() || b >= vertices.len() || c >= vertices.len() {
300            continue;
301        }
302
303        let pa = vertices[a].position;
304        let pb = vertices[b].position;
305        let pc = vertices[c].position;
306
307        // Face center
308        let center = [
309            (pa[0] + pb[0] + pc[0]) / 3.0,
310            (pa[1] + pb[1] + pc[1]) / 3.0,
311            (pa[2] + pb[2] + pc[2]) / 3.0,
312        ];
313
314        // Averaged vertex normal (more robust than cross product for degenerate faces)
315        let na = vertices[a].normal;
316        let nb = vertices[b].normal;
317        let nc = vertices[c].normal;
318        let avg = [
319            (na[0] + nb[0] + nc[0]) / 3.0,
320            (na[1] + nb[1] + nc[1]) / 3.0,
321            (na[2] + nb[2] + nc[2]) / 3.0,
322        ];
323        let len = (avg[0] * avg[0] + avg[1] * avg[1] + avg[2] * avg[2]).sqrt();
324        if len < 1e-8 {
325            continue;
326        }
327        let normal = [avg[0] / len, avg[1] / len, avg[2] / len];
328
329        let tip = [
330            center[0] + normal[0] * arrow_length,
331            center[1] + normal[1] * arrow_length,
332            center[2] + normal[2] * arrow_length,
333        ];
334
335        arrow_verts.push(Vertex {
336            position: center,
337            normal,
338            color: cyan,
339            uv: [0.0, 0.0],
340        });
341        arrow_verts.push(Vertex {
342            position: tip,
343            normal,
344            color: cyan,
345            uv: [0.0, 0.0],
346        });
347
348        arrow_indices.push(idx);
349        arrow_indices.push(idx + 1);
350        idx += 2;
351    }
352
353    Mesh {
354        vertices: arrow_verts,
355        indices: arrow_indices,
356    }
