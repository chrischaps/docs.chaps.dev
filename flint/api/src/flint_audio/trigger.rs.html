1//! Audio trigger system: maps GameEvents to sound playback
2//!
3//! Reads `audio_trigger` components from entities and fires sounds
4//! when matching events occur (collisions, interactions, trigger volumes).
5
6use flint_core::{EntityId, Vec3};
7use flint_ecs::FlintWorld;
8use flint_runtime::GameEvent;
9use std::collections::HashMap;
10
11/// A sound to play in response to a game event
12#[derive(Debug, Clone)]
13pub enum AudioCommand {
14    /// Play a sound, optionally at a 3D position
15    Play {
16        sound: String,
17        position: Option<Vec3>,
18        volume: f64,
19    },
20    /// Stop all sounds on an entity's spatial track
21    Stop { entity: EntityId },
22}
23
24/// Rules for which sounds to play on which events
25#[derive(Debug, Clone, Default)]
26struct TriggerRules {
27    on_collision: Option<String>,
28    on_interact: Option<String>,
29    on_enter: Option<String>,
30    on_exit: Option<String>,
31}
32
33/// Reads audio_trigger components and generates AudioCommands from GameEvents
34pub struct AudioTrigger {
35    entity_rules: HashMap<EntityId, TriggerRules>,
36}
37
38impl Default for AudioTrigger {
39    fn default() -> Self {
40        Self::new()
41    }
42}
43
44impl AudioTrigger {
45    pub fn new() -> Self {
46        Self {
47            entity_rules: HashMap::new(),
48        }
49    }
50
51    /// Scan the world for entities with `audio_trigger` components
52    pub fn load_rules(&mut self, world: &FlintWorld) {
53        for entity in world.all_entities() {
54            if self.entity_rules.contains_key(&entity.id) {
55                continue;
56            }
57
58            let components = match world.get_components(entity.id) {
59                Some(c) => c,
60                None => continue,
61            };
62
63            let trigger_data = match components.get("audio_trigger") {
64                Some(v) => v,
65                None => continue,
66            };
67
68            let rules = TriggerRules {
69                on_collision: trigger_data
70                    .get("on_collision")
71                    .and_then(|v| v.as_str().map(String::from)),
72                on_interact: trigger_data
73                    .get("on_interact")
74                    .and_then(|v| v.as_str().map(String::from)),
75                on_enter: trigger_data
76                    .get("on_enter")
77                    .and_then(|v| v.as_str().map(String::from)),
78                on_exit: trigger_data
79                    .get("on_exit")
80                    .and_then(|v| v.as_str().map(String::from)),
81            };
82
83            self.entity_rules.insert(entity.id, rules);
84        }
85    }
86
87    /// Process a list of game events and generate audio commands
88    pub fn process_events(
89        &self,
90        events: &[GameEvent],
91        world: &FlintWorld,
92    ) -> Vec<AudioCommand> {
93        let mut commands = Vec::new();
94
95        for event in events {
96            match event {
97                GameEvent::CollisionStarted { entity_a, entity_b } => {
98                    // Check both entities for on_collision triggers
99                    if let Some(cmd) = self.check_collision(*entity_a, world) {
100                        commands.push(cmd);
101                    }
102                    if let Some(cmd) = self.check_collision(*entity_b, world) {
103                        commands.push(cmd);
104                    }
105                }
106                GameEvent::TriggerEntered { entity: _, trigger } => {
107                    if let Some(rules) = self.entity_rules.get(trigger) {
108                        if let Some(sound) = &rules.on_enter {
109                            let pos = world
110                                .get_transform(*trigger)
111                                .map(|t| t.position);
112                            commands.push(AudioCommand::Play {
113                                sound: sound.clone(),
114                                position: pos,
115                                volume: 1.0,
116                            });
117                        }
118                    }
119                }
120                GameEvent::TriggerExited { entity: _, trigger } => {
121                    if let Some(rules) = self.entity_rules.get(trigger) {
122                        if let Some(sound) = &rules.on_exit {
123                            let pos = world
124                                .get_transform(*trigger)
125                                .map(|t| t.position);
126                            commands.push(AudioCommand::Play {
127                                sound: sound.clone(),
128                                position: pos,
129                                volume: 1.0,
130                            });
131                        }
132                    }
133                }
134                GameEvent::ActionPressed(action) => {
135                    if action == "interact" {
136                        // For interact, we'd need to know which entity is targeted
137                        // This will be expanded when interaction raycasting is added
138                    }
139                }
140                _ => {}
141            }
142        }
143
144        commands
145    }
146
147    fn check_collision(&self, entity: EntityId, world: &FlintWorld) -> Option<AudioCommand> {
148        let rules = self.entity_rules.get(&entity)?;
149        let sound = rules.on_collision.as_ref()?;
150        let pos = world.get_transform(entity).map(|t| t.position);
151
152        Some(AudioCommand::Play {
153            sound: sound.clone(),
154            position: pos,
155            volume: 1.0,
156        })
157    }
158}

164    #[test]
165    fn test_trigger_loads_rules() {
166        let mut world = FlintWorld::new();
167        let id = world.spawn("door").unwrap();
168
169        let trigger_data = toml::Value::Table({
170            let mut t = toml::map::Map::new();
171            t.insert(
172                "on_collision".into(),
173                toml::Value::String("door_bump.ogg".into()),
174            );
175            t.insert(
176                "on_interact".into(),
177                toml::Value::String("door_open.ogg".into()),
178            );
179            t
180        });
181        world
182            .set_component(id, "audio_trigger", trigger_data)
183            .unwrap();
184
185        let mut trigger = AudioTrigger::new();
186        trigger.load_rules(&world);
187
188        assert!(trigger.entity_rules.contains_key(&id));
189        let rules = &trigger.entity_rules[&id];
190        assert_eq!(rules.on_collision.as_deref(), Some("door_bump.ogg"));
191        assert_eq!(rules.on_interact.as_deref(), Some("door_open.ogg"));
192        assert!(rules.on_enter.is_none());
193    }

195    #[test]
196    fn test_collision_event_generates_command() {
197        let mut world = FlintWorld::new();
198        let door_id = world.spawn("door").unwrap();
199        let player_id = world.spawn("player").unwrap();
200
201        // Add transform to door
202        let transform = toml::Value::Table({
203            let mut t = toml::map::Map::new();
204            t.insert(
205                "position".into(),
206                toml::Value::Array(vec![
207                    toml::Value::Float(5.0),
208                    toml::Value::Float(0.0),
209                    toml::Value::Float(3.0),
210                ]),
211            );
212            t
213        });
214        world.set_component(door_id, "transform", transform).unwrap();
215
216        // Add audio_trigger to door
217        let trigger_data = toml::Value::Table({
218            let mut t = toml::map::Map::new();
219            t.insert(
220                "on_collision".into(),
221                toml::Value::String("door_bump.ogg".into()),
222            );
223            t
224        });
225        world
226            .set_component(door_id, "audio_trigger", trigger_data)
227            .unwrap();
228
229        let mut trigger = AudioTrigger::new();
230        trigger.load_rules(&world);
231
232        let events = vec![GameEvent::CollisionStarted {
233            entity_a: player_id,
234            entity_b: door_id,
235        }];
236
237        let commands = trigger.process_events(&events, &world);
238        assert_eq!(commands.len(), 1);
239
240        match &commands[0] {
241            AudioCommand::Play {
242                sound, position, ..
243            } => {
244                assert_eq!(sound, "door_bump.ogg");
245                assert!(position.is_some());
246                let pos = position.unwrap();
247                assert!((pos.x - 5.0).abs() < 0.01);
248            }
249            _ => panic!("Expected Play command"),
250        }
251    }
