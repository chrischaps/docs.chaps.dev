1//! Query result formatting
2
3use flint_ecs::EntityInfo;
4use serde::Serialize;
5
6/// Query result types
7#[derive(Debug, Clone)]
8pub enum QueryResult {
9    Entities(Vec<EntityInfo>),
10    Components(Vec<String>),
11}
12
13impl QueryResult {
14    pub fn is_empty(&self) -> bool {
15        match self {
16            QueryResult::Entities(e) => e.is_empty(),
17            QueryResult::Components(c) => c.is_empty(),
18        }
19    }
20
21    pub fn len(&self) -> usize {
22        match self {
23            QueryResult::Entities(e) => e.len(),
24            QueryResult::Components(c) => c.len(),
25        }
26    }
27}
28
29/// Format query result as JSON
30pub fn format_json(result: &QueryResult) -> String {
31    match result {
32        QueryResult::Entities(entities) => {
33            serde_json::to_string_pretty(entities).unwrap_or_else(|_| "[]".to_string())
34        }
35        QueryResult::Components(components) => {
36            serde_json::to_string_pretty(components).unwrap_or_else(|_| "[]".to_string())
37        }
38    }
39}
40
41/// Format query result as TOML
42pub fn format_toml(result: &QueryResult) -> String {
43    match result {
44        QueryResult::Entities(entities) => {
45            let wrapper = EntityListWrapper { entities: entities.clone() };
46            toml::to_string_pretty(&wrapper).unwrap_or_else(|_| "".to_string())
47        }
48        QueryResult::Components(components) => {
49            let wrapper = ComponentListWrapper { components: components.clone() };
50            toml::to_string_pretty(&wrapper).unwrap_or_else(|_| "".to_string())
51        }
52    }
53}
54
55#[derive(Serialize)]
56struct EntityListWrapper {
57    entities: Vec<EntityInfo>,
58}
59
60#[derive(Serialize)]
61struct ComponentListWrapper {
62    components: Vec<String>,
63}
