7/// Execute a query against a world
8pub fn execute_query(world: &FlintWorld, query: &Query) -> QueryResult {
9    match query.resource.as_str() {
10        "entities" => {
11            let entities = world.all_entities();
12            let filtered = filter_entities(entities, &query.condition, world);
13            QueryResult::Entities(filtered)
14        }
15        "components" => {
16            // List all unique component names across all entities
17            let mut components: Vec<String> = world
18                .all_entities()
19                .iter()
20                .flat_map(|e| e.components.clone())
21                .collect();
22            components.sort();
23            components.dedup();
24            QueryResult::Components(components)
25        }
26        _ => QueryResult::Entities(vec![]),
27    }
28}

30fn filter_entities(
31    entities: Vec<EntityInfo>,
32    condition: &Option<Condition>,
33    world: &FlintWorld,
34) -> Vec<EntityInfo> {
35    match condition {
36        None => entities,
37        Some(cond) => entities
38            .into_iter()
39            .filter(|e| matches_condition(e, cond, world))
40            .collect(),
41    }
42}
43
44fn matches_condition(entity: &EntityInfo, condition: &Condition, world: &FlintWorld) -> bool {
45    let field_value = get_field_value(entity, &condition.field, world);
46
47    match &field_value {
48        None => false,
49        Some(value) => compare_values(value, &condition.operator, &condition.value),
50    }
51}

53fn get_field_value(entity: &EntityInfo, field: &str, world: &FlintWorld) -> Option<FieldValue> {
54    // Handle special fields
55    match field {
56        "archetype" => {
57            return entity.archetype.clone().map(FieldValue::String);
58        }
59        "name" => {
60            return Some(FieldValue::String(entity.name.clone()));
61        }
62        "parent" => {
63            return entity.parent.clone().map(FieldValue::String);
64        }
65        _ => {}
66    }
67
68    // Handle nested component fields (e.g., "door.locked")
69    let parts: Vec<&str> = field.split('.').collect();
70    if parts.len() >= 2 {
71        let component_name = parts[0];
72        let field_path = &parts[1..];
73
74        if let Some(components) = world.get_components(entity.id) {
75            if let Some(comp_data) = components.get(component_name) {
76                return extract_toml_value(comp_data, field_path);
77            }
78        }
79    }
80
81    // Check if field matches a component name (for existence check)
82    if entity.components.contains(&field.to_string()) {
83        return Some(FieldValue::Bool(true));
84    }
85
86    None
87}

89fn extract_toml_value(value: &toml::Value, path: &[&str]) -> Option<FieldValue> {
90    if path.is_empty() {
91        return toml_to_field_value(value);
92    }
93
94    match value {
95        toml::Value::Table(table) => {
96            table.get(path[0]).and_then(|v| extract_toml_value(v, &path[1..]))
97        }
98        _ => None,
99    }
100}
101
102fn toml_to_field_value(value: &toml::Value) -> Option<FieldValue> {
103    match value {
104        toml::Value::String(s) => Some(FieldValue::String(s.clone())),
105        toml::Value::Integer(i) => Some(FieldValue::Number(*i as f64)),
106        toml::Value::Float(f) => Some(FieldValue::Number(*f)),
107        toml::Value::Boolean(b) => Some(FieldValue::Bool(*b)),
108        _ => None,
109    }
110}
111
112#[derive(Debug, Clone)]
113enum FieldValue {
114    String(String),
115    Number(f64),
116    Bool(bool),
117}
118
119fn compare_values(field: &FieldValue, op: &Operator, query: &QueryValue) -> bool {
120    match (field, query) {
121        (FieldValue::String(f), QueryValue::String(q)) => match op {
122            Operator::Equal => f == q,
123            Operator::NotEqual => f != q,
124            Operator::Contains => f.contains(q),
125            _ => false,
126        },
127        (FieldValue::Number(f), QueryValue::Number(q)) => match op {
128            Operator::Equal => (f - q).abs() < f64::EPSILON,
129            Operator::NotEqual => (f - q).abs() >= f64::EPSILON,
130            Operator::GreaterThan => f > q,
131            Operator::LessThan => f < q,
132            Operator::GreaterThanOrEqual => f >= q,
133            Operator::LessThanOrEqual => f <= q,
134            Operator::Contains => false,
135        },
136        (FieldValue::Bool(f), QueryValue::Boolean(q)) => match op {
137            Operator::Equal => f == q,
138            Operator::NotEqual => f != q,
139            _ => false,
140        },
141        // Type mismatch
142        _ => false,
143    }
144}
