7/// Clip registry â€” holds all loaded animation clips by name.

12impl AnimationPlayer {

14        Self {

19    /// Register a clip. Overwrites any existing clip with the same name.
20    pub fn add_clip(&mut self, clip: AnimationClip) {
21        self.clips.insert(clip.name.clone(), clip);
22    }
23
24    /// Look up a clip by name.
25    pub fn get_clip(&self, name: &str) -> Option<&AnimationClip> {
26        self.clips.get(name)
27    }
28
29    /// Check if a clip is registered.
30    pub fn has_clip(&self, name: &str) -> bool {
31        self.clips.contains_key(name)
32    }
33
34    /// Number of registered clips.
35    pub fn clip_count(&self) -> usize {
36        self.clips.len()
37    }
38}

40impl Default for AnimationPlayer {

46/// Per-entity playback state for one animation clip.
47#[derive(Debug, Clone)]
48pub struct PlaybackState {
49    /// Name of the clip being played
50    pub clip_name: String,
51    /// Current playback time in seconds
52    pub time: f64,
53    /// Playback speed multiplier (1.0 = normal, negative = reverse)
54    pub speed: f64,
55    /// Whether the clip loops
56    pub looping: bool,
57    /// Whether the clip is currently playing
58    pub playing: bool,
59    /// Set of event times already fired this loop (prevents re-firing)
60    fired_events: HashSet<u64>,
61}
62
63impl PlaybackState {
64    pub fn new(clip_name: String, speed: f64, looping: bool, playing: bool) -> Self {
65        Self {
66            clip_name,
67            time: 0.0,
68            speed,
69            looping,
70            playing,
71            fired_events: HashSet::new(),
72        }
73    }
74
75    /// Clear fired event tracking so events can fire again on replay.
76    pub fn clear_fired_events(&mut self) {
77        self.fired_events.clear();
78    }
79}

81/// Result of advancing a playback: sampled values per track + fired events.
82pub struct AdvanceResult {
83    /// One sampled [f32; 3] per track in the clip
84    pub samples: Vec<[f32; 3]>,
85    /// Events that fired during this advance
86    pub events: Vec<AnimationEvent>,
87}
88
89/// Advance a playback state by `dt` seconds, returning sampled values and events.
90///
91/// Returns `None` if the clip isn't found or playback is stopped.
92pub fn advance(
93    state: &mut PlaybackState,
94    clip: &AnimationClip,
95    dt: f64,
96) -> Option<AdvanceResult> {
97    if !state.playing {
98        // Still sample at current time so static poses work
99        let samples: Vec<[f32; 3]> = clip.tracks.iter().map(|t| sample_track(t, state.time)).collect();
100        return Some(AdvanceResult {
101            samples,
102            events: vec![],
103        });
104    }
105
106    // Advance time
107    state.time += dt * state.speed;
108
109    // Handle looping / clamping
110    if state.looping {
111        if clip.duration > 0.0 {
112            if state.time >= clip.duration {
113                state.time %= clip.duration;
114                state.fired_events.clear(); // reset events for new loop
115            } else if state.time < 0.0 {
116                state.time = clip.duration - (-state.time % clip.duration);
117                state.fired_events.clear();
118            }
119        }
120    } else if state.time >= clip.duration {
121        state.time = clip.duration;
122        state.playing = false;
123    } else if state.time < 0.0 {
124        state.time = 0.0;
125        state.playing = false;
126    }
127
128    // Sample all tracks
129    let samples: Vec<[f32; 3]> = clip
130        .tracks
131        .iter()
132        .map(|t| sample_track(t, state.time))
133        .collect();
134
135    // Collect newly-fired events
136    let mut events = Vec::new();
137    for ev in &clip.events {
138        // Use bits of the float time as a hash key to deduplicate
139        let key = ev.time.to_bits();
140        if ev.time <= state.time && !state.fired_events.contains(&key) {
141            state.fired_events.insert(key);
142            events.push(ev.clone());
143        }
144    }
145
146    Some(AdvanceResult { samples, events })
147}

149#[cfg(test)]
150mod tests {
151    use super::*;
152    use crate::clip::*;

154    fn simple_clip() -> AnimationClip {
155        AnimationClip {
156            name: "test_bob".into(),
157            duration: 2.0,
158            tracks: vec![AnimationTrack {
159                target: TrackTarget::Position,
160                interpolation: Interpolation::Linear,
161                keyframes: vec![
162                    Keyframe { time: 0.0, value: [0.0, 0.0, 0.0], in_tangent: None, out_tangent: None },
163                    Keyframe { time: 2.0, value: [0.0, 4.0, 0.0], in_tangent: None, out_tangent: None },
164                ],
165            }],
166            events: vec![AnimationEvent {
167                time: 1.0,
168                event_name: "halfway".into(),
169            }],
170        }
171    }
172
173    #[test]
174    fn advance_samples_at_current_time() {
175        let clip = simple_clip();
176        let mut state = PlaybackState::new("test_bob".into(), 1.0, false, true);
177        let result = advance(&mut state, &clip, 1.0).unwrap();
178        // After 1.0s of a 2.0s linear track: halfway
179        assert!((result.samples[0][1] - 2.0).abs() < 1e-4);
180    }
181
182    #[test]
183    fn advance_fires_event_once() {
184        let clip = simple_clip();
185        let mut state = PlaybackState::new("test_bob".into(), 1.0, false, true);
186        let r1 = advance(&mut state, &clip, 1.5).unwrap();
187        assert_eq!(r1.events.len(), 1);
188        assert_eq!(r1.events[0].event_name, "halfway");
189        // Advancing again should NOT re-fire the same event
190        let r2 = advance(&mut state, &clip, 0.1).unwrap();
191        assert_eq!(r2.events.len(), 0);
192    }
193
194    #[test]
195    fn advance_loops_and_resets_events() {
196        let clip = simple_clip();
197        let mut state = PlaybackState::new("test_bob".into(), 1.0, true, true);
198        // Advance past the end to trigger a loop
199        let _ = advance(&mut state, &clip, 1.5);
200        let _ = advance(&mut state, &clip, 1.0); // wraps around
201        // After wrap, events should be clearable and re-firable
202        assert!(state.time < clip.duration);
203    }
204
205    #[test]
206    fn advance_stops_at_end_when_not_looping() {
207        let clip = simple_clip();
208        let mut state = PlaybackState::new("test_bob".into(), 1.0, false, true);
209        let _ = advance(&mut state, &clip, 3.0); // overshoot
210        assert_eq!(state.time, 2.0);
211        assert!(!state.playing);
212    }
213
214    #[test]
215    fn paused_still_samples() {
216        let clip = simple_clip();
217        let mut state = PlaybackState::new("test_bob".into(), 1.0, false, false);
218        state.time = 1.0; // manually set to midpoint
219        let result = advance(&mut state, &clip, 1.0).unwrap();
220        // Time should NOT advance (paused)
221        assert_eq!(state.time, 1.0);
222        // But it should still return a sample
223        assert!((result.samples[0][1] - 2.0).abs() < 1e-4);
224    }
