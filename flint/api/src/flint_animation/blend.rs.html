1//! Animation blending utilities for skeletal animation
2//!
3//! Provides crossfade blending between two full pose arrays and
4//! additive blending that layers a delta clip onto a base pose.
5
6use crate::skeleton::JointPose;

8/// Linearly blend two full pose arrays.
9///
10/// Translation and scale use component-wise lerp.
11/// Rotation uses quaternion slerp for correct interpolation.
12/// `weight` of 0.0 = fully `a`, 1.0 = fully `b`.
13pub fn blend_poses(a: &[JointPose], b: &[JointPose], weight: f32, out: &mut [JointPose]) {
14    let count = a.len().min(b.len()).min(out.len());
15    let w = weight.clamp(0.0, 1.0);
16    let iw = 1.0 - w;
17
18    for i in 0..count {
19        // Translation: lerp
20        out[i].translation = [
21            a[i].translation[0] * iw + b[i].translation[0] * w,
22            a[i].translation[1] * iw + b[i].translation[1] * w,
23            a[i].translation[2] * iw + b[i].translation[2] * w,
24        ];
25
26        // Scale: lerp
27        out[i].scale = [
28            a[i].scale[0] * iw + b[i].scale[0] * w,
29            a[i].scale[1] * iw + b[i].scale[1] * w,
30            a[i].scale[2] * iw + b[i].scale[2] * w,
31        ];
32
33        // Rotation: slerp
34        out[i].rotation = quat_slerp(&a[i].rotation, &b[i].rotation, w);
35    }
36}

38/// Additive blend: layer a partial clip's delta onto a base pose.
39///
40/// For each joint: `out = base + (additive - reference) * weight`
41/// Translation/scale are additive differences; rotation uses quaternion multiplication.
42pub fn additive_blend(
43    base: &[JointPose],
44    additive: &[JointPose],
45    reference: &[JointPose],
46    weight: f32,
47    out: &mut [JointPose],
48) {
49    let count = base.len().min(additive.len()).min(reference.len()).min(out.len());
50    let w = weight.clamp(0.0, 1.0);
51
52    for i in 0..count {
53        // Translation: base + (additive - reference) * weight
54        let dt = [
55            (additive[i].translation[0] - reference[i].translation[0]) * w,
56            (additive[i].translation[1] - reference[i].translation[1]) * w,
57            (additive[i].translation[2] - reference[i].translation[2]) * w,
58        ];
59        out[i].translation = [
60            base[i].translation[0] + dt[0],
61            base[i].translation[1] + dt[1],
62            base[i].translation[2] + dt[2],
63        ];
64
65        // Scale: base * lerp(1, additive/reference, weight)
66        let ds = [
67            1.0 + (additive[i].scale[0] / reference[i].scale[0].max(1e-10) - 1.0) * w,
68            1.0 + (additive[i].scale[1] / reference[i].scale[1].max(1e-10) - 1.0) * w,
69            1.0 + (additive[i].scale[2] / reference[i].scale[2].max(1e-10) - 1.0) * w,
70        ];
71        out[i].scale = [
72            base[i].scale[0] * ds[0],
73            base[i].scale[1] * ds[1],
74            base[i].scale[2] * ds[2],
75        ];
76
77        // Rotation: base * slerp(identity, inv(reference) * additive, weight)
78        let ref_inv = quat_conjugate(&reference[i].rotation);
79        let delta_rot = quat_mul(&ref_inv, &additive[i].rotation);
80        let identity = [0.0, 0.0, 0.0, 1.0];
81        let weighted_delta = quat_slerp(&identity, &delta_rot, w);
82        out[i].rotation = quat_normalize(&quat_mul(&base[i].rotation, &weighted_delta));
83    }
84}

86/// Quaternion slerp with shortest-path correction
87fn quat_slerp(a: &[f32; 4], b: &[f32; 4], t: f32) -> [f32; 4] {
88    let mut dot = a[0] * b[0] + a[1] * b[1] + a[2] * b[2] + a[3] * b[3];
89
90    // Ensure shortest path
91    let mut b_adj = *b;
92    if dot < 0.0 {
93        b_adj = [-b[0], -b[1], -b[2], -b[3]];
94        dot = -dot;
95    }
96
97    // If very close, use lerp to avoid division by zero
98    if dot > 0.9995 {
99        let result = [
100            a[0] + t * (b_adj[0] - a[0]),
101            a[1] + t * (b_adj[1] - a[1]),
102            a[2] + t * (b_adj[2] - a[2]),
103            a[3] + t * (b_adj[3] - a[3]),
104        ];
105        return quat_normalize(&result);
106    }
107
108    let theta = dot.acos();
109    let sin_theta = theta.sin();
110    let wa = ((1.0 - t) * theta).sin() / sin_theta;
111    let wb = (t * theta).sin() / sin_theta;
112
113    [
114        a[0] * wa + b_adj[0] * wb,
115        a[1] * wa + b_adj[1] * wb,
116        a[2] * wa + b_adj[2] * wb,
117        a[3] * wa + b_adj[3] * wb,
118    ]
119}
120
121fn quat_normalize(q: &[f32; 4]) -> [f32; 4] {
122    let len = (q[0] * q[0] + q[1] * q[1] + q[2] * q[2] + q[3] * q[3]).sqrt();
123    if len < 1e-10 {
124        return [0.0, 0.0, 0.0, 1.0];
125    }
126    [q[0] / len, q[1] / len, q[2] / len, q[3] / len]
127}
128
129fn quat_conjugate(q: &[f32; 4]) -> [f32; 4] {
130    [-q[0], -q[1], -q[2], q[3]]
131}
132
133fn quat_mul(a: &[f32; 4], b: &[f32; 4]) -> [f32; 4] {
134    // Hamilton product: (x,y,z,w)
135    [
136        a[3] * b[0] + a[0] * b[3] + a[1] * b[2] - a[2] * b[1],
137        a[3] * b[1] - a[0] * b[2] + a[1] * b[3] + a[2] * b[0],
138        a[3] * b[2] + a[0] * b[1] - a[1] * b[0] + a[2] * b[3],
139        a[3] * b[3] - a[0] * b[0] - a[1] * b[1] - a[2] * b[2],
140    ]
141}

156    fn blend_weight_zero_returns_a() {
157        let a = [JointPose {
158            translation: [1.0, 2.0, 3.0],
159            rotation: [0.0, 0.0, 0.0, 1.0],
160            scale: [1.0, 1.0, 1.0],
161        }];
162        let b = [JointPose {
163            translation: [10.0, 20.0, 30.0],
164            rotation: [0.0, 0.0, 0.0, 1.0],
165            scale: [2.0, 2.0, 2.0],
166        }];
167        let mut out = [identity_pose()];
168        blend_poses(&a, &b, 0.0, &mut out);
169        assert!((out[0].translation[0] - 1.0).abs() < 1e-5);
170        assert!((out[0].translation[1] - 2.0).abs() < 1e-5);
171        assert!((out[0].scale[0] - 1.0).abs() < 1e-5);
172    }
