3use anyhow::{Context, Result};
4use flint_schema::{FieldType, SchemaRegistry};
5use std::path::Path;
6
7pub fn run(name: &str, schemas_path: &str) -> Result<()> {
8    if !Path::new(schemas_path).exists() {
9        anyhow::bail!("Schemas directory not found: {}", schemas_path);
10    }
11
12    let registry =
13        SchemaRegistry::load_from_directory(schemas_path).context("Failed to load schemas")?;
14
15    // Try component first, then archetype
16    if let Some(component) = registry.get_component(name) {
17        println!("Component: {}", component.name);
18        if let Some(desc) = &component.description {
19            println!("Description: {}", desc);
20        }
21        println!("");
22        println!("Fields:");
23
24        let mut fields: Vec<_> = component.fields.iter().collect();
25        fields.sort_by_key(|(name, _)| *name);
26
27        for (field_name, field_schema) in fields {
28            let type_str = format_field_type(&field_schema.field_type);
29            let required = if field_schema.required { " (required)" } else { "" };
30
31            print!("  {} : {}{}", field_name, type_str, required);
32
33            if let Some(default) = &field_schema.default {
34                print!(" = {}", format_toml_value(default));
35            }
36
37            println!();
38
39            if let Some(desc) = &field_schema.description {
40                println!("    # {}", desc);
41            }
42
43            if field_schema.min.is_some() || field_schema.max.is_some() {
44                let min = field_schema.min.map(|v| v.to_string()).unwrap_or_default();
45                let max = field_schema.max.map(|v| v.to_string()).unwrap_or_default();
46                println!("    # Range: {} .. {}", min, max);
47            }
48        }
49
50        return Ok(());
51    }
52
53    if let Some(archetype) = registry.get_archetype(name) {
54        println!("Archetype: {}", archetype.name);
55        if let Some(desc) = &archetype.description {
56            println!("Description: {}", desc);
57        }
58        println!("");
59        println!("Components:");
60        for comp_name in &archetype.components {
61            let exists = if registry.get_component(comp_name).is_some() {
62                ""
63            } else {
64                " (not found)"
65            };
66            println!("  - {}{}", comp_name, exists);
67        }
68
69        if !archetype.defaults.is_empty() {
70            println!("");
71            println!("Defaults:");
72            for (comp_name, defaults) in &archetype.defaults {
73                println!("  [{}]", comp_name);
74                if let Some(table) = defaults.as_table() {
75                    for (k, v) in table {
76                        println!("    {} = {}", k, format_toml_value(v));
77                    }
78                } else {
79                    println!("    {}", format_toml_value(defaults));
80                }
81            }
82        }
83
84        return Ok(());
85    }
86
87    // List available schemas
88    println!("Schema '{}' not found.", name);
89    println!("");
90
91    let components = registry.component_names();
92    let archetypes = registry.archetype_names();
93
94    if !components.is_empty() {
95        println!("Available components:");
96        for c in components {
97            println!("  - {}", c);
98        }
99    }
100
101    if !archetypes.is_empty() {
102        println!("");
103        println!("Available archetypes:");
104        for a in archetypes {
105            println!("  - {}", a);
106        }
107    }
108
109    Ok(())
110}

112fn format_field_type(ft: &FieldType) -> String {
113    match ft {
114        FieldType::Bool => "bool".to_string(),
115        FieldType::I32 => "i32".to_string(),
116        FieldType::I64 => "i64".to_string(),
117        FieldType::F32 => "f32".to_string(),
118        FieldType::F64 => "f64".to_string(),
119        FieldType::String => "string".to_string(),
120        FieldType::Vec3 => "vec3".to_string(),
121        FieldType::Transform => "transform".to_string(),
122        FieldType::Color => "color".to_string(),
123        FieldType::Enum { values } => format!("enum[{}]", values.join(", ")),
124        FieldType::Array { element } => format!("array<{}>", format_field_type(element)),
125    }
126}
127
128fn format_toml_value(v: &toml::Value) -> String {
129    match v {
130        toml::Value::String(s) => format!("\"{}\"", s),
131        toml::Value::Integer(i) => i.to_string(),
132        toml::Value::Float(f) => f.to_string(),
133        toml::Value::Boolean(b) => b.to_string(),
134        toml::Value::Array(arr) => {
135            let items: Vec<String> = arr.iter().map(format_toml_value).collect();
136            format!("[{}]", items.join(", "))
137        }
138        toml::Value::Table(t) => {
139            let items: Vec<String> = t
140                .iter()
141                .map(|(k, v)| format!("{} = {}", k, format_toml_value(v)))
142                .collect();
143            format!("{{ {} }}", items.join(", "))
144        }
145        toml::Value::Datetime(d) => d.to_string(),
146    }
147}
