3use anyhow::{Context, Result};
4use flint_scene::load_scene;
5use flint_schema::SchemaRegistry;
6use std::path::Path;
7use winit::event_loop::{ControlFlow, EventLoop};
8
9pub struct PlayArgs {
10    pub scene: String,
11    pub schemas: Vec<String>,
12    pub fullscreen: bool,
13}
14
15pub fn run(args: PlayArgs) -> Result<()> {
16    // Load schemas from all directories
17    let existing: Vec<&str> = args.schemas.iter().map(|s| s.as_str()).filter(|p| Path::new(p).exists()).collect();
18    let registry = if !existing.is_empty() {
19        SchemaRegistry::load_from_directories(&existing).context("Failed to load schemas")?
20    } else {
21        println!("Warning: No schemas directories found");
22        SchemaRegistry::new()
23    };
24
25    // Load scene
26    let (world, scene_file) =
27        load_scene(&args.scene, &registry).context("Failed to load scene")?;
28
29    println!("Loaded scene: {}", scene_file.scene.name);
30    println!("Entities: {}", world.entity_count());
31    println!();
32    println!("Controls:");
33    println!("  WASD     - Move");
34    println!("  Mouse    - Look");
35    println!("  Space    - Jump");
36    println!("  Shift    - Sprint");
37    println!("  Escape   - Release cursor / Exit");
38    println!("  F1       - Cycle debug mode");
39    println!("  F4       - Toggle shadows");
40    println!("  F11      - Toggle fullscreen");
41
42    // Create and run the event loop
43    let event_loop = EventLoop::new()?;
44    event_loop.set_control_flow(ControlFlow::Poll);
45
46    let mut app = flint_player::PlayerApp::new(world, args.scene, args.fullscreen);
47    event_loop.run_app(&mut app)?;
48
49    Ok(())
50}
