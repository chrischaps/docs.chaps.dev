1//! Scene validation command
2
3use anyhow::Result;
4use flint_constraint::{
5    compute_scene_diff, ConstraintEvaluator, ConstraintFixer, ConstraintRegistry, Severity,
6};
7use flint_scene::{load_scene, save_scene_string};
8use flint_schema::SchemaRegistry;
9
10pub struct ValidateArgs {
11    pub scene: String,
12    pub fix: bool,
13    pub dry_run: bool,
14    pub output_diff: bool,
15    pub schemas: String,
16    pub format: String,
17}
18
19pub fn run(args: ValidateArgs) -> Result<()> {
20    let schema_registry = SchemaRegistry::load_from_directory(&args.schemas)?;
21    let constraint_registry = ConstraintRegistry::load_from_directory(&args.schemas)?;
22
23    if constraint_registry.is_empty() {
24        println!("No constraints found in {}/constraints/", args.schemas);
25        println!("Create constraint files to enable validation.");
26        return Ok(());
27    }
28
29    let (world, _scene_file) = load_scene(&args.scene, &schema_registry)?;
30
31    if args.fix || args.dry_run {
32        // Save before state for diff
33        let before_toml = if args.output_diff {
34            Some(save_scene_string(&world, "scene")?)
35        } else {
36            None
37        };
38
39        let fixer = ConstraintFixer::new(&schema_registry, &constraint_registry);
40
41        if args.dry_run {
42            let report = fixer.dry_run(&world)?;
43            println!("Dry run results ({} iteration(s)):", report.iterations);
44
45            if report.actions.is_empty() {
46                println!("  No fixes would be applied.");
47            } else {
48                for action in &report.actions {
49                    println!(
50                        "  [{}] {} — {}",
51                        action.strategy, action.entity_name, action.description
52                    );
53                }
54            }
55
56            if report.remaining_violations > 0 {
57                println!(
58                    "\n  {} violation(s) would remain after fixes.",
59                    report.remaining_violations
60                );
61            }
62
63            if report.cycle_detected {
64                println!("  Warning: Fix cycle detected.");
65            }
66        } else {
67            // Actually fix by reloading, modifying, and saving
68            let (mut world, _) = load_scene(&args.scene, &schema_registry)?;
69            let report = fixer.fix(&mut world)?;
70
71            println!("Fix results ({} iteration(s)):", report.iterations);
72
73            for action in &report.actions {
74                println!(
75                    "  [{}] {} — {}",
76                    action.strategy, action.entity_name, action.description
77                );
78            }
79
80            if report.remaining_violations > 0 {
81                println!(
82                    "\n  {} violation(s) remain after fixes.",
83                    report.remaining_violations
84                );
85            }
86
87            if report.cycle_detected {
88                println!("  Warning: Fix cycle detected.");
89            }
90
91            if !report.actions.is_empty() {
92                // Save the fixed scene
93                flint_scene::save_scene(&args.scene, &world, "scene")?;
94                println!("\nScene saved to {}", args.scene);
95
96                if args.output_diff {
97                    if let Some(before) = before_toml {
98                        let after = save_scene_string(&world, "scene")?;
99                        let diff = compute_scene_diff(&before, &after);
100                        println!("\nDiff:");
101                        print!("{}", diff);
102                    }
103                }
104            }
105        }
106    } else {
107        // Validation only
108        let evaluator =
109            ConstraintEvaluator::new(&world, &schema_registry, &constraint_registry);
110        let report = evaluator.validate();
111
112        if args.format == "json" {
113            print_report_json(&report);
114        } else {
115            print_report_text(&report);
116        }
117
118        if !report.is_valid() {
119            std::process::exit(1);
120        }
121    }
122
123    Ok(())
124}

126fn print_report_text(report: &flint_constraint::ValidationReport) {
127    if report.violations.is_empty() {
128        println!("All constraints passed.");
129        return;
130    }
131
132    println!("{}", report.summary());
133    println!();
134
135    for violation in &report.violations {
136        let severity_str = match violation.severity {
137            Severity::Error => "ERROR",
138            Severity::Warning => "WARN ",
139            Severity::Info => "INFO ",
140        };
141
142        let fix_marker = if violation.has_auto_fix { " [fixable]" } else { "" };
143
144        println!(
145            "  [{}] {}: {}{}",
146            severity_str, violation.entity_name, violation.message, fix_marker
147        );
148    }
149}
150
151fn print_report_json(report: &flint_constraint::ValidationReport) {
152    let violations: Vec<serde_json::Value> = report
153        .violations
154        .iter()
155        .map(|v| {
156            serde_json::json!({
157                "constraint": v.constraint_name,
158                "entity": v.entity_name,
159                "severity": format!("{:?}", v.severity).to_lowercase(),
160                "message": v.message,
161                "has_auto_fix": v.has_auto_fix,
162            })
163        })
164        .collect();
165
166    let output = serde_json::json!({
167        "valid": report.is_valid(),
168        "summary": report.summary(),
169        "errors": report.error_count(),
170        "warnings": report.warning_count(),
171        "info": report.info_count(),
172        "violations": violations,
173    });
174
175    println!("{}", serde_json::to_string_pretty(&output).unwrap());
176}
