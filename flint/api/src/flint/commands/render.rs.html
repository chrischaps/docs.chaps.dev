1//! Headless scene-to-PNG render command
2
3use anyhow::{Context, Result};
4use flint_core::Vec3;
5use flint_import::import_gltf;
6use flint_render::{Camera, DebugMode, HeadlessContext, SceneRenderer};
7use flint_scene::load_scene;
8use flint_schema::SchemaRegistry;
9use std::path::Path;
10
11pub struct RenderArgs {
12    pub scene: String,
13    pub output: String,
14    pub width: u32,
15    pub height: u32,
16    pub schemas: Vec<String>,
17    pub distance: Option<f32>,
18    pub yaw: Option<f32>,
19    pub pitch: Option<f32>,
20    pub target: Option<[f32; 3]>,
21    pub fov: Option<f32>,
22    pub no_grid: bool,
23    pub debug_mode: Option<String>,
24    pub wireframe_overlay: bool,
25    pub show_normals: bool,
26    pub no_tonemapping: bool,
27    pub shadows: bool,
28    pub shadow_resolution: u32,
29}
30
31pub fn run(args: RenderArgs) -> Result<()> {
32    // Load schemas from all directories
33    let existing: Vec<&str> = args.schemas.iter().map(|s| s.as_str()).filter(|p| Path::new(p).exists()).collect();
34    let registry = if !existing.is_empty() {
35        SchemaRegistry::load_from_directories(&existing).context("Failed to load schemas")?
36    } else {
37        println!("Warning: No schemas directories found");
38        SchemaRegistry::new()
39    };
40
41    // Load scene
42    let (world, scene_file) =
43        load_scene(&args.scene, &registry).context("Failed to load scene")?;
44    println!("Loaded scene: {}", scene_file.scene.name);
45    println!("Entities: {}", world.entity_count());
46
47    // Create headless context
48    let ctx = pollster::block_on(HeadlessContext::new(args.width, args.height))
49        .context("Failed to create headless render context")?;
50
51    // Configure camera
52    let mut camera = Camera::new();
53    camera.aspect = ctx.aspect_ratio();
54
55    if let Some(d) = args.distance {
56        camera.distance = d;
57    }
58    if let Some(y) = args.yaw {
59        camera.yaw = y.to_radians();
60    }
61    if let Some(p) = args.pitch {
62        camera.pitch = p.to_radians();
63    }
64    if let Some(t) = args.target {
65        camera.target = Vec3::new(t[0], t[1], t[2]);
66    }
67    if let Some(f) = args.fov {
68        camera.fov = f;
69    }
70    camera.update_orbit();
71
72    // Create scene renderer
73    let mut renderer = SceneRenderer::new_headless(&ctx.device, &ctx.queue, ctx.format);
74    if args.no_grid {
75        renderer.disable_grid();
76    }
77
78    // Load models from the scene
79    let scene_dir = Path::new(&args.scene)
80        .parent()
81        .unwrap_or_else(|| Path::new("."));
82
83    for entity in world.all_entities() {
84        let model_asset = world
85            .get_components(entity.id)
86            .and_then(|components| components.get("model").cloned())
87            .and_then(|model| {
88                model
89                    .get("asset")
90                    .and_then(|v| v.as_str().map(String::from))
91            });
92
93        if let Some(asset_name) = model_asset {
94            if renderer.mesh_cache().contains(&asset_name) {
95                continue;
96            }
97
98            let model_path = scene_dir.join("models").join(format!("{}.glb", asset_name));
99
100            if model_path.exists() {
101                match import_gltf(&model_path) {
102                    Ok(import_result) => {
103                        println!(
104                            "Loaded model: {} ({} meshes, {} materials)",
105                            asset_name,
106                            import_result.meshes.len(),
107                            import_result.materials.len()
108                        );
109                        renderer.load_model(&ctx.device, &ctx.queue, &asset_name, &import_result);
110                    }
111                    Err(e) => {
112                        eprintln!("Failed to load model '{}': {:?}", asset_name, e);
113                    }
114                }
115            } else {
116                eprintln!(
117                    "Model file not found: {} (tried {})",
118                    asset_name,
119                    model_path.display()
120                );
121            }
122        }
123    }
124
125    // Load texture files referenced by material components
126    {
127        let mut loaded: std::collections::HashSet<String> = std::collections::HashSet::new();
128
129        for entity in world.all_entities() {
130            let texture_name = world
131                .get_components(entity.id)
132                .and_then(|components| components.get("material").cloned())
133                .and_then(|material| {
134                    material
135                        .get("texture")
136                        .and_then(|v| v.as_str().map(String::from))
137                });
138
139            if let Some(tex_name) = texture_name {
140                if loaded.contains(&tex_name) {
141                    continue;
142                }
143                loaded.insert(tex_name.clone());
144
145                let tex_path = scene_dir.join(&tex_name);
146                if tex_path.exists() {
147                    match renderer.load_texture_file(&ctx.device, &ctx.queue, &tex_name, &tex_path) {
148                        Ok(true) => {
149                            println!("Loaded texture: {}", tex_name);
150                        }
151                        Ok(false) => {}
152                        Err(e) => {
153                            eprintln!("Failed to load texture '{}': {}", tex_name, e);
154                        }
155                    }
156                } else {
157                    eprintln!(
158                        "Texture file not found: {} (tried {})",
159                        tex_name,
160                        tex_path.display()
161                    );
162                }
163            }
164        }
165    }
166
167    // Apply debug state
168    if let Some(mode_str) = &args.debug_mode {
169        let mode = match mode_str.as_str() {
170            "wireframe" => DebugMode::WireframeOnly,
171            "normals" => DebugMode::Normals,
172            "depth" => DebugMode::Depth,
173            "uv" => DebugMode::UvChecker,
174            "unlit" => DebugMode::Unlit,
175            "metalrough" => DebugMode::MetallicRoughness,
176            _ => DebugMode::Pbr,
177        };
178        renderer.set_debug_mode(mode);
179    }
180    if args.wireframe_overlay {
181        renderer.toggle_wireframe_overlay();
182    }
183    if args.show_normals {
184        renderer.toggle_normal_arrows();
185    }
186    if args.no_tonemapping {
187        renderer.set_tonemapping(false);
188    }
189    if args.shadows {
190        renderer.set_shadows(true);
191    }
192    if args.shadow_resolution != 1024 {
193        renderer.set_shadow_resolution(&ctx.device, args.shadow_resolution);
194    }
195
196    renderer.update_from_world(&world, &ctx.device);
197
198    // Render
199    renderer.render_to(
200        &ctx.device,
201        &ctx.queue,
202        &ctx.depth_view,
203        &camera,
204        &ctx.color_view,
205    );
206
207    // Read back pixels
208    let pixels = pollster::block_on(ctx.read_pixels()).context("Failed to read rendered pixels")?;
209
210    // Encode as PNG
211    let img = image::RgbaImage::from_raw(args.width, args.height, pixels)
212        .context("Failed to create image from pixel data")?;
213    img.save(&args.output)
214        .context(format!("Failed to save image to {}", args.output))?;
215
216    println!(
217        "Rendered {}x{} image to {}",
218        args.width, args.height, args.output
219    );
220
221    Ok(())
222}
