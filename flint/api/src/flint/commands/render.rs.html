1//! Headless scene-to-PNG render command
2
3use anyhow::{Context, Result};
4use flint_core::Vec3;
5use flint_import::import_gltf;
6use flint_render::{Camera, DebugMode, HeadlessContext, SceneRenderer};
7use flint_scene::load_scene;
8use flint_schema::SchemaRegistry;
9use std::path::Path;
10
11pub struct RenderArgs {
12    pub scene: String,
13    pub output: String,
14    pub width: u32,
15    pub height: u32,
16    pub schemas: String,
17    pub distance: Option<f32>,
18    pub yaw: Option<f32>,
19    pub pitch: Option<f32>,
20    pub target: Option<[f32; 3]>,
21    pub fov: Option<f32>,
22    pub no_grid: bool,
23    pub debug_mode: Option<String>,
24    pub wireframe_overlay: bool,
25    pub show_normals: bool,
26    pub no_tonemapping: bool,
27    pub shadows: bool,
28    pub shadow_resolution: u32,
29}
30
31pub fn run(args: RenderArgs) -> Result<()> {
32    // Load schemas
33    let registry = if Path::new(&args.schemas).exists() {
34        SchemaRegistry::load_from_directory(&args.schemas).context("Failed to load schemas")?
35    } else {
36        println!("Warning: Schemas directory not found: {}", args.schemas);
37        SchemaRegistry::new()
38    };
39
40    // Load scene
41    let (world, scene_file) =
42        load_scene(&args.scene, &registry).context("Failed to load scene")?;
43    println!("Loaded scene: {}", scene_file.scene.name);
44    println!("Entities: {}", world.entity_count());
45
46    // Create headless context
47    let ctx = pollster::block_on(HeadlessContext::new(args.width, args.height))
48        .context("Failed to create headless render context")?;
49
50    // Configure camera
51    let mut camera = Camera::new();
52    camera.aspect = ctx.aspect_ratio();
53
54    if let Some(d) = args.distance {
55        camera.distance = d;
56    }
57    if let Some(y) = args.yaw {
58        camera.yaw = y.to_radians();
59    }
60    if let Some(p) = args.pitch {
61        camera.pitch = p.to_radians();
62    }
63    if let Some(t) = args.target {
64        camera.target = Vec3::new(t[0], t[1], t[2]);
65    }
66    if let Some(f) = args.fov {
67        camera.fov = f;
68    }
69    camera.update_orbit();
70
71    // Create scene renderer
72    let mut renderer = SceneRenderer::new_headless(&ctx.device, &ctx.queue, ctx.format);
73    if args.no_grid {
74        renderer.disable_grid();
75    }
76
77    // Load models from the scene
78    let scene_dir = Path::new(&args.scene)
79        .parent()
80        .unwrap_or_else(|| Path::new("."));
81
82    for entity in world.all_entities() {
83        let model_asset = world
84            .get_components(entity.id)
85            .and_then(|components| components.get("model").cloned())
86            .and_then(|model| {
87                model
88                    .get("asset")
89                    .and_then(|v| v.as_str().map(String::from))
90            });
91
92        if let Some(asset_name) = model_asset {
93            if renderer.mesh_cache().contains(&asset_name) {
94                continue;
95            }
96
97            let model_path = scene_dir.join("models").join(format!("{}.glb", asset_name));
98
99            if model_path.exists() {
100                match import_gltf(&model_path) {
101                    Ok(import_result) => {
102                        println!(
103                            "Loaded model: {} ({} meshes, {} materials)",
104                            asset_name,
105                            import_result.meshes.len(),
106                            import_result.materials.len()
107                        );
108                        renderer.load_model(&ctx.device, &ctx.queue, &asset_name, &import_result);
109                    }
110                    Err(e) => {
111                        eprintln!("Failed to load model '{}': {:?}", asset_name, e);
112                    }
113                }
114            } else {
115                eprintln!(
116                    "Model file not found: {} (tried {})",
117                    asset_name,
118                    model_path.display()
119                );
120            }
121        }
122    }
123
124    // Apply debug state
125    if let Some(mode_str) = &args.debug_mode {
126        let mode = match mode_str.as_str() {
127            "wireframe" => DebugMode::WireframeOnly,
128            "normals" => DebugMode::Normals,
129            "depth" => DebugMode::Depth,
130            "uv" => DebugMode::UvChecker,
131            "unlit" => DebugMode::Unlit,
132            "metalrough" => DebugMode::MetallicRoughness,
133            _ => DebugMode::Pbr,
134        };
135        renderer.set_debug_mode(mode);
136    }
137    if args.wireframe_overlay {
138        renderer.toggle_wireframe_overlay();
139    }
140    if args.show_normals {
141        renderer.toggle_normal_arrows();
142    }
143    if args.no_tonemapping {
144        renderer.set_tonemapping(false);
145    }
146    if args.shadows {
147        renderer.set_shadows(true);
148    }
149    if args.shadow_resolution != 1024 {
150        renderer.set_shadow_resolution(&ctx.device, args.shadow_resolution);
151    }
152
153    renderer.update_from_world(&world, &ctx.device);
154
155    // Render
156    renderer.render_to(
157        &ctx.device,
158        &ctx.queue,
159        &ctx.depth_view,
160        &camera,
161        &ctx.color_view,
162    );
163
164    // Read back pixels
165    let pixels = pollster::block_on(ctx.read_pixels()).context("Failed to read rendered pixels")?;
166
167    // Encode as PNG
168    let img = image::RgbaImage::from_raw(args.width, args.height, pixels)
169        .context("Failed to create image from pixel data")?;
170    img.save(&args.output)
171        .context(format!("Failed to save image to {}", args.output))?;
172
173    println!(
174        "Rendered {}x{} image to {}",
175        args.width, args.height, args.output
176    );
177
178    Ok(())
179}
