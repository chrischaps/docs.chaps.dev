1//! Headless scene-to-PNG render command
2
3use anyhow::{Context, Result};
4use flint_core::Vec3;
5use flint_render::{Camera, HeadlessContext, SceneRenderer};
6use flint_scene::load_scene;
7use flint_schema::SchemaRegistry;
8use std::path::Path;
9
10pub struct RenderArgs {
11    pub scene: String,
12    pub output: String,
13    pub width: u32,
14    pub height: u32,
15    pub schemas: String,
16    pub distance: Option<f32>,
17    pub yaw: Option<f32>,
18    pub pitch: Option<f32>,
19    pub target: Option<[f32; 3]>,
20    pub fov: Option<f32>,
21    pub no_grid: bool,
22}
23
24pub fn run(args: RenderArgs) -> Result<()> {
25    // Load schemas
26    let registry = if Path::new(&args.schemas).exists() {
27        SchemaRegistry::load_from_directory(&args.schemas).context("Failed to load schemas")?
28    } else {
29        println!("Warning: Schemas directory not found: {}", args.schemas);
30        SchemaRegistry::new()
31    };
32
33    // Load scene
34    let (world, scene_file) =
35        load_scene(&args.scene, &registry).context("Failed to load scene")?;
36    println!("Loaded scene: {}", scene_file.scene.name);
37    println!("Entities: {}", world.entity_count());
38
39    // Create headless context
40    let ctx = pollster::block_on(HeadlessContext::new(args.width, args.height))
41        .context("Failed to create headless render context")?;
42
43    // Configure camera
44    let mut camera = Camera::new();
45    camera.aspect = ctx.aspect_ratio();
46
47    if let Some(d) = args.distance {
48        camera.distance = d;
49    }
50    if let Some(y) = args.yaw {
51        camera.yaw = y.to_radians();
52    }
53    if let Some(p) = args.pitch {
54        camera.pitch = p.to_radians();
55    }
56    if let Some(t) = args.target {
57        camera.target = Vec3::new(t[0], t[1], t[2]);
58    }
59    if let Some(f) = args.fov {
60        camera.fov = f;
61    }
62    camera.update_orbit();
63
64    // Create scene renderer
65    let mut renderer = SceneRenderer::new_headless(&ctx.device, ctx.format);
66    if args.no_grid {
67        renderer.disable_grid();
68    }
69    renderer.update_from_world(&world, &ctx.device);
70
71    // Render
72    renderer.render_to(
73        &ctx.device,
74        &ctx.queue,
75        &ctx.depth_view,
76        &camera,
77        &ctx.color_view,
78    );
79
80    // Read back pixels
81    let pixels = pollster::block_on(ctx.read_pixels()).context("Failed to read rendered pixels")?;
82
83    // Encode as PNG
84    let img = image::RgbaImage::from_raw(args.width, args.height, pixels)
85        .context("Failed to create image from pixel data")?;
86    img.save(&args.output)
87        .context(format!("Failed to save image to {}", args.output))?;
88
89    println!(
90        "Rendered {}x{} image to {}",
91        args.width, args.height, args.output
92    );
93
94    Ok(())
95}
