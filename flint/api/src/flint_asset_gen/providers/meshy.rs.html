1//! Meshy 3D model generation provider
2//!
3//! Generates 3D models via the Meshy v2 text-to-3d API.
4//! Model generation is long-running (~2-5 min), so `generate()` polls
5//! with progress output, and `submit_job()` is available for async use.

16const DEFAULT_MESHY_URL: &str = "https://api.meshy.ai/openapi/v2/text-to-3d";
17const POLL_INTERVAL_SECS: u64 = 10;
18const REQUEST_TIMEOUT_SECS: u64 = 60;
19const MAX_RETRIES: usize = 3;
20const RETRY_BASE_DELAY_MS: u64 = 500;
21const MAX_POLL_ATTEMPTS: u32 = 180;

30    /// Create a new MeshyProvider from config
31    pub fn from_config(config: &FlintConfig) -> Result<Self> {
32        let api_key = config
33            .api_key("meshy")
34            .ok_or_else(|| {
35                FlintError::GenerationError(
36                    "Meshy API key not configured. Set FLINT_MESHY_API_KEY or add to .flint/config.toml".to_string(),
37                )
38            })?
39            .to_string();
40
41        let api_url = config
42            .api_url("meshy")
43            .unwrap_or(DEFAULT_MESHY_URL)
44            .to_string();
45
46        Ok(Self { api_key, api_url })
47    }

49    /// Submit a text-to-3d task and return the task ID
50    fn submit_task(&self, prompt: &str, negative: Option<&str>) -> Result<String> {
51        let mut payload = serde_json::json!({
52            "mode": "refine",
53            "prompt": prompt,
54            "should_remesh": true
55        });
56
57        if let Some(neg) = negative {
58            payload["negative_prompt"] = serde_json::json!(neg);
59        }
60
61        let response = self.post_json_with_retry(&self.api_url, &payload)?;
62
63        response
64            .get("result")
65            .and_then(|r| r.as_str())
66            .map(|s| s.to_string())
67            .ok_or_else(|| {
68                FlintError::GenerationError(format!(
69                    "Unexpected Meshy submit response: {}",
70                    serde_json::to_string_pretty(&response).unwrap_or_default()
71                ))
72            })
73    }
74
75    /// Poll task status
76    fn poll_task(&self, task_id: &str) -> Result<MeshyTaskStatus> {
77        let url = format!("{}/{}", self.api_url, task_id);
78
79        let response = self.get_json_with_retry(&url)?;
80
81        let status = response
82            .get("status")
83            .and_then(|s| s.as_str())
84            .unwrap_or("UNKNOWN");
85
86        let progress = response
87            .get("progress")
88            .and_then(|p| p.as_u64())
89            .unwrap_or(0) as u8;
90
91        match status {
92            "SUCCEEDED" => {
93                let model_url = response
94                    .get("model_urls")
95                    .and_then(|u| u.get("glb"))
96                    .and_then(|u| u.as_str())
97                    .map(|s| s.to_string());
98
99                Ok(MeshyTaskStatus::Complete { model_url })
100            }
101            "FAILED" | "EXPIRED" => {
102                let msg = response
103                    .get("task_error")
104                    .and_then(|e| e.get("message"))
105                    .and_then(|m| m.as_str())
106                    .unwrap_or("Unknown error")
107                    .to_string();
108                Ok(MeshyTaskStatus::Failed(msg))
109            }
110            _ => Ok(MeshyTaskStatus::Processing(progress)),
111        }
112    }

114    /// Download a GLB file from URL
115    fn download_glb(&self, url: &str, output_path: &Path) -> Result<()> {
116        let bytes = self.download_bytes_with_retry(url)?;
117        std::fs::write(output_path, &bytes)?;
118        Ok(())
119    }

121    fn post_json_with_retry(
122        &self,
123        url: &str,
124        payload: &serde_json::Value,
125    ) -> Result<serde_json::Value> {
126        for attempt in 0..MAX_RETRIES {
127            let agent = build_agent();
128            let response = agent
129                .post(url)
130                .header("Authorization", &format!("Bearer {}", self.api_key))
131                .header("Content-Type", "application/json")
132                .send_json(payload);
133
134            match response {
135                Ok(mut ok) => {
136                    return ok.body_mut().read_json().map_err(|e| {
137                        FlintError::GenerationError(format!(
138                            "Failed to parse Meshy response: {}",
139                            e
140                        ))
141                    });
142                }
143                Err(e) => {
144                    if attempt + 1 < MAX_RETRIES && is_retryable_error(&e) {
145                        sleep_backoff(attempt);
146                        continue;
147                    }
148                    return Err(FlintError::GenerationError(format!(
149                        "Meshy API request failed: {}",
150                        e
151                    )));
152                }
153            }
154        }
155
156        Err(FlintError::GenerationError(
157            "Meshy API request failed after retries".to_string(),
158        ))
159    }
160
161    fn get_json_with_retry(&self, url: &str) -> Result<serde_json::Value> {
162        for attempt in 0..MAX_RETRIES {
163            let agent = build_agent();
164            let response = agent
165                .get(url)
166                .header("Authorization", &format!("Bearer {}", self.api_key))
167                .call();
168
169            match response {
170                Ok(mut ok) => {
171                    return ok.body_mut().read_json().map_err(|e| {
172                        FlintError::GenerationError(format!(
173                            "Failed to parse poll response: {}",
174                            e
175                        ))
176                    });
177                }
178                Err(e) => {
179                    if attempt + 1 < MAX_RETRIES && is_retryable_error(&e) {
180                        sleep_backoff(attempt);
181                        continue;
182                    }
183                    return Err(FlintError::GenerationError(format!("Meshy poll failed: {}", e)));
184                }
185            }
186        }
187
188        Err(FlintError::GenerationError(
189            "Meshy poll failed after retries".to_string(),
190        ))
191    }
192
193    fn download_bytes_with_retry(&self, url: &str) -> Result<Vec<u8>> {
194        for attempt in 0..MAX_RETRIES {
195            let agent = build_agent();
196            let response = agent.get(url).call();
197
198            match response {
199                Ok(ok) => {
200                    let mut reader = ok.into_body().into_reader();
201                    let mut bytes = Vec::new();
202                    std::io::Read::read_to_end(&mut reader, &mut bytes).map_err(|e| {
203                        FlintError::GenerationError(format!("Failed to read model data: {}", e))
204                    })?;
205                    return Ok(bytes);
206                }
207                Err(e) => {
208                    if attempt + 1 < MAX_RETRIES && is_retryable_error(&e) {
209                        sleep_backoff(attempt);
210                        continue;
211                    }
212                    return Err(FlintError::GenerationError(format!(
213                        "Failed to download model: {}",
214                        e
215                    )));
216                }
217            }
218        }
219
220        Err(FlintError::GenerationError(
221            "Model download failed after retries".to_string(),
222        ))
223    }

232impl GenerationProvider for MeshyProvider {
233    fn name(&self) -> &str {
234        "meshy"
235    }
236
237    fn supported_kinds(&self) -> Vec<AssetKind> {
238        vec![AssetKind::Model]
239    }
240
241    fn health_check(&self) -> Result<ProviderStatus> {
242        if self.api_key.is_empty() {
243            return Ok(ProviderStatus::NoApiKey);
244        }
245        Ok(ProviderStatus::Available)
246    }
247
248    fn generate(
249        &self,
250        request: &GenerateRequest,
251        style: Option<&StyleGuide>,
252        output_dir: &Path,
253    ) -> Result<GenerateResult> {
254        let start = std::time::Instant::now();
255        let prompt = self.build_prompt(request, style);
256        let negative = style.and_then(|s| s.negative()).map(|s| s.to_string());
257
258        std::fs::create_dir_all(output_dir)?;
259
260        // Submit task
261        let task_id = self.submit_task(&prompt, negative.as_deref())?;
262        eprintln!("  Submitted job: {}", task_id);
263
264        // Poll until complete
265        let mut poll_attempts = 0u32;
266        loop {
267            poll_attempts += 1;
268            if poll_attempts > MAX_POLL_ATTEMPTS {
269                return Err(FlintError::GenerationError(format!(
270                    "Meshy generation timed out after {} poll attempts",
271                    MAX_POLL_ATTEMPTS
272                )));
273            }
274
275            std::thread::sleep(std::time::Duration::from_secs(POLL_INTERVAL_SECS));
276
277            match self.poll_task(&task_id)? {
278                MeshyTaskStatus::Processing(progress) => {
279                    eprintln!("  Processing... {}%", progress);
280                }
281                MeshyTaskStatus::Complete { model_url } => {
282                    let url = model_url.ok_or_else(|| {
283                        FlintError::GenerationError("No GLB URL in completion response".to_string())
284                    })?;
285
286                    let output_path = output_dir.join(format!("{}.glb", request.name));
287                    self.download_glb(&url, &output_path)?;
288
289                    let duration = start.elapsed().as_secs_f64();
290                    let hash = flint_core::ContentHash::from_file(&output_path)
291                        .map(|h| h.to_prefixed_hex())
292                        .ok();
293
294                    let mut metadata = HashMap::new();
295                    metadata.insert("task_id".to_string(), task_id);
296
297                    return Ok(GenerateResult {
298                        output_path: output_path.to_string_lossy().to_string(),
299                        prompt_used: prompt,
300                        provider: "meshy".to_string(),
301                        duration_secs: duration,
302                        content_hash: hash,
303                        metadata,
304                    });
305                }
306                MeshyTaskStatus::Failed(msg) => {
307                    return Err(FlintError::GenerationError(format!(
308                        "Meshy generation failed: {}",
309                        msg
310                    )));
311                }
312            }
313        }
314    }
315
316    fn submit_job(
317        &self,
318        request: &GenerateRequest,
319        style: Option<&StyleGuide>,
320    ) -> Result<GenerationJob> {
321        let prompt = self.build_prompt(request, style);
322        let negative = style.and_then(|s| s.negative()).map(|s| s.to_string());
323        let task_id = self.submit_task(&prompt, negative.as_deref())?;
324
325        let mut job = GenerationJob::new("meshy", &request.name);
326        job.remote_id = Some(task_id);
327        job.prompt = Some(prompt);
328        Ok(job)
329    }
330
331    fn poll_job(&self, job: &GenerationJob) -> Result<JobPollResult> {
332        let remote_id = job.remote_id.as_deref().ok_or_else(|| {
333            FlintError::GenerationError("Job has no remote ID".to_string())
334        })?;
335
336        match self.poll_task(remote_id)? {
337            MeshyTaskStatus::Processing(p) => Ok(JobPollResult::Processing(p)),
338            MeshyTaskStatus::Complete { .. } => Ok(JobPollResult::Complete),
339            MeshyTaskStatus::Failed(msg) => Ok(JobPollResult::Failed(msg)),
340        }
341    }
342
343    fn download_result(&self, job: &GenerationJob, output_dir: &Path) -> Result<GenerateResult> {
344        let remote_id = job.remote_id.as_deref().ok_or_else(|| {
345            FlintError::GenerationError("Job has no remote ID".to_string())
346        })?;
347
348        // Poll one more time to get the model URL
349        match self.poll_task(remote_id)? {
350            MeshyTaskStatus::Complete { model_url } => {
351                let url = model_url.ok_or_else(|| {
352                    FlintError::GenerationError("No GLB URL in completion response".to_string())
353                })?;
354
355                std::fs::create_dir_all(output_dir)?;
356                let output_path = output_dir.join(format!("{}.glb", job.asset_name));
357                self.download_glb(&url, &output_path)?;
358
359                let hash = flint_core::ContentHash::from_file(&output_path)
360                    .map(|h| h.to_prefixed_hex())
361                    .ok();
362
363                Ok(GenerateResult {
364                    output_path: output_path.to_string_lossy().to_string(),
365                    prompt_used: job.prompt.clone().unwrap_or_default(),
366                    provider: "meshy".to_string(),
367                    duration_secs: 0.0,
368                    content_hash: hash,
369                    metadata: HashMap::new(),
370                })
371            }
372            MeshyTaskStatus::Processing(_) => Err(FlintError::GenerationError(
373                "Job not yet complete".to_string(),
374            )),
375            MeshyTaskStatus::Failed(msg) => Err(FlintError::GenerationError(format!(
376                "Job failed: {}",
377                msg
378            ))),
379        }
380    }
381
382    fn build_prompt(&self, request: &GenerateRequest, style: Option<&StyleGuide>) -> String {
383        match style {
384            Some(s) => s.enrich_prompt(&request.description),
385            None => request.description.clone(),
386        }
387    }
388}

390fn build_agent() -> ureq::Agent {
391    let config = ureq::Agent::config_builder()
392        .timeout_global(Some(Duration::from_secs(REQUEST_TIMEOUT_SECS)))
393        .build();
394    config.into()
395}

397fn is_retryable_error(e: &ureq::Error) -> bool {
398    match e {
399        ureq::Error::Timeout(_)
400        | ureq::Error::Io(_)
401        | ureq::Error::ConnectionFailed
402        | ureq::Error::HostNotFound => true,
403        ureq::Error::StatusCode(code) => matches!(code, 429 | 500 | 502 | 503 | 504),
404        _ => false,
405    }
406}
407
408fn sleep_backoff(attempt: usize) {
409    let delay_ms = RETRY_BASE_DELAY_MS.saturating_mul(1u64 << attempt);
410    std::thread::sleep(Duration::from_millis(delay_ms));
411}

413/// Parse a Meshy submit response for testing
414pub fn parse_meshy_submit(json: &str) -> Result<String> {
415    let response: serde_json::Value = serde_json::from_str(json)
416        .map_err(|e| FlintError::GenerationError(format!("Invalid JSON: {}", e)))?;
417
418    response
419        .get("result")
420        .and_then(|r| r.as_str())
421        .map(|s| s.to_string())
422        .ok_or_else(|| FlintError::GenerationError("No task ID in response".to_string()))
423}

425/// Parse a Meshy poll response for testing
426pub fn parse_meshy_poll(json: &str) -> Result<(String, u8, Option<String>)> {
427    let response: serde_json::Value = serde_json::from_str(json)
428        .map_err(|e| FlintError::GenerationError(format!("Invalid JSON: {}", e)))?;
429
430    let status = response
431        .get("status")
432        .and_then(|s| s.as_str())
433        .unwrap_or("UNKNOWN")
434        .to_string();
435
436    let progress = response
437        .get("progress")
438        .and_then(|p| p.as_u64())
439        .unwrap_or(0) as u8;
440
441    let model_url = response
442        .get("model_urls")
443        .and_then(|u| u.get("glb"))
444        .and_then(|u| u.as_str())
445        .map(|s| s.to_string());
446
447    Ok((status, progress, model_url))
448}
