23impl GenerationProvider for MockProvider {
24    fn name(&self) -> &str {
25        "mock"
26    }
27
28    fn supported_kinds(&self) -> Vec<AssetKind> {
29        vec![AssetKind::Texture, AssetKind::Model, AssetKind::Audio]
30    }
31
32    fn health_check(&self) -> Result<ProviderStatus> {
33        Ok(ProviderStatus::Available)
34    }
35
36    fn generate(
37        &self,
38        request: &GenerateRequest,
39        style: Option<&StyleGuide>,
40        output_dir: &Path,
41    ) -> Result<GenerateResult> {
42        let start = std::time::Instant::now();
43        let prompt = self.build_prompt(request, style);
44
45        std::fs::create_dir_all(output_dir)?;
46
47        let output_path = match request.kind {
48            AssetKind::Texture => {
49                let params = request
50                    .texture_params
51                    .as_ref()
52                    .cloned()
53                    .unwrap_or_default();
54                generate_solid_png(output_dir, &request.name, params.width, params.height)?
55            }
56            AssetKind::Model => generate_minimal_glb(output_dir, &request.name)?,
57            AssetKind::Audio => {
58                let params = request
59                    .audio_params
60                    .as_ref()
61                    .cloned()
62                    .unwrap_or_default();
63                generate_silence_wav(output_dir, &request.name, params.duration)?
64            }
65        };
66
67        let duration = start.elapsed().as_secs_f64();
68
69        // Compute content hash
70        let hash = flint_core::ContentHash::from_file(&output_path)
71            .map(|h| h.to_prefixed_hex())
72            .ok();
73
74        Ok(GenerateResult {
75            output_path: output_path.to_string_lossy().to_string(),
76            prompt_used: prompt,
77            provider: "mock".to_string(),
78            duration_secs: duration,
79            content_hash: hash,
80            metadata: HashMap::new(),
81        })
82    }
83
84    fn submit_job(
85        &self,
86        request: &GenerateRequest,
87        _style: Option<&StyleGuide>,
88    ) -> Result<GenerationJob> {
89        let mut job = GenerationJob::new("mock", &request.name);
90        job.prompt = Some(request.description.clone());
91        Ok(job)
92    }
93
94    fn poll_job(&self, _job: &GenerationJob) -> Result<JobPollResult> {
95        // Mock jobs complete instantly
96        Ok(JobPollResult::Complete)
97    }
98
99    fn download_result(&self, job: &GenerationJob, output_dir: &Path) -> Result<GenerateResult> {
100        // For mock, just generate inline
101        let request = GenerateRequest {
102            name: job.asset_name.clone(),
103            description: job.prompt.clone().unwrap_or_default(),
104            kind: AssetKind::Texture,
105            texture_params: None,
106            model_params: None,
107            audio_params: None,
108            tags: vec![],
109        };
110        self.generate(&request, None, output_dir)
111    }
112
113    fn build_prompt(&self, request: &GenerateRequest, style: Option<&StyleGuide>) -> String {
114        match style {
115            Some(s) => s.enrich_prompt(&request.description),
116            None => request.description.clone(),
117        }
118    }
119}

121/// Generate a solid-color PNG texture
122fn generate_solid_png(
123    output_dir: &Path,
124    name: &str,
125    width: u32,
126    height: u32,
127) -> Result<std::path::PathBuf> {
128    // Generate a warm, earthy color from the name hash for visual interest
129    let hash_val = name.bytes().fold(0u32, |acc, b| acc.wrapping_mul(31).wrapping_add(b as u32));
130    let r = ((hash_val >> 16) & 0xFF) as u8;
131    let g = ((hash_val >> 8) & 0xFF) as u8;
132    let b = (hash_val & 0xFF) as u8;
133
134    let mut img_data = Vec::with_capacity((width * height * 4) as usize);
135    for _ in 0..(width * height) {
136        img_data.extend_from_slice(&[r, g, b, 255]);
137    }
138
139    let path = output_dir.join(format!("{}.png", name));
140    let img = image::RgbaImage::from_raw(width, height, img_data).ok_or_else(|| {
141        FlintError::GenerationError("Failed to create image buffer".to_string())
142    })?;
143    img.save(&path).map_err(|e| {
144        FlintError::GenerationError(format!("Failed to save PNG: {}", e))
145    })?;
146
147    Ok(path)
148}

150/// Generate a minimal valid GLB file (single triangle)
151fn generate_minimal_glb(output_dir: &Path, name: &str) -> Result<std::path::PathBuf> {
152    // Minimal valid glTF 2.0 binary with a single triangle
153    // This is the smallest possible valid GLB that importers will accept
154    let json = serde_json::json!({
155        "asset": { "version": "2.0", "generator": "flint-mock" },
156        "scene": 0,
157        "scenes": [{ "nodes": [0] }],
158        "nodes": [{ "mesh": 0 }],
159        "meshes": [{
160            "primitives": [{
161                "attributes": { "POSITION": 0 },
162                "indices": 1
163            }]
164        }],
165        "accessors": [
166            {
167                "bufferView": 0,
168                "componentType": 5126,
169                "count": 3,
170                "type": "VEC3",
171                "max": [1.0, 1.0, 0.0],
172                "min": [-1.0, 0.0, 0.0]
173            },
174            {
175                "bufferView": 1,
176                "componentType": 5123,
177                "count": 3,
178                "type": "SCALAR",
179                "max": [2],
180                "min": [0]
181            }
182        ],
183        "bufferViews": [
184            { "buffer": 0, "byteOffset": 0, "byteLength": 36, "target": 34962 },
185            { "buffer": 0, "byteOffset": 36, "byteLength": 6, "target": 34963 }
186        ],
187        "buffers": [{ "byteLength": 44 }]
188    });
189
190    let json_str = serde_json::to_string(&json).map_err(|e| {
191        FlintError::GenerationError(format!("Failed to serialize GLB JSON: {}", e))
192    })?;
193
194    // Pad JSON to 4-byte alignment
195    let json_bytes = json_str.as_bytes();
196    let json_padded_len = (json_bytes.len() + 3) & !3;
197    let mut json_padded = json_bytes.to_vec();
198    json_padded.resize(json_padded_len, b' ');
199
200    // Binary buffer: 3 vertices (3 * 12 bytes) + 3 indices (3 * 2 bytes) + 2 padding
201    #[allow(clippy::approx_constant)]
202    let vertices: [f32; 9] = [
203        -1.0, 0.0, 0.0, // v0
204        1.0, 0.0, 0.0, // v1
205        0.0, 1.0, 0.0, // v2
206    ];
207    let indices: [u16; 3] = [0, 1, 2];
208
209    let mut bin_data = Vec::new();
210    for v in &vertices {
211        bin_data.extend_from_slice(&v.to_le_bytes());
212    }
213    for i in &indices {
214        bin_data.extend_from_slice(&i.to_le_bytes());
215    }
216    // Pad to 4-byte alignment
217    let bin_padded_len = (bin_data.len() + 3) & !3;
218    bin_data.resize(bin_padded_len, 0);
219
220    // GLB header
221    let total_len =
222        12 + 8 + json_padded.len() as u32 + 8 + bin_data.len() as u32;
223
224    let path = output_dir.join(format!("{}.glb", name));
225    let mut file = std::fs::File::create(&path)?;
226    use std::io::Write;
227
228    // Header
229    file.write_all(b"glTF")?; // magic
230    file.write_all(&2u32.to_le_bytes())?; // version
231    file.write_all(&total_len.to_le_bytes())?; // length
232
233    // JSON chunk
234    file.write_all(&(json_padded.len() as u32).to_le_bytes())?;
235    file.write_all(&0x4E4F534Au32.to_le_bytes())?; // "JSON"
236    file.write_all(&json_padded)?;
237
238    // BIN chunk
239    file.write_all(&(bin_data.len() as u32).to_le_bytes())?;
240    file.write_all(&0x004E4942u32.to_le_bytes())?; // "BIN\0"
241    file.write_all(&bin_data)?;
242
243    Ok(path)
244}

246/// Generate a WAV file with silence
247fn generate_silence_wav(
248    output_dir: &Path,
249    name: &str,
250    duration_secs: f64,
251) -> Result<std::path::PathBuf> {
252    let sample_rate: u32 = 44100;
253    let num_channels: u16 = 1;
254    let bits_per_sample: u16 = 16;
255    let num_samples = (sample_rate as f64 * duration_secs) as u32;
256    let data_size = num_samples * (bits_per_sample / 8) as u32 * num_channels as u32;
257
258    let path = output_dir.join(format!("{}.wav", name));
259    let mut file = std::fs::File::create(&path)?;
260    use std::io::Write;
261
262    // RIFF header
263    file.write_all(b"RIFF")?;
264    file.write_all(&(36 + data_size).to_le_bytes())?;
265    file.write_all(b"WAVE")?;
266
267    // fmt chunk
268    file.write_all(b"fmt ")?;
269    file.write_all(&16u32.to_le_bytes())?; // chunk size
270    file.write_all(&1u16.to_le_bytes())?; // PCM format
271    file.write_all(&num_channels.to_le_bytes())?;
272    file.write_all(&sample_rate.to_le_bytes())?;
273    let byte_rate = sample_rate * num_channels as u32 * (bits_per_sample / 8) as u32;
274    file.write_all(&byte_rate.to_le_bytes())?;
275    let block_align = num_channels * (bits_per_sample / 8);
276    file.write_all(&block_align.to_le_bytes())?;
277    file.write_all(&bits_per_sample.to_le_bytes())?;
278
279    // data chunk
280    file.write_all(b"data")?;
281    file.write_all(&data_size.to_le_bytes())?;
282
283    // Write silence (zeros)
284    let silence = vec![0u8; data_size as usize];
285    file.write_all(&silence)?;
286
287    Ok(path)
288}
