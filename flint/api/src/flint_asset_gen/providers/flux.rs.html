1//! Flux texture generation provider (fal.ai)
2//!
3//! Generates textures via the Flux image generation API.
4//! Textures are fast (~10s) so `generate()` blocks synchronously.

15const DEFAULT_FLUX_URL: &str = "https://queue.fal.run/fal-ai/flux/dev";
16const REQUEST_TIMEOUT_SECS: u64 = 60;
17const MAX_RETRIES: usize = 3;
18const RETRY_BASE_DELAY_MS: u64 = 500;

20/// Flux provider for AI texture generation via fal.ai
21pub struct FluxProvider {
22    api_key: String,
23    api_url: String,
24}

26impl FluxProvider {
27    /// Create a new FluxProvider from config
28    pub fn from_config(config: &FlintConfig) -> Result<Self> {
29        let api_key = config
30            .api_key("flux")
31            .ok_or_else(|| {
32                FlintError::GenerationError(
33                    "Flux API key not configured. Set FLINT_FLUX_API_KEY or add to .flint/config.toml".to_string(),
34                )
35            })?
36            .to_string();
37
38        let api_url = config
39            .api_url("flux")
40            .unwrap_or(DEFAULT_FLUX_URL)
41            .to_string();
42
43        Ok(Self { api_key, api_url })
44    }
45
46    /// Submit a request to fal.ai and poll for completion
47    fn submit_and_wait(
48        &self,
49        prompt: &str,
50        width: u32,
51        height: u32,
52        seed: Option<u64>,
53    ) -> Result<serde_json::Value> {
54        let mut payload = serde_json::json!({
55            "prompt": prompt,
56            "image_size": {
57                "width": width,
58                "height": height
59            },
60            "num_images": 1,
61            "enable_safety_checker": false
62        });
63
64        if let Some(s) = seed {
65            payload["seed"] = serde_json::json!(s);
66        }
67
68        self.post_json_with_retry(&payload)
69    }
70
71    /// Download an image from a URL to a local file
72    fn download_image(&self, url: &str, output_path: &Path) -> Result<()> {
73        let bytes = self.download_bytes_with_retry(url)?;
74        std::fs::write(output_path, &bytes)?;
75        Ok(())
76    }
77
78    fn post_json_with_retry(&self, payload: &serde_json::Value) -> Result<serde_json::Value> {
79        for attempt in 0..MAX_RETRIES {
80            let agent = build_agent();
81            let response = agent
82                .post(&self.api_url)
83                .header("Authorization", &format!("Key {}", self.api_key))
84                .header("Content-Type", "application/json")
85                .send_json(payload);
86
87            match response {
88                Ok(mut ok) => {
89                    return ok.body_mut().read_json().map_err(|e| {
90                        FlintError::GenerationError(format!(
91                            "Failed to parse Flux response: {}",
92                            e
93                        ))
94                    });
95                }
96                Err(e) => {
97                    if attempt + 1 < MAX_RETRIES && is_retryable_error(&e) {
98                        sleep_backoff(attempt);
99                        continue;
100                    }
101                    return Err(FlintError::GenerationError(format!(
102                        "Flux API request failed: {}",
103                        e
104                    )));
105                }
106            }
107        }
108
109        Err(FlintError::GenerationError(
110            "Flux API request failed after retries".to_string(),
111        ))
112    }
113
114    fn download_bytes_with_retry(&self, url: &str) -> Result<Vec<u8>> {
115        for attempt in 0..MAX_RETRIES {
116            let agent = build_agent();
117            let response = agent.get(url).call();
118
119            match response {
120                Ok(ok) => {
121                    let mut reader = ok.into_body().into_reader();
122                    let mut bytes = Vec::new();
123                    std::io::Read::read_to_end(&mut reader, &mut bytes).map_err(|e| {
124                        FlintError::GenerationError(format!("Failed to read image data: {}", e))
125                    })?;
126                    return Ok(bytes);
127                }
128                Err(e) => {
129                    if attempt + 1 < MAX_RETRIES && is_retryable_error(&e) {
130                        sleep_backoff(attempt);
131                        continue;
132                    }
133                    return Err(FlintError::GenerationError(format!(
134                        "Failed to download image: {}",
135                        e
136                    )));
137                }
138            }
139        }
140
141        Err(FlintError::GenerationError(
142            "Image download failed after retries".to_string(),
143        ))
144    }
145}
146
147fn build_agent() -> ureq::Agent {
148    let config = ureq::Agent::config_builder()
149        .timeout_global(Some(Duration::from_secs(REQUEST_TIMEOUT_SECS)))
150        .build();
151    config.into()
152}
153
154fn is_retryable_error(e: &ureq::Error) -> bool {
155    match e {
156        ureq::Error::Timeout(_)
157        | ureq::Error::Io(_)
158        | ureq::Error::ConnectionFailed
159        | ureq::Error::HostNotFound => true,
160        ureq::Error::StatusCode(code) => matches!(code, 429 | 500 | 502 | 503 | 504),
161        _ => false,
162    }
163}
164
165fn sleep_backoff(attempt: usize) {
166    let delay_ms = RETRY_BASE_DELAY_MS.saturating_mul(1u64 << attempt);
167    std::thread::sleep(Duration::from_millis(delay_ms));
168}

170impl GenerationProvider for FluxProvider {
171    fn name(&self) -> &str {
172        "flux"
173    }
174
175    fn supported_kinds(&self) -> Vec<AssetKind> {
176        vec![AssetKind::Texture]
177    }
178
179    fn health_check(&self) -> Result<ProviderStatus> {
180        if self.api_key.is_empty() {
181            return Ok(ProviderStatus::NoApiKey);
182        }
183        Ok(ProviderStatus::Available)
184    }
185
186    fn generate(
187        &self,
188        request: &GenerateRequest,
189        style: Option<&StyleGuide>,
190        output_dir: &Path,
191    ) -> Result<GenerateResult> {
192        let start = std::time::Instant::now();
193        let prompt = self.build_prompt(request, style);
194
195        let params = request
196            .texture_params
197            .as_ref()
198            .cloned()
199            .unwrap_or_default();
200
201        std::fs::create_dir_all(output_dir)?;
202
203        // Submit and wait for result
204        let response = self.submit_and_wait(&prompt, params.width, params.height, params.seed)?;
205
206        // Extract image URL from response
207        let image_url = response
208            .get("images")
209            .and_then(|imgs| imgs.as_array())
210            .and_then(|arr| arr.first())
211            .and_then(|img| img.get("url"))
212            .and_then(|u| u.as_str())
213            .ok_or_else(|| {
214                FlintError::GenerationError(format!(
215                    "Unexpected Flux response format: {}",
216                    serde_json::to_string_pretty(&response).unwrap_or_default()
217                ))
218            })?;
219
220        // Download the image
221        let output_path = output_dir.join(format!("{}.png", request.name));
222        self.download_image(image_url, &output_path)?;
223
224        let duration = start.elapsed().as_secs_f64();
225
226        // Compute content hash
227        let hash = flint_core::ContentHash::from_file(&output_path)
228            .map(|h| h.to_prefixed_hex())
229            .ok();
230
231        let mut metadata = HashMap::new();
232        if let Some(seed) = response.get("seed").and_then(|s| s.as_u64()) {
233            metadata.insert("seed".to_string(), seed.to_string());
234        }
235
236        Ok(GenerateResult {
237            output_path: output_path.to_string_lossy().to_string(),
238            prompt_used: prompt,
239            provider: "flux".to_string(),
240            duration_secs: duration,
241            content_hash: hash,
242            metadata,
243        })
244    }
245
246    fn submit_job(
247        &self,
248        request: &GenerateRequest,
249        style: Option<&StyleGuide>,
250    ) -> Result<GenerationJob> {
251        // Flux is fast enough to be synchronous, but we support async for consistency
252        let mut job = GenerationJob::new("flux", &request.name);
253        job.prompt = Some(self.build_prompt(request, style));
254        Ok(job)
255    }
256
257    fn poll_job(&self, _job: &GenerationJob) -> Result<JobPollResult> {
258        // Flux completes synchronously
259        Ok(JobPollResult::Complete)
260    }
261
262    fn download_result(&self, job: &GenerationJob, output_dir: &Path) -> Result<GenerateResult> {
263        // For Flux, we generate synchronously via generate()
264        let request = GenerateRequest {
265            name: job.asset_name.clone(),
266            description: job.prompt.clone().unwrap_or_default(),
267            kind: AssetKind::Texture,
268            texture_params: None,
269            model_params: None,
270            audio_params: None,
271            tags: vec![],
272        };
273        self.generate(&request, None, output_dir)
274    }
275
276    fn build_prompt(&self, request: &GenerateRequest, style: Option<&StyleGuide>) -> String {
277        match style {
278            Some(s) => s.enrich_prompt(&request.description),
279            None => request.description.clone(),
280        }
281    }
282}

284/// Parse a Flux API response for testing
285pub fn parse_flux_response(json: &str) -> Result<String> {
286    let response: serde_json::Value = serde_json::from_str(json)
287        .map_err(|e| FlintError::GenerationError(format!("Invalid JSON: {}", e)))?;
288
289    response
290        .get("images")
291        .and_then(|imgs| imgs.as_array())
292        .and_then(|arr| arr.first())
293        .and_then(|img| img.get("url"))
294        .and_then(|u| u.as_str())
295        .map(|s| s.to_string())
296        .ok_or_else(|| FlintError::GenerationError("No image URL in response".to_string()))
297}
