1//! Style guide system for enriching generation prompts
2//!
3//! Style guides define a visual vocabulary (palette, materials, geometry constraints)
4//! that providers use to enrich prompts for consistent asset generation.
5
6use flint_core::{FlintError, Result};
7use serde::{Deserialize, Serialize};
8use std::path::Path;
9
10/// A style guide that enriches generation prompts
11#[derive(Debug, Clone, Serialize, Deserialize)]
12pub struct StyleGuide {
13    /// Style name (e.g., "medieval_tavern")
14    pub name: String,
15    /// Human-readable description
16    #[serde(default)]
17    pub description: Option<String>,
18    /// Prompt prefix prepended to all generation prompts
19    #[serde(default)]
20    pub prompt_prefix: Option<String>,
21    /// Prompt suffix appended to all generation prompts
22    #[serde(default)]
23    pub prompt_suffix: Option<String>,
24    /// Negative prompt (things to avoid)
25    #[serde(default)]
26    pub negative_prompt: Option<String>,
27    /// Color palette as hex strings
28    #[serde(default)]
29    pub palette: Vec<String>,
30    /// Material constraints
31    #[serde(default)]
32    pub materials: MaterialConstraints,
33    /// Geometry constraints
34    #[serde(default)]
35    pub geometry: GeometryConstraints,
36}
37
38/// Constraints on material properties
39#[derive(Debug, Clone, Default, Serialize, Deserialize)]
40pub struct MaterialConstraints {
41    /// Roughness range [min, max]
42    #[serde(default)]
43    pub roughness_range: Option<[f64; 2]>,
44    /// Metallic range [min, max]
45    #[serde(default)]
46    pub metallic_range: Option<[f64; 2]>,
47    /// Preferred material descriptors
48    #[serde(default)]
49    pub preferred_materials: Vec<String>,
50}
51
52/// Constraints on geometry
53#[derive(Debug, Clone, Default, Serialize, Deserialize)]
54pub struct GeometryConstraints {
55    /// Maximum triangle count for meshes
56    #[serde(default)]
57    pub max_triangles: Option<u32>,
58    /// Required features
59    #[serde(default)]
60    pub require_uvs: Option<bool>,
61    #[serde(default)]
62    pub require_normals: Option<bool>,
63}
64
65/// TOML file wrapper
66#[derive(Debug, Deserialize)]
67struct StyleFile {
68    style: StyleGuide,
69}
70
71impl StyleGuide {
72    /// Load a style guide from a TOML file
73    pub fn load(path: &Path) -> Result<Self> {
74        let content = std::fs::read_to_string(path)?;
75        let file: StyleFile = toml::from_str(&content).map_err(|e| {
76            FlintError::GenerationError(format!(
77                "Failed to parse style guide {}: {}",
78                path.display(),
79                e
80            ))
81        })?;
82        Ok(file.style)
83    }
84
85    /// Find and load a style guide by name, searching standard locations
86    pub fn find(name: &str) -> Result<Self> {
87        let candidates = [
88            format!("styles/{}.style.toml", name),
89            format!(".flint/styles/{}.style.toml", name),
90        ];
91
92        for candidate in &candidates {
93            let path = Path::new(candidate);
94            if path.exists() {
95                return Self::load(path);
96            }
97        }
98
99        Err(FlintError::GenerationError(format!(
100            "Style guide '{}' not found (searched: {})",
101            name,
102            candidates.join(", ")
103        )))
104    }
105
106    /// Enrich a prompt with style guide context
107    pub fn enrich_prompt(&self, base_prompt: &str) -> String {
108        let mut parts = Vec::new();
109
110        if let Some(ref prefix) = self.prompt_prefix {
111            parts.push(prefix.clone());
112        }
113
114        parts.push(base_prompt.to_string());
115
116        if !self.palette.is_empty() {
117            parts.push(format!("Color palette: {}", self.palette.join(", ")));
118        }
119
120        if !self.materials.preferred_materials.is_empty() {
121            parts.push(format!(
122                "Materials: {}",
123                self.materials.preferred_materials.join(", ")
124            ));
125        }
126
127        if let Some(ref suffix) = self.prompt_suffix {
128            parts.push(suffix.clone());
129        }
130
131        parts.join(". ")
132    }
133
134    /// Get the negative prompt (if any)
135    pub fn negative(&self) -> Option<&str> {
136        self.negative_prompt.as_deref()
137    }
138}

145    fn temp_style(content: &str) -> std::path::PathBuf {
146        let dir =
147            std::env::temp_dir().join(format!("flint_style_test_{}", uuid::Uuid::new_v4()));
148        std::fs::create_dir_all(&dir).unwrap();
149        let path = dir.join("test.style.toml");
150        let mut f = std::fs::File::create(&path).unwrap();
151        f.write_all(content.as_bytes()).unwrap();
152        path
153    }

155    #[test]
156    fn test_load_style_guide() {
157        let style_str = r##"
158[style]
159name = "medieval_tavern"
160description = "Weathered medieval fantasy tavern aesthetic"
161prompt_prefix = "Medieval fantasy tavern style"
162prompt_suffix = "Photorealistic, detailed textures, warm lighting"
163negative_prompt = "modern, sci-fi, neon, plastic"
164palette = ["#8B4513", "#A0522D", "#4A4A4A", "#D4A574", "#2F1B0E"]
165
166[style.materials]
167roughness_range = [0.6, 0.95]
168metallic_range = [0.0, 0.15]
169preferred_materials = ["aged wood", "rough stone", "hammered iron", "worn leather"]
170
171[style.geometry]
172max_triangles = 5000
173require_uvs = true
174require_normals = true
175"##;
176        let path = temp_style(style_str);
177        let style = StyleGuide::load(&path).unwrap();
178
179        assert_eq!(style.name, "medieval_tavern");
180        assert_eq!(style.palette.len(), 5);
181        assert_eq!(style.materials.roughness_range, Some([0.6, 0.95]));
182        assert_eq!(style.geometry.max_triangles, Some(5000));
183
184        std::fs::remove_file(&path).ok();
185        std::fs::remove_dir(path.parent().unwrap()).ok();
186    }

190        let style = StyleGuide {
191            name: "test".to_string(),
192            description: None,
193            prompt_prefix: Some("Fantasy style".to_string()),
