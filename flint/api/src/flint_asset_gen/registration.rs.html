3use crate::provider::{AssetKind, GenerateRequest, GenerateResult};

10const DEFAULT_STORE_ROOT: &str = ".flint/assets";
11const DEFAULT_ASSETS_ROOT: &str = "assets";
12
13#[derive(Debug, Clone)]
14pub struct RegisteredAsset {
15    pub hash: String,
16    pub sidecar_path: PathBuf,
17    pub meta: AssetMeta,
18}
19
20/// Register a generated asset using default project paths:
21/// - content store: `.flint/assets`
22/// - sidecars: `assets/<kind>/*.asset.toml`
23pub fn register_generated_asset(
24    request: &GenerateRequest,
25    result: &GenerateResult,
26) -> Result<RegisteredAsset> {
27    register_generated_asset_with_roots(
28        request,
29        result,
30        Path::new(DEFAULT_STORE_ROOT),
31        Path::new(DEFAULT_ASSETS_ROOT),
32    )
33}
34
35/// Register a generated asset into a specific content store and sidecar root.
36pub fn register_generated_asset_with_roots(
37    request: &GenerateRequest,
38    result: &GenerateResult,
39    store_root: &Path,
40    assets_root: &Path,
41) -> Result<RegisteredAsset> {
42    let output_path = Path::new(&result.output_path);
43    if !output_path.exists() {
44        return Err(FlintError::GenerationError(format!(
45            "Generated file not found: {}",
46            output_path.display()
47        )));
48    }
49
50    let store = ContentStore::new(store_root);
51    let hash = store.store(output_path)?;
52    let hash_str = hash.to_prefixed_hex();
53
54    let meta = AssetMeta {
55        name: request.name.clone(),
56        asset_type: asset_type_for_kind(request.kind),
57        hash: hash_str.clone(),
58        source_path: Some(result.output_path.clone()),
59        format: output_path
60            .extension()
61            .and_then(|e| e.to_str())
62            .map(|s| s.to_string()),
63        properties: build_generation_properties(result),
64        tags: request.tags.clone(),
65    };
66
67    let sidecar_path = write_asset_sidecar_with_root(&meta, assets_root)?;
68
69    Ok(RegisteredAsset {
70        hash: hash_str,
71        sidecar_path,
72        meta,
73    })
74}

76/// Write a sidecar file using default `assets/` root.
77pub fn write_asset_sidecar(meta: &AssetMeta) -> Result<PathBuf> {
78    write_asset_sidecar_with_root(meta, Path::new(DEFAULT_ASSETS_ROOT))
79}
80
81/// Write a sidecar file under a specific assets root directory.
82pub fn write_asset_sidecar_with_root(meta: &AssetMeta, assets_root: &Path) -> Result<PathBuf> {
83    #[derive(Serialize)]
84    struct Sidecar<'a> {
85        asset: &'a AssetMeta,
86    }
87
88    let subdir = asset_subdirectory(meta.asset_type);
89    let sidecar_dir = assets_root.join(subdir);
90    std::fs::create_dir_all(&sidecar_dir)?;
91
92    let sidecar_path = sidecar_dir.join(format!("{}.asset.toml", meta.name));
93    let toml_str = toml::to_string_pretty(&Sidecar { asset: meta })?;
94    std::fs::write(&sidecar_path, toml_str)?;
95    Ok(sidecar_path)
96}

98fn build_generation_properties(result: &GenerateResult) -> HashMap<String, toml::Value> {
99    let mut props = HashMap::new();
100    props.insert(
101        "prompt".to_string(),
102        toml::Value::String(result.prompt_used.clone()),
103    );
104    props.insert(
105        "provider".to_string(),
106        toml::Value::String(result.provider.clone()),
107    );
108    for (k, v) in &result.metadata {
109        props.insert(k.clone(), toml::Value::String(v.clone()));
110    }
111    props
112}
113
114fn asset_type_for_kind(kind: AssetKind) -> AssetType {
115    match kind {
116        AssetKind::Texture => AssetType::Texture,
117        AssetKind::Model => AssetType::Mesh,
118        AssetKind::Audio => AssetType::Audio,
119    }
120}
121
122fn asset_subdirectory(asset_type: AssetType) -> &'static str {
123    match asset_type {
124        AssetType::Mesh => "meshes",
125        AssetType::Texture => "textures",
126        AssetType::Material => "materials",
127        AssetType::Audio => "audio",
128        AssetType::Script => "scripts",
129    }
130}

171    #[test]
172    fn test_register_generated_asset_with_roots() {
173        let dir = temp_dir();
174        let store_root = dir.join("store");
175        let assets_root = dir.join("assets");
176        let out_dir = dir.join("out");
177        std::fs::create_dir_all(&out_dir).unwrap();
178
179        let output_path = out_dir.join("brick wall.png");
180        let mut f = std::fs::File::create(&output_path).unwrap();
181        f.write_all(b"fake-image").unwrap();
182
183        let request = sample_request(AssetKind::Texture, "brick_wall");
184        let result = GenerateResult {
185            output_path: output_path.to_string_lossy().to_string(),
186            prompt_used: "weathered \"brick\" wall".to_string(),
187            provider: "mock".to_string(),
188            duration_secs: 0.1,
189            content_hash: None,
190            metadata: HashMap::from([("seed".to_string(), "42".to_string())]),
191        };
192
193        let registered = register_generated_asset_with_roots(
194            &request,
195            &result,
196            &store_root,
197            &assets_root,
198        )
199        .unwrap();
200
201        assert!(registered.hash.starts_with("sha256:"));
202        assert!(registered.sidecar_path.exists());
203        assert_eq!(registered.meta.name, "brick_wall");
204        assert_eq!(registered.meta.asset_type, AssetType::Texture);
205
206        let sidecar = std::fs::read_to_string(&registered.sidecar_path).unwrap();
207        let parsed: toml::Value = toml::from_str(&sidecar).unwrap();
208        assert_eq!(
209            parsed
210                .get("asset")
211                .and_then(|a| a.get("name"))
212                .and_then(|v| v.as_str()),
213            Some("brick_wall")
214        );
215        assert_eq!(
216            parsed
217                .get("asset")
218                .and_then(|a| a.get("properties"))
219                .and_then(|p| p.get("prompt"))
220                .and_then(|v| v.as_str()),
221            Some("weathered \"brick\" wall")
222        );
223
224        std::fs::remove_dir_all(dir).ok();
225    }
